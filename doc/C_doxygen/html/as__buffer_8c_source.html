200  as_hardware_address_t start = (as_hardware_address_t)buf->buffer_baseaddr_phys;
201  as_hardware_address_t end = (as_hardware_address_t)buf->buffer_baseaddr_phys + buf->transfer_size;
202 
203  /* Get current operating address of HW module to update HW address of as_memio */
204  as_hardware_address_t temp_addr = as_reader_writer_get_cur_hw_addr(module_address);
205  if (temp_addr == 0) {
206  temp_addr = buf->buffer_baseaddr_phys;
207  }
208 
209  // If outside of bounds or not busy, transfer is finished
210  if (temp_addr >= start && temp_addr <= end && as_reader_writer_is_busy(module_address)) {
211  // AS_INFO("Incomplete read\n");
212  } else {
213  AS_INFO("Buffer complete (address is outside of buffer): %x\n", temp_addr);
214  AS_INFO("Unmap buffers at: %x, %p\n", buf->dma_addr, buf->buffer_baseaddr_virt);
215  as_unmap_single(buf->dma_addr, buf->buffer_size, buf->direction, manage_cache);
216  if ((buf->direction == AS_BUFFER_DIR_FROM_DEV) && supports_data_unit) {
217  buf->transfer_size = start - (as_hardware_address_t)as_writer_get_last_data_unit_complete_addr(module_address);
218  }
219  buf->dma_addr = 0;

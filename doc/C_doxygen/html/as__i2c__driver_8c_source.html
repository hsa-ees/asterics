40 int as_driver_i2c_probe(struct platform_device *pdev) {
41  int ret = 0;
42  struct asterics_i2c *priv_data = NULL;
43 
44  // Create driver data and set it for this platform device, lifetime managed by kernel
45  priv_data = devm_kzalloc(&pdev->dev, sizeof(*priv_data), GFP_KERNEL);
46  if (!priv_data)
47  return -ENOMEM;
48 
49  platform_set_drvdata(pdev, priv_data);
50 
51  // Initialize adapter
52  priv_data->adapter.owner = THIS_MODULE;
53 
54  priv_data->adapter.algo = &asterics_i2c_algo;
55  priv_data->adapter.algo_data = priv_data;
56  priv_data->adapter.retries = 5;
57  priv_data->adapter.timeout = msecs_to_jiffies(1000);
58  priv_data->adapter.dev.of_node = pdev->dev.of_node;
59  priv_data->adapter.dev.parent = &pdev->dev;
60  strlcpy(priv_data->adapter.name, "ASTERICS I2C", sizeof(priv_data->adapter.name));
61 
62  // TODO: Use clock to initialize clock speed
63  priv_data->clk = devm_clk_get(&pdev->dev, NULL);
64 
65  ret = i2c_add_adapter(&priv_data->adapter);
66  return ret;
67 }
68 
69 int as_driver_i2c_remove(struct platform_device *pdev) {
70  struct asterics_i2c *priv_data = platform_get_drvdata(pdev);
71 
72  i2c_del_adapter(&priv_data->adapter);
73 
74  return 0;

208 uint8_t as_iic_init(as_hardware_address_t base_addr, uint32_t iic_bus_freq) {
209  as_iic_reset(base_addr);
210 
211  // check if a valid frequency is entered
212  // note that valid is not always possible. This depends on the hardware implementation
213  if (iic_bus_freq < 10000 || iic_bus_freq > 1000000) {
214  return AS_IIC_ERR_FREQ_INVALID;
215  }
216 
217  // wait for the hardware to set the iic_ready bit
218  if (as_iic_busy_wait_for_hwready(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_SHORT_TIMEOUT) == AS_FALSE) {
219  // if it takes too long, return with an error
220  return AS_IIC_ERR_HW_NOT_READY;
221  }
222  // calculates and sets the contents of the scl-div register.
223  // SCL_DIV = (System Freq / 4 * Desired IIC Bus Freq) - 2.
224  as_iic_set_scl_div(base_addr, (uint32_t)((AS_SYSTEM_CLOCK_HZ / (4 * iic_bus_freq)) - 2));
225 
226  // set the initialized bit.
227  as_iic_set_initialized(1);
228 
229  return AS_IIC_OK;
230 }

234 uint8_t as_iic_start_transaction(as_hardware_address_t base_addr, uint8_t *slave_addr, uint8_t *data, uint8_t readwrite, uint8_t modifier) {
235  // if the bus frequency hasn't been set yet, return an error
236  if (as_iic_is_initialized() != AS_TRUE) {
237  return AS_ERR_NOT_INITIALIZED;
238  }
239 
240  // wait for the hardware to set the iic_ready bit
241  if (as_iic_busy_wait_for_hwready(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_SHORT_TIMEOUT) == AS_FALSE) {
242  // if it takes too long, set the reset bit
243  as_iic_reset_hw_state(base_addr);
244  if (as_iic_is_ready(base_addr) == AS_FALSE) {
245  // if it's still not ready, return an error. Something's seriously wrong.
246  return AS_IIC_ERR_HW_NOT_READY;
247  }
248  }
249 
250  // make sure the address is in the correct form:
251  if (readwrite) { // read address:
252  (*slave_addr) |= 0x01; // add one to the address, if it isn't already there.
253  } else { // write address:
254  (*slave_addr) &= 0xFE; // force a 0 for the last bit, if it isn't already there.
255  }
256 
257  // the hardware is ready. Copy the slave address to the tx data register
258  as_iic_set_data_tx(base_addr, slave_addr);
259 
260  // start the transaction
261  as_iic_set_control(base_addr, AS_IIC_TSTART);
262 
263  // wait for the hardware to send the address. If it takes too long, reset and return an error.
264  if (as_iic_busy_wait_for_wait(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_LONG_TIMEOUT) == AS_FALSE) {
265  as_iic_reset_hw_state(base_addr);
266  return AS_IIC_ERR_HW_TO_SENDADR;
267  }
268  // do we only need to send the address?
269  if (modifier & AS_IIC_MOD_ONLY_ADDRESS) {
270  // yes -> stop the transaction.
271  as_iic_stop_write_transaction(base_addr);
272  } else { // no -> check what transaction is next:
273  if (readwrite) { // read ...
274  if (modifier & AS_IIC_MOD_MASTER_ACK) { // ... with master acknowledge.
275  as_iic_set_control(base_addr, AS_IIC_CONT_READ | AS_IIC_ACK_MODIFIER);
276  } else { // ... without master acknowledge.
277  as_iic_set_control(base_addr, AS_IIC_CONT_READ);
278  }
279  } else { // write ...
280  // (set the byte to write)
281  as_iic_set_data_tx(base_addr, data);
282 
283  if (modifier & AS_IIC_MOD_MASTER_ACK) { // ... with master acknowledge.
284  as_iic_set_control(base_addr, AS_IIC_CONT_WRITE | AS_IIC_ACK_MODIFIER);
285  } else { // ... without master acknowledge.
286  as_iic_set_control(base_addr, AS_IIC_CONT_WRITE);
287  }
288  }
289 
290  // wait for the hardware to send/receive the next byte.
291  // If it takes too long, reset and return an error.
292  if (as_iic_busy_wait_for_wait(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_LONG_TIMEOUT) == AS_FALSE) {
293  as_iic_reset_hw_state(base_addr);
294  return AS_IIC_ERR_HW_TO_SENDADR;
295  }
296  // check if the slave acknowledged the address.
297  // If not, stop the transaction and return an error.
298  if (as_iic_ack_received(base_addr) == AS_FALSE && !(modifier & (AS_IIC_MOD_MASTER_ACK | AS_IIC_MOD_IGNORE_ACKNOWLEDGE))) {
299  as_iic_set_control(base_addr, AS_IIC_STOP_TRANS);
300  return AS_IIC_ERR_ADR_NACK;
301  }
302  }
303  // everything seems to have worked. Great!
304  return AS_IIC_OK;
305 }

307 uint8_t as_iic_stop_read_transaction(as_hardware_address_t base_addr, uint8_t *read_data) {
308  // get the received data byte
309  *read_data = as_iic_get_rx_byte(base_addr);
310 
311  // stop the transaction.
312  as_iic_set_control(base_addr, AS_IIC_STOP_TRANS);
313 
314  // wait for the hardware to enter the "ready" state again
315  if (as_iic_busy_wait_for_hwready(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_SHORT_TIMEOUT) == AS_FALSE) {
316  as_iic_reset_hw_state(base_addr);
317  return AS_IIC_ERR_HW_TO_GENERAL;
318  }
319 
320  // wait for a bit. When stringing transaction together
321  // the slave needs some time between transactions
322  as_sleep(50000);
323 
324  return AS_IIC_OK;
325 }

327 uint8_t as_iic_stop_write_transaction(as_hardware_address_t base_addr) {
328  // stop the transaction.
329  as_iic_set_control(base_addr, AS_IIC_STOP_TRANS);
330 
331  // wait for the hardware to enter the "ready" state again
332  if (as_iic_busy_wait_for_hwready(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_SHORT_TIMEOUT) == AS_FALSE) {
333  as_iic_reset_hw_state(base_addr);
334  return AS_IIC_ERR_HW_TO_GENERAL;
335  }
336 
337  // check if the slave acknowledged the data. If not, return an error.
338  if (as_iic_ack_received(base_addr) == AS_FALSE) {
339  return AS_IIC_ERR_DATA_NACK;
340  }
341 
342  // wait for a bit. When stringing transaction together
343  // the slave needs some time between transactions
344  as_sleep(50000);
345 
346  return AS_IIC_OK;
347 }

349 uint8_t as_iic_write_transaction(as_hardware_address_t base_addr, uint8_t *byte, uint8_t modifier) {
350  // check whether the previous transaction was read or write.
351  AS_BOOL prev_rw = as_iic_is_readwrite(base_addr);
352 
353  // copy the next byte into the tx data register.
354  as_iic_set_data_tx(base_addr, byte);
355 
356  if (!(modifier & AS_IIC_MOD_IGNORE_ACKNOWLEDGE) && !prev_rw) {
357  // check if the slave acknowledged the last byte.
358  // If not, stop the transaction and return an error.
359  if (as_iic_ack_received(base_addr) == AS_FALSE) {
360  as_iic_set_control(base_addr, AS_IIC_STOP_TRANS);
361  return AS_IIC_ERR_DATA_NACK;
362  }
363  }
364  // start the next transaction ...
365  if (modifier & AS_IIC_MOD_MASTER_ACK) { // ... with master acknowledge.
366  as_iic_set_control(base_addr, AS_IIC_CONT_WRITE | AS_IIC_ACK_MODIFIER);
367  } else { // ... without master acknowledge.
368  as_iic_set_control(base_addr, AS_IIC_CONT_WRITE);
369  }
370 
371  // wait for the hardware to send the byte. If it takes too long, reset and return an error.
372  if (as_iic_busy_wait_for_wait(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_LONG_TIMEOUT) == AS_FALSE) {
373  as_iic_reset_hw_state(base_addr);
374  return AS_IIC_ERR_HW_TO_GENERAL;
375  }
376  return AS_IIC_OK;
377 }

379 uint8_t as_iic_read_transaction(as_hardware_address_t base_addr, uint8_t *read_data, uint8_t modifier) {
380  // check whether the previous transaction was read or write.
381  AS_BOOL prev_rw = as_iic_is_readwrite(base_addr);
382 
383  // read the byte from the rx data register.
384  *read_data = as_iic_get_rx_byte(base_addr);
385 
386  // if the previous transaction was a write transaction and acknowledges are not ignored ...
387  if (!(modifier & AS_IIC_MOD_IGNORE_ACKNOWLEDGE) && !prev_rw) {
388  // ... check if the slave acknowledged the last byte.
389  // If not, stop the transaction and return an error.
390  if (as_iic_ack_received(base_addr) == AS_FALSE) {
391  as_iic_set_control(base_addr, AS_IIC_STOP_TRANS);
392  return AS_IIC_ERR_DATA_NACK;
393  }
394  }
395 
396  // Continue with the read transaction
397  as_iic_set_control(base_addr, AS_IIC_CONT_READ);
398 
399  // wait for the hardware to receive the byte. If it takes too long, reset and return an error.
400  if (as_iic_busy_wait_for_wait(base_addr, AS_IIC_TIMEOUT_BASE * AS_IIC_LONG_TIMEOUT) == AS_FALSE) {
401  as_iic_reset_hw_state(base_addr);
402  return AS_IIC_ERR_HW_TO_GENERAL;
403  }
404  return AS_IIC_OK;
405 }

419 uint8_t as_iic_get_byte_mod(as_hardware_address_t base_addr, uint8_t slave_addr, uint8_t *read_data, uint8_t modifier) {
420  uint8_t report;

439 uint8_t as_iic_get_bytes(as_hardware_address_t base_addr, uint8_t slave_addr, uint16_t byte_count, uint8_t *read_data) {
440  uint8_t report = 0;
441  uint16_t count = 0;
442  byte_count--;
443  // start a read transaction (send the address).
444  report = as_iic_start_transaction(base_addr, &slave_addr, 0, AS_IIC_READ, AS_IIC_MOD_NONE);
445  if (report != AS_IIC_OK) return report;
446  // for every data byte:
447  for (; count < byte_count; count++) {
448  // let the slave write the data byte to the bus.
449  report = as_iic_read_transaction(base_addr, &(read_data[count]), AS_IIC_MOD_NONE);
450  if (report != AS_IIC_OK) return report;
451  }
452  // finish the transaction and receive the last byte.
453  return as_iic_stop_read_transaction(base_addr, &(read_data[byte_count]));
454 }

475 uint8_t as_iic_read_reg(as_hardware_address_t base_addr, uint8_t slave_addr, uint8_t *reg, uint8_t *read_data) {
476  uint8_t report = 0;
477  // write the register address to the slave.
478  report = as_iic_write_byte(base_addr, slave_addr, reg);
479  if (report != AS_IIC_OK) return report;

482  // have the slave send the byte of the register address.

495 uint8_t as_iic_read_regs(as_hardware_address_t base_addr, uint8_t slave_addr, uint8_t *reg, uint16_t byte_count, uint8_t *read_data) {
496  uint8_t report = 0;
497  uint16_t count = 0;
498  byte_count--;
499  // send the starting register to the slave.
500  as_iic_write_byte(base_addr, slave_addr, reg);
501  // wait a bit, so the slave has enough time to recognize the next transaction.
502  as_sleep(50000);
503  // start a read transaction (send the address).
504  report = as_iic_start_transaction(base_addr, &slave_addr, 0, AS_IIC_READ, AS_IIC_MOD_NONE);
505  if (report != AS_IIC_OK) return report;
506  // for every data byte:
507  for (; count < byte_count; count++) {
508  // let the slave write the data byte to the bus.
509  report = as_iic_read_transaction(base_addr, &(read_data[count]), AS_IIC_MOD_NONE);
510  if (report != AS_IIC_OK) return report;
511  }
512  // finish the transaction and receive the last byte.
513  return as_iic_stop_read_transaction(base_addr, &(read_data[byte_count]));
514 }

44 typedef struct {
45  char name[20]; /* Device name used for printing feedback if device creation was successful */
46  int type; /* One of the AS_DEV_TYPE */
47  as_hardware_address_t address; /* Hardware address of the memory module for DEV_TYPE_MEMIO, start address for DEV_TYPE_REGIO and DEV_TYPE_I2C */
48  uint32_t address_range_size; /* Number of bytes to be mapped for DEV_TYPE_REGIO and DEV_TYPE_I2C. Number of bytes to allocate for DEV_TYPE_MMAP */
49  uint32_t interface_width; /* Bit width of the bus interface of the memory module for the DEV_TYPE_MEMIO device */
50  uint8_t flags; /* O_RDWR, O_RDONLY or O_WRONLY; DEV_TYPE_MEMIO requires O_RDONLY for as_memwriter and O_WRONLY for as_memreader */
51  char manage_cache; /* If "AS_TRUE", the device driver has to handle cache coherency for the device (currently: as_memio only) */
52  char support_data_unit; /* If "AS_TRUE", the device is capable of providing the end address of the last transferred unit */
53 } device_info_t;

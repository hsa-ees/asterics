30 void as_memio_module_setup(as_memio_hw_settings_t *hw_settings) {
31  if ((*hw_settings).baseaddr != 0) {
32  /* HW module supports only 1 section for as_memio configuration */
33  as_reader_writer_set_section_count((*hw_settings).baseaddr, 1);
34  /* There is no section offset since there is only one section */
35  as_reader_writer_set_section_offset((*hw_settings).baseaddr, 0);
36  /* Set maximum possible burst length of HW module */
37  as_reader_writer_set_max_burst_length((*hw_settings).baseaddr, (*hw_settings).burst_length);
38 
39  /* Bring HW module to defined state */
40  as_reader_writer_reset((*hw_settings).baseaddr);
41 
42  /* Enable the memwriter module to receive data */
43  if ((*hw_settings).read_not_write == AS_TRUE) {
44  as_writer_set_enable((*hw_settings).baseaddr);
45  }
46  }
47 }

93 as_memio_file_t *as_memio_open(as_hardware_address_t as_reader_writer_baseaddr, as_memio_config_t *memio_config, uint8_t flags) {
94  as_memio_config_t memio_config_init;
95  as_memio_file_t *fd = NULL;
96  // Hardcode 1280*960 for now
97  as_buffer_config_t config;
98  config.size = 0x12c000;
99  config.type = AS_BUFFER_KERN_MEM;
100 
101  if (memio_config == NULL) {
102  as_memio_config_init(&memio_config_init);
103  } else {
104  memio_config_init = *memio_config;
105  }
106 
107  print_memio_config(&memio_config_init);
108 
109  /* Create new structure for as_memio operations */
110  fd = (as_memio_file_t *)as_malloc(sizeof(as_memio_file_t));
111  if (fd == NULL) {
112  return NULL;
113  }
114 
115 
116  fd->memio_buffer_handler.available_buffers = 0;
117 
118  /* Allocate memory for as_memio buffer */
119  fd->memio_buffer_handler.buffers[0] = as_buffer_obj_create(&config);
120  if (fd->memio_buffer_handler.buffers[0] == NULL) {
121  AS_ERROR("Failed to allocate buffer 1\n");
122  as_free(fd);
123  return NULL;
124  }
125  fd->memio_buffer_handler.buffers[1] = as_buffer_obj_create(&config);
126  if (fd->memio_buffer_handler.buffers[1] == NULL) {
127  AS_ERROR("Failed to allocate buffer 2\n");
128  as_buffer_obj_destroy(fd->memio_buffer_handler.buffers[0]);
129  as_free(fd);
130  return NULL;
131  }
132  fd->memio_buffer_handler.available_buffers = 2;
133  as_buffer_queue_init(&fd->memio_buffer_handler.queue, 2);
134 
135  /* Memio config assignments */
136  /* Set address of associated hardware module */
137  fd->memio_hw_settings.baseaddr = as_reader_writer_baseaddr;
138  /* Set burst length for hardware module */
139  fd->memio_hw_settings.burst_length = (memio_config_init).as_reader_writer_max_burst_length;
140  /* Set word size of associated hardware module */
141  fd->memio_hw_settings.wordsize = (memio_config_init).as_reader_writer_interface_width / 8;
142  /* Initialize as_memio for read operations if at least one of the following flags were provided */
143  if ((flags & O_ACCMODE) == O_RDONLY) {
144  fd->memio_hw_settings.read_not_write = AS_TRUE;
145  }
146  /* Initialize as_memio for write operations if at least one of the following flags were provided */
147  else if ((flags & O_ACCMODE) == O_WRONLY) {
148  fd->memio_hw_settings.read_not_write = AS_FALSE;
149  } else {
150  as_buffer_obj_destroy(fd->memio_buffer_handler.buffers[0]);
151  as_buffer_obj_destroy(fd->memio_buffer_handler.buffers[1]);
152  as_free(fd);
153  return NULL;
154  }
155  fd->memio_hw_settings.manage_cache = (memio_config_init).manage_cache;
156 
157  print_buffer_handler(&fd->memio_buffer_handler);
158  print_hw_settings(&fd->memio_hw_settings);
159  /* Setup hardware modules for as_memio usage */
160  as_memio_module_setup(&(fd->memio_hw_settings));
161 
162  /* Return the as_memio structure to caller for as_memio operations */
163  return fd;
164 }

166 void as_memio_hw_write_update(as_memio_file_t *fd) {
167  as_buffer_obj_t *buf = NULL;
168 
169  //AS_INFO("Call to hw write update\n");
170  /* check if next go can already be accepted by hardware */
171  if (as_reader_writer_is_pending_go(fd->memio_hw_settings.baseaddr) == AS_FALSE) {
172  buf = as_buffer_find_waiting_buffer(fd->memio_buffer_handler.buffers, fd->memio_buffer_handler.available_buffers);
173  if (buf != NULL) {
174  buf->dma_addr = as_map_single(buf->buffer_baseaddr_virt, buf->buffer_size, AS_BUFFER_DIR_TO_DEV, fd->memio_hw_settings.manage_cache);
175  /* Set start address for next write operation */
176  #if AS_OS_LINUX_KERNEL
177  AS_INFO("Set MEMREADER at %x to address %x, with size: %x", fd->memio_hw_settings.baseaddr,
178  (unsigned int)virt_to_phys((volatile void *)buf->buffer_baseaddr_virt), buf->transfer_size);
179  //AS_INFO("Buffer is %p\n", buf);
180  as_reader_writer_set_section_addr(fd->memio_hw_settings.baseaddr, virt_to_phys((volatile void *)buf->buffer_baseaddr_virt));
181  #else /* AS_OS_NONE */
182  as_reader_writer_set_section_addr(fd->memio_hw_settings.baseaddr, fd->memio_buffer_handler.cur_hw_start_addr);
183  #endif
184  /* Set blocksize in byte! (Required by HW module) */
185  as_reader_writer_set_section_size(fd->memio_hw_settings.baseaddr, buf->transfer_size);
186  /* Start as_memreader */
187  as_reader_writer_set_go(fd->memio_hw_settings.baseaddr);
188  buf->state = AS_BUFFER_STATE_OWN_DEVICE;
189  as_buffer_enqueue(&fd->memio_buffer_handler.queue, buf);
190  }
191  }
192 
193  // Check if there is a buffer that is ready to be transferred back to CPU
194  buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
195  //AS_INFO("Buffer %p\n", buf);
196  if (buf != NULL && buf->state == AS_BUFFER_STATE_OWN_DEVICE) {
197  as_hardware_address_t start = (as_hardware_address_t)buf->buffer_baseaddr_phys;
198  as_hardware_address_t end = (as_hardware_address_t)buf->buffer_baseaddr_phys + buf->transfer_size;
199 
200  /* Get current operating address of HW module to update HW address of as_memio */
201  as_hardware_address_t temp_addr = as_reader_writer_get_cur_hw_addr(fd->memio_hw_settings.baseaddr);
202  if (temp_addr == 0) {
203  temp_addr = fd->memio_buffer_handler.buffers[0]->buffer_baseaddr_phys;
204  }
205 
206  // If outside of bounds or not busy, transfer is finished
207  if (temp_addr >= start && temp_addr < end && as_reader_writer_is_busy(fd->memio_hw_settings.baseaddr)) {
208  AS_INFO("Incomplete write\n");
209  } else {
210  AS_INFO("Buffer write complete (address is outside of buffer): %x\n", temp_addr);
211  AS_INFO("Unmap buffers at: %x, %p\n", buf->dma_addr, buf->buffer_baseaddr_virt);
212  as_unmap_single(buf->dma_addr, buf->buffer_size, AS_BUFFER_DIR_TO_DEV, fd->memio_hw_settings.manage_cache);
213  buf->dma_addr = 0;
214  buf->state = AS_BUFFER_STATE_OWN_CPU;
215  }
216  }

225 void as_memio_hw_read_update(as_memio_file_t *fd) {
226  as_buffer_obj_t *buf = NULL;
227 
228  /* Check if a new go signal can already be set */
229  if (as_reader_writer_is_pending_go(fd->memio_hw_settings.baseaddr) == AS_FALSE) {
230  buf = as_buffer_find_waiting_buffer(fd->memio_buffer_handler.buffers, fd->memio_buffer_handler.available_buffers);
231  if (buf != NULL) {
232  if (buf->buffer_size < buf->transfer_size) {
233  //AS_ERROR("Tried to transfer more bytes than available\n");
234  buf->transfer_size = buf->buffer_size;
235  }
236  AS_INFO("Set MEMWRITER at %x to address %x, with size: %x", fd->memio_hw_settings.baseaddr,
237  (unsigned int)virt_to_phys((volatile void *)buf->buffer_baseaddr_virt), buf->buffer_size);
238 #if AS_OS_LINUX_KERNEL
239  trace_asterics_start_transfer(buf->buffer_baseaddr_phys, as_reader_writer_get_cur_hw_addr(fd->memio_hw_settings.baseaddr));
240  buf->dma_addr = as_map_single(buf->buffer_baseaddr_virt, buf->buffer_size, AS_BUFFER_DIR_FROM_DEV, fd->memio_hw_settings.manage_cache);
241  as_reader_writer_set_section_addr(fd->memio_hw_settings.baseaddr, virt_to_phys((volatile void *)buf->buffer_baseaddr_virt));
242 #else /* AS_OS_NONE */
243  if (fd->memio_hw_settings.manage_cache) {
244  as_dcache_invalidate_range(((volatile void *)fd->memio_buffer_handler.cur_hw_start_addr), buf->transfer_size);
245  }
246  as_reader_writer_set_section_addr(fd->memio_hw_settings.baseaddr, fd->memio_buffer_handler.cur_hw_start_addr);
247 #endif
248  /* Set blocksize in byte! (Required by HW module) */
249  as_reader_writer_set_section_size(fd->memio_hw_settings.baseaddr, buf->transfer_size);
250  /* Start as_memwriter */
251  as_reader_writer_set_go(fd->memio_hw_settings.baseaddr);
252  buf->state = AS_BUFFER_STATE_OWN_DEVICE;
253  AS_BOOL success = as_buffer_enqueue(&fd->memio_buffer_handler.queue, buf);
254  if (success == AS_FALSE) {
255  AS_ERROR("Unable to add more transfers to queue\n");
256  }
257  }
258  }
259 
260  // Check if there is a buffer that is ready to be transferred back to CPU
261  buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
262  if (buf != NULL && buf->state == AS_BUFFER_STATE_OWN_DEVICE) {
263  as_hardware_address_t start = (as_hardware_address_t)buf->buffer_baseaddr_phys;
264  as_hardware_address_t end = (as_hardware_address_t)buf->buffer_baseaddr_phys + buf->transfer_size;
265 
266  /* Get current operating address of HW module to update HW address of as_memio */
267  as_hardware_address_t temp_addr = as_reader_writer_get_cur_hw_addr(fd->memio_hw_settings.baseaddr);
268  if (temp_addr == 0) {
269  temp_addr = fd->memio_buffer_handler.buffers[0]->buffer_baseaddr_phys;
270  }
271 
272  // If outside of bounds or not busy, transfer is finished
273  if (temp_addr >= start && temp_addr < end && as_reader_writer_is_busy(fd->memio_hw_settings.baseaddr)) {
274  // AS_INFO("Incomplete read\n");
275  } else {
276  AS_INFO("Buffer complete (address is outside of buffer): %x for %x\n", temp_addr, buf->buffer_baseaddr_phys);
277  AS_INFO("Unmap buffers at: %x, %p\n", buf->dma_addr, buf->buffer_baseaddr_virt);
278  trace_asterics_finished_transfer(buf->buffer_baseaddr_phys, temp_addr);
279  as_unmap_single(buf->dma_addr, buf->buffer_size, AS_BUFFER_DIR_FROM_DEV, fd->memio_hw_settings.manage_cache);
280  buf->dma_addr = 0;
281  buf->state = AS_BUFFER_STATE_OWN_CPU;
282  }
283  }

335 ssize_t as_memio_read(as_memio_file_t *fd, void *buffer, uint32_t count) {
336  /* Temporary variables used for copying data */
337  uint8_t *dst = NULL;
338  uint8_t *src = NULL;
339 #if AS_OS_NONE
340  uint32_t n = 0;
341 #endif
342  uint32_t copied = 0;
343  // Is a buffer already waiting for it's transfer?
344  AS_BOOL alreadyOnQueue = AS_FALSE;
345 
346  if (as_writer_is_sync_error(fd->memio_hw_settings.baseaddr)) {
347  AS_ERROR("Tried to use MemWriter after sync error occured\n");
348  return -ENODEV;
349  }
350 
351  // AS_INFO("read: %x bytes\n", count);
352  as_buffer_obj_t *buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
353  if (buf != NULL) {
354  if (buf->buffer_baseaddr_user == buffer) {
355  alreadyOnQueue = AS_TRUE;
356  }
357  }
358 
359  // If this transfer isn't yet on queue, add it
360  if (alreadyOnQueue == AS_FALSE && count > 0) {
361  as_buffer_obj_t *wait_buf = as_buffer_find_inactive_buffer(fd->memio_buffer_handler.buffers, fd->memio_buffer_handler.available_buffers);
362  if (wait_buf != NULL) {
363  wait_buf->buffer_baseaddr_user = buffer;
364  wait_buf->transfer_size = count;
365  wait_buf->state = AS_BUFFER_STATE_WAITING;
366  } else {
367  return -EAGAIN;
368  }
369  }
370 
371  /* Start transfer and track progress */
372  as_memio_hw_read_update(fd);
373 
374  /* Check if a completed object is on queue */
375  buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
376 
377  if (buf != NULL) {
378  if (buf->state == AS_BUFFER_STATE_OWN_CPU) {
379  as_buffer_dequeue(&fd->memio_buffer_handler.queue);
380 
381  /* Copy buffer (from user) and SW address to temporary variables for copy
382  * process */
383  src = (uint8_t *)buf->buffer_baseaddr_virt;
384  dst = (uint8_t *)buffer;
385 
386  copied = as_copy_to_user(dst, src, buf->transfer_size);
387  AS_INFO("copied: %x\n", copied);
388  buf->state = AS_BUFFER_STATE_INACTIVE;
389  } else {
390  return -EAGAIN;
391  }
392  }
393 
394  return copied;
395 }

397 ssize_t as_memio_write(as_memio_file_t *fd, const void *buffer, uint32_t count) {
398  uint32_t copied = 0;
399  AS_BOOL alreadyOnQueue = AS_FALSE;
400  as_buffer_obj_t *buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
401  if (buf != NULL) {
402  if (buf->buffer_baseaddr_user == buffer) {
403  alreadyOnQueue = AS_TRUE;
404  }
405  }
406 
407  if (alreadyOnQueue == AS_FALSE && count > 0) {
408  as_buffer_obj_t *wait_buf = as_buffer_find_inactive_buffer(fd->memio_buffer_handler.buffers, fd->memio_buffer_handler.available_buffers);
409  if (wait_buf == NULL) {
410  AS_ERROR("Failed to find inactive buffer\n");
411  return -EAGAIN;
412  }
413 
414  wait_buf->buffer_baseaddr_user = buffer;
415 
416  // Copy as many bytes as possible
417  if (wait_buf->buffer_size < count) {
418  count = wait_buf->buffer_size;
419  }
420  wait_buf->transfer_size = as_copy_from_user(wait_buf->buffer_baseaddr_virt, buffer, count);
421  if (wait_buf->transfer_size != 0) {
422  wait_buf->state = AS_BUFFER_STATE_WAITING;
423  }
424  }
425 
426  // Initiate transfer and update state
427  as_memio_hw_write_update(fd);
428 
429  /* Check if a completed object is on queue */
430  buf = as_buffer_inspect(&fd->memio_buffer_handler.queue);
431 
432  if (buf != NULL) {
433  if (buf->state == AS_BUFFER_STATE_OWN_CPU) {
434  // Writing is done, dequeue object
435  as_buffer_dequeue(&fd->memio_buffer_handler.queue);
436 
437  // AS_INFO("copied (write): %x\n", buf->transfer_size);
438  copied = buf->transfer_size;
439  buf->state = AS_BUFFER_STATE_INACTIVE;
440  }
441  }
442 
443  return copied;
444 }

62 void write_iic_register(XIicPs *instancePtr, u8 chipAddr, u16 regAddr, u8 regVal) {
63  u8 writeData[3];
64  int status;
65 
66  writeData[0] = regAddr >> 8;
67  writeData[1] = regAddr & 0xFF;
68  writeData[2] = regVal;
69  status = XIicPs_MasterSendPolled(instancePtr, writeData, 3, chipAddr);
70 
71  /* Wait until bus is idle to start another transfer */
72  while (XIicPs_BusIsBusy(instancePtr)) {
73  /* NOP */
74  }
75 }

77 int read_iic_register(XIicPs *instancePtr, u8 chipAddr, u16 regAddr) {
78  u8 writeData[2];
79  u8 readData;
80 
81  writeData[0] = regAddr >> 8;
82  writeData[1] = regAddr & 0xFF;
83  XIicPs_MasterSendPolled(instancePtr, writeData, 2, chipAddr);
84  XIicPs_MasterRecvPolled(instancePtr, &readData, 1, chipAddr);
85 
86  /* Wait until bus is idle */
87  while (XIicPs_BusIsBusy(instancePtr)) {
88  /* NOP */
89  }
90 
91  return readData;
92 }

317 int init_iic(uint16_t device_id) {
318  int status;
319  int Index;
320  XIicPs_Config *Config;
321  XIicPs *instancePtr = &instanceIIC;
322 
323  /* Initialize the IIC driver so that it's ready to use
324  * Look up the configuration in the config table,
325  * then initialize it.
326  */
327  Config = XIicPs_LookupConfig(device_id);
328  if (NULL == Config)
329  {
330  return XST_FAILURE;
331  }
332 
333  status = XIicPs_CfgInitialize(instancePtr, Config, Config->BaseAddress);
334  if (status != XST_SUCCESS)
335  {
336  return XST_FAILURE;
337  }
338 
339  /* Perform a self-test to ensure that the hardware was built correctly. */
340  status = XIicPs_SelfTest(instancePtr);
341  if (status != XST_SUCCESS)
342  {
343  return XST_FAILURE;
344  }
345 
346  /* Set the IIC serial clock rate. */
347  XIicPs_SetSClk(instancePtr, IIC_SCLK_RATE);
348 
349  return status;
350 }

353  int status;
354  u8 writeData[1];
355  u8 readData[1];
356  u16 modelID;

358  /* Set the I2C Multiplexer to the channel required for addressing the picam */

377 int configure_picam_v1_iic(XIicPs *instancePtr) {
378  int status;
379  u8 writeData[1];
380  u8 readData[2];
381 
382  /* Set the I2C Multiplexer to the channel required for addressing the picam */
383  writeData[0] = 0x07;
384  status = XIicPs_MasterSendPolled(instancePtr, writeData, 1, IIC_MULTIPLEXER_ADDR);
385  if (status != XST_SUCCESS)
386  {
387  return XST_FAILURE;
388  }
389 
390  /* Wait until bus is idle to start another transfer. */
391  while (XIicPs_BusIsBusy(instancePtr))
392  {
393  /* NOP */
394  }
395 
396  write_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, 0x0100, 0x00); // Disable
397  write_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, 0x0103, 0x01); // Reset
398  for(int i=1000; i != 0; i--) {
399  /* wait for camera to reset */
400  }
401  write_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, 0x0103, 0x01); // Reset
402  for(int i=1000000; i != 0; i--) {
403  /* wait for camera to reset again but longer */
404  }
405  // Load common configuration
406  write_config_picam_iic(instancePtr, CAMERA_V1_3_IIC_ADDR,ov5647_sensor_common_10bit);
407  // load specific configuration
408  write_config_picam_iic(instancePtr, CAMERA_V1_3_IIC_ADDR,ov5647_sensor_1280_720_30);
409  write_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, 0x0100, 0x01); // Reset
410 
411  return XST_SUCCESS;
412 }

414 int as_picam_init(uint16_t iic_device_id) {
415  int status;
416  u8 writeData[1];
417  u8 readData[2];
418  u16 modelID;
419 
420  XIicPs *instancePtr = &instanceIIC;
421 
422  /* Init the I2C module */
423  init_iic(iic_device_id);
424 
425  /* Set the I2C Multiplexer to the channel required for addressing the picam */
426  writeData[0] = 0x07;
427  status = XIicPs_MasterSendPolled(instancePtr, writeData, 1, IIC_MULTIPLEXER_ADDR);
428  if (status != XST_SUCCESS)
429  {
430  return XST_FAILURE;
431  }
432 
433  /* Wait until bus is idle to start another transfer. */
434  while (XIicPs_BusIsBusy(instancePtr))
435  {
436  /* NOP */
437  }
438 
439  /* Check for version 2.19 camera */
440  readData[0] = read_iic_register(instancePtr, CAMERA_V2_1_IIC_ADDR, 0x0000);
441  if (status != XST_SUCCESS)
442  {
443  return XST_FAILURE;
444  }
445  modelID = readData[0] << 8;
446 
447  readData[0] = read_iic_register(instancePtr, CAMERA_V2_1_IIC_ADDR, 0x0001);
448  if (status != XST_SUCCESS)
449  {
450  return XST_FAILURE;
451  }
452  modelID |= readData[0];
453 
454  if(modelID == 0x0219) {
455 #ifdef DEBUG_PRINT
456  xil_printf("Raspberry Pi camera v2.19 found.\n\r");
457 #endif
458  status = configure_picam_v2_iic(instancePtr);
459  if (status != XST_SUCCESS)
460  {
461  return XST_FAILURE;
462  }
463  return XST_SUCCESS;
464  }
465 
466  /* Check for version 1.3 camera */
467  readData[0] = read_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, CS_CMMN_CHIP_ID_H);
468  if (status != XST_SUCCESS)
469  {
470  return XST_FAILURE;
471  }
472 
473  readData[1] = read_iic_register(instancePtr, CAMERA_V1_3_IIC_ADDR, CS_CMMN_CHIP_ID_L);
474  if (status != XST_SUCCESS)
475  {
476  return XST_FAILURE;
477  }
478 
479  if (!((readData[0] != 0x56) || (readData[1] != 0x47))) {
480 #ifdef DEBUG_PRINT
481  xil_printf("Raspberry Pi camera v1.3 found.\n\r");
482 #endif
483  status = configure_picam_v1_iic(instancePtr);
484  if (status != XST_SUCCESS)
485  {
486  return XST_FAILURE;
487  }
488  return XST_SUCCESS;
489  }
490 
491  return XST_FAILURE;
492 }

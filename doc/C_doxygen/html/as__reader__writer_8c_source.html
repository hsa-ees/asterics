38 /* Module Init Function */
39 
40 void as_reader_writer_init(as_hardware_address_t base_addr, as_reader_writer_config_t *config) {
41  if (config != NULL) {
42  as_reader_writer_set_section_offset(base_addr, (*config).section_offset);
43  as_reader_writer_set_section_addr(base_addr, (*config).first_section_addr);
44  as_reader_writer_set_section_size(base_addr, (*config).section_size);
45  as_reader_writer_set_section_count(base_addr, (*config).section_count);
46  as_reader_writer_set_max_burst_length(base_addr, (*config).max_burst_length);
47  }
48  /* Write Default Values to Registers */
49  else {
50  as_reader_writer_set_section_offset(base_addr, (uint32_t)AS_READER_WRITER_DEFAULT_SECTION_OFFSET);
51  as_reader_writer_set_section_size(base_addr, (uint32_t)AS_READER_WRITER_DEFAULT_SECTION_SIZE);
52  as_reader_writer_set_section_count(base_addr, (uint32_t)AS_READER_WRITER_DEFAULT_SECTION_COUNT);
53  as_reader_writer_set_max_burst_length(base_addr, (uint32_t)AS_READER_WRITER_DEFAULT_MAX_BURST_LENGTH);
54  }
55  /* Reset */
56  as_reader_writer_reset(base_addr);
57 }

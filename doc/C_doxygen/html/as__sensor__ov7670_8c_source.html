319 static int as_sensor_ov7670_identify(uint32_t *base_addr) {
320  uint8_t midl, midh = 0;
321  SENS_CONF *channel = ov7670_get_iic_channel(base_addr);
322 #ifdef DEBUG_PRINT
323  xil_printf("Getting Sensor ID. \n\r");
324 #endif
325  ov7670_iic_reg_get(channel, OV7670_IIC_SLV_ADDR, MIDH_REG, &midh);
326  ov7670_iic_reg_get(channel, OV7670_IIC_SLV_ADDR, MIDL_REG, &midl);
327 
328  if ((midh == MIDH_REG_VAL) && (midl == MIDL_REG_VAL)) {
329 #ifdef DEBUG_PRINT
330  xil_printf("Sensor ID ok: ");
331  xil_printf("MIDH=[0x%02X] MDIL=[0x%02X] \n\r", midh, midl);
332 #endif
333  return 1;
334  } else {
335 #ifdef DEBUG_PRINT
336  xil_printf("Sensor ID fail: ");
337  xil_printf("MIDH=[0x%02X] MDIL=[0x%02X] \n\r", midh, midl);
338 #endif
339  return 0;
340  }
341 }

343 uint8_t as_sensor_ov7670_init(uint32_t *base_addr, uint8_t iic_type,
344  uint32_t *iic_base_addr) {
345  SENS_CONF *channel;
346  // If this is the first ov7670 module to be initialized:
347  if (ov7670_sensors == NULL) {
348  ov7670_sensors = (SENS_CONF *)calloc(1, sizeof(SENS_CONF));
349  channel = ov7670_sensors;
350  } else {
351  // If an ov7670 module was already initialized
352  channel = ov7670_get_iic_channel(base_addr);
353  if (channel == NULL) {
354  // This is an unkown ov7670 base address => New module!
355  // Resize sensor array 'ov7670_sensors'
356  ov7670_sensor_count++;
357  realloc(ov7670_sensors, ov7670_sensor_count * sizeof(SENS_CONF));
358  channel = &(ov7670_sensors[ov7670_sensor_count - 1]);
359  }
360  }
361  // Store the IIC configuration along with the base address
362  channel->hw_base_addr = base_addr;
363  channel->iic_type = iic_type;
364  channel->iic_base_addr = iic_base_addr;
365 
366  IIC_SETREG(channel, COM2_REG, 0x00) /* Set Video Output Strength to 1x (lowest
367  possible value to avoid crosstalk) */
368 
369  // Check for sensor ID:
370  if (!as_sensor_ov7670_identify(base_addr))
371  return 0;
372 
373  as_sensor_ov7670_resolution_set(base_addr);
374  as_sensor_ov7670_reset(base_addr);
375 
376  /* Perform a reset to the sensors core: */
377  IIC_SETREG(channel, 0x12, 0x80) /* COM7 SCCB Register Reset */
378  /* wait some time after reset: */
379  usleep(10000);
380 
381  IIC_SETREG(channel, COM2_REG, 0x00) /* Set Video Output Strength to 1x (lowest
382  possible value to avoid crosstalk) */
383 
384  /* Set default iic register values for grayscale at VGA resolution: */
385  ov7670_set_registers_vga_gray(base_addr);
386 
387  /* Set 25fps for 50HZ light environment (0x66=25fps; 0x0=30fps): */
388  IIC_SETREG(channel, 0x92, 0x66)
389 
390  /* Set Contrast: */
391  IIC_SETREG(channel, 0x56, OV7670_INIT_CONTRAST_CONTROL)
392  IIC_SETREG(channel, 0x57, OV7670_INIT_CONTRAST_CENTER)
393 
394  /* Set Exposure and Gain to auto: */
395  as_sensor_ov7670_exposure_auto(base_addr, 1);
396  as_sensor_ov7670_gain_auto(base_addr, 1);
397 
398  return 1;
399 }

414  AS_SENSOR_OV7670_EXT_RESET_MASK, 0x0);

416 }

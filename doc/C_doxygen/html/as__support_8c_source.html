38 uint64_t as_swap64(uint64_t x)
39 {
40  return (x >> 56) | ((x & 0x00ff000000000000) >> 40) | ((x & 0x0000ff0000000000) >> 24) | ((x & 0x000000ff00000000) >> 8) | ((x & 0x00000000ff000000) << 8) | ((x & 0x0000000000ff0000) << 24) | ((x & 0x000000000000ff00) << 40) | (x << 56);
41 }

44 {

46 }

154 /* Tasklets are managed in a static array, so that no heap is required.
155  * If we are sure to have a heap, this may be changed to a chained list
156  * in order to get rid of the 'MAX_TASKLETS' limit.
157  */
158 
159 #define MAX_TASKLETS 32
160 
161 struct as_tasklet_s
162 {
163  as_tasklet_f *func;
164  void *data;
165 };
166 
167 static struct as_tasklet_s tasklets[MAX_TASKLETS];
168 static int tasklets_top = 0;
169 
170 struct as_tasklet_s *as_tasklet_new(as_tasklet_f *func, void *data)
171 {
172  struct as_tasklet_s *tasklet;
173 
174  AS_ASSERT(tasklets_top < MAX_TASKLETS);
175  tasklet = &(tasklets[tasklets_top++]);
176  tasklet->func = func;
177  tasklet->data = data;
178  return tasklet;
179 }
180 
181 void as_tasklet_del(struct as_tasklet_s *tasklet)
182 {
183  int n;
184 
185  n = tasklet - tasklets;
186  AS_ASSERT(n >= 0 && n < tasklets_top);
187  if (n < tasklets_top - 1)
188  tasklets[n] = tasklets[tasklets_top - 1];
189  tasklets_top--;
190 }

214 #define AS_ITER_NANOSECS 100
215 /* average/minimum number of nanoseconds of one delay loop iteration;
216  * to be adjusted for present processor */
217 
218 void as_sleep(int nanoseconds)
219 {
220  /* FIXME: This should be replaced or complemented by a version that uses a timer.
221  * A timer interface would have to be added to the support interface.
222  */
223  static volatile int i;
224 
225  i = 0;
226  while (nanoseconds > 0)
227  {
228  i += AS_ITER_NANOSECS;
229  nanoseconds -= AS_ITER_NANOSECS;
230  if (i >= 1000000)
231  { /* one 'as_yield' call approx. every millisecond */
232  as_yield();
233  i -= 1000000;
234  }
235  }
236 }

309 static void *posix_thread_routine(void *data)
310 {
311  struct timespec ts;
312  int n;
313 
314  ts.tv_sec = 0;
315  ts.tv_nsec = TASKLETS_SLEEP_TIME;
316  while (!tasklets_thread_quit)
317  {
318 #if SERIALIZE_TASKLETS
319  as_mutex_lock(&tasklet_m);
320 #endif
321  for (n = 0; n < tasklets_top; n++)
322  tasklets[n].func(tasklets[n].data);
323 #if SERIALIZE_TASKLETS
324  as_mutex_unlock(&tasklet_m);
325 #endif
326  nanosleep(&ts, NULL); /* We do not care about a potential pre-mature wakeup due to a signal. */
327  }
328 }

330 /********** as_sleep ***********/
331 
332 void as_sleep(int nanoseconds)
333 {
334  struct timespec ts, rem;
335 
336  ts.tv_sec = nanoseconds / 1000000000;
337  ts.tv_nsec = nanoseconds % 1000000000;
338  while (nanosleep(&ts, &rem) < 0)
339  ts = rem;
340  /* Notes: 1. This loop may in total last longer than 'nanoseconds' (see nanosleep(2) ).
341  * 2. The loop should probably do some more sanity checks.
342  */
343 }

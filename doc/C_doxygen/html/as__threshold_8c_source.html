122 void as_threshold_set_c_value(uint32_t* base_addr, uint32_t value){
123  uint32_t regval;
124  /* upper 4 bits of c are within reg 1 */
125  regval = as_reg_read( as_module_reg(base_addr, AS_THRESHOLD_PARAMETER_1_REG_OFFSET));
126  regval &= 0xf0ffffff; /* clear old c_upper value */
127  regval = regval | ((value<<16) & 0x0f000000); /* set new c_upper value */
128  as_reg_write( as_module_reg(base_addr, AS_THRESHOLD_PARAMETER_1_REG_OFFSET), regval);
129  /* lower 8 bits of c are within reg 2 */
130  regval = as_reg_read( as_module_reg(base_addr, AS_THRESHOLD_PARAMETER_2_REG_OFFSET));
131  regval &= 0x00ffffff; /* clear old c_lower value */
132  regval = regval | ((value<<24) & 0xff000000); /* set new c_lower value */
133  as_reg_write( as_module_reg(base_addr, AS_THRESHOLD_PARAMETER_2_REG_OFFSET), regval);

78 ea_bool ea_draw_int(uint32_t x, uint32_t y, uint32_t muster, uint32_t color)
79 {
80  ea_bool returnval= ea_true;
81  uint32_t temp=0;
82  uint32_t int_cnt = 0;
83  uint32_t color_cnt = 0;
84  uint32_t value = 0;
85  uint32_t rest = 0;
86  uint32_t rpos =x+y*WIDTH; //real position
87  uint32_t pos = rpos/ ( PIXEL_PER_INT); //table pos
88  uint32_t shift = (rpos%(PIXEL_PER_INT));
89  uint32_t xl_off = shift * COLOR_DEPTH;
90  uint32_t xr_off = BUS_WIDTH;
91 
92  if ( y > Y_LOW || y < Y_HIGH) return ea_false; //Check Y-Direction
93  if ( x > X_RIGHT) return ea_false; //Check X_RIGHT-Direction nichts mehr zu tun
94  if ( x < X_LEFT ) //Check X_LEFT-Direction
95  {
96  xl_off = (X_LEFT -x)*COLOR_DEPTH;
97  if (xl_off > BUS_WIDTH)
98  return ea_false; //nichts mehr zu tun
99  returnval = ea_false;
100  }
101 
102  rest = muster & ((1<<shift)-1); // get rest
103  rest = ((rest <<(PIXEL_PER_INT))>> shift); //bring rest to right position
104  muster >>=shift; //shift muster
105 
106  for (int_cnt = 0 ; int_cnt <= COLOR_DEPTH ; int_cnt++)
107  {
108  temp = 0; //zuruecksetzen fuer neues Pixel
109  if (muster == 0) return returnval; //Cancel if nothing to do
110  for (color_cnt = xl_off; color_cnt < xr_off ;color_cnt+= COLOR_DEPTH) // 0 - 1 bei 2 bit
111  {
112 
113  #if COLOR_DEPTH == 2
114  temp |= (0x80000000 & (muster<< (color_cnt / 2))) >> (color_cnt);
115  temp |= (0x80000000 & (muster<< (color_cnt / 2))) >> (color_cnt | 1);
116  #else
117 
118  for ( bit_cnt = 0 ; bit_cnt < COLOR_DEPTH ; bit_cnt ++)
119  temp |= (0x80000000 & (muster<< (color_cnt / COLOR_DEPTH))) >> (color_cnt | bit_cnt);
120  #endif
121  }
122  value = overlay_table[pos + int_cnt] & ~temp; //Read existing Overlay
123  overlay_table[pos + int_cnt] = value | (temp & mask[color]); //Write muster
124 
125  muster <<= (PIXEL_PER_INT);
126  muster |=rest;
127  rest=0;
128  xl_off = 0; //zur�ckstellen damit der rest gedruckt werden kann
129  }
130 
131 
132  return returnval;
133 }

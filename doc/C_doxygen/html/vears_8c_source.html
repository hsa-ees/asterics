95 int8_t vears_init(uint32_t vears_iobase, uint8_t *image_base) {
96  uint32_t frame_width, frame_height;
97 
98  vears_reset (vears_iobase);
99 
100  if (!vears_get_resolution(vears_iobase, &frame_width, &frame_height)){
101  vears_initialized = 0;
102  return -1;
103  } else {
104  vears_frame_width = frame_width;
105  vears_frame_height = frame_height;
106  vears_initialized = 1;
107  }
108 
109  if (!image_base) {
110  return -1;
111  } else {
112  vears_image_show(vears_iobase, image_base);
113  vears_enable (vears_iobase);
114  }
115 
116  ea_init ();
117  flush_cache_for_image ();
118 
119  return 0;
120 }

145 void vears_overlay_on (uint32_t vears_iobase){
146 
147  // Switch to (and create) internal overlay if non is set ...
148  if (!curOverlay) {
149  if (!internalOverlay) {
150  internalOverlay = malloc (vears_frame_width * vears_frame_height / 4);
151  ASSERT (internalOverlay != NULL);
152  vears_overlay_drawto (internalOverlay);
153  vears_overlay_clear ();
154  }
155  vears_overlay_show (vears_iobase, internalOverlay);
156  vears_overlay_drawto (internalOverlay);
157  }
158 
159  // Switch on the overlay...
160  ctrlReg |= VEARS_CONTROL_REG_OVERLAY_ENABLE_MASK;
161  Xil_Out32 (vears_iobase + VEARS_REG_CONTROL, ctrlReg);
162 
163  // Reset color map...
164  vears_overlay_set_color(vears_iobase, 1, STD_COLOR_1);
165  vears_overlay_set_color(vears_iobase, 2, STD_COLOR_2);
166  vears_overlay_set_color(vears_iobase, 3, STD_COLOR_3);
167 }

204 int8_t vears_get_resolution (uint32_t vears_iobase, uint32_t *screen_width, uint32_t *screen_height){
205  // read status register
206  uint32_t reg_status = Xil_In32(vears_iobase + VEARS_REG_STATUS);
207  uint8_t video_group = (uint8_t)(reg_status);
208  uint8_t video_mode = (uint8_t)(reg_status>>8);
209 
210  if ( ( video_group != 0 ) && ( video_mode != 0 ) ) {
211  *screen_width = video_settings_group_mode_array[video_group-1][video_mode-1].H_Tdisp;
212  *screen_height = video_settings_group_mode_array[video_group-1][video_mode-1].V_Tdisp;
213  return 1;
214  } else {
215  *screen_width = 0;
216  *screen_height = 0;
217  return -1;
218  }
219 }

334 void vears_draw_icon (uint32_t x, uint32_t y, uint32_t color, const char *icon, uint32_t w, uint32_t h) {
335  const char *p;
336  char c;
337  uint32_t dx, dy;
338 
339  p = icon;
340  for (dy = 0; dy < h; dy++) {
341  for (dx = 0; dx < w; dx++) {
342  switch ( (c = (*p++)) ) {
343  case ' ':
344  case '.':
345  break;
346  case '+':
347  ea_draw_point (x + dx, y + dy, (1 << COLOR_DEPTH) - 1);
348  break;
349  case '0':
350  case '1':
351  case '2':
352  ea_draw_point (x + dx, y + dy, c - '0');
353  break;
354  default:
355  ea_draw_point (x + dx, y + dy, color);
356  }
357  }
358  }
359  flush_cache_for_overlay ();
360 }

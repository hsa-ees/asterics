84 def resolve_data_width(data_width: Port.DataWidth, generics: list) -> list:
85  """! @brief Resolve data_width using a list of generics."""
86  if data_width.is_resolved():
87  return data_width
88 
89  gendict = {}
90  for gen in generics:
91  gendict[gen.code_name] = gen.get_value()
92 
93  spliton = data_width.sep
94  value0 = eval_vhdl_expr(
95  str(data_width.a).strip().upper(), "data width", gendict
96  )
97  value1 = eval_vhdl_expr(
98  str(data_width.b).strip().upper(), "data width", gendict
99  )
100  return Port.DataWidth(a=value0, sep=spliton, b=value1)

103 def report_buffer_statistics(
104  buffer_rows: list, minimum_bram_size: int, verbosity: int = 0
105 ):
106  """! @brief Prints buffer statistics to console.
107  Calculates statistics for used flip-flops and BRAM resources by image
108  buffers and prints to console.
109  A summary (default) or a per-buffer report can be created using 'verbosity'.
110  """
111  count = len(buffer_rows)
112  total_size = 0
113  count_bram = 0
114  count_reg = 0
115  size_bram = 0
116  size_reg = 0
117 
118  for buff in buffer_rows:
119  buffsize = buff.get_size()
120  total_size += buffsize
121  if buffsize > minimum_bram_size:
122  count_bram += 1
123  window_size = buff.window_width * buff.get_bit_width()
124  size_bram += buffsize - window_size
125  size_reg += window_size
126  else:
127  count_reg += 1
128  size_reg += buffsize
129  print(
130  (
131  "\n"
132  "###############################################\n"
133  "Automatics 2D Window Pipeline Buffer Row Report\n"
134  "###############################################\n"
135  "\n"
136  "Total number of buffers: {count}\n"
137  "Total buffer size in bits: {size}\n"
138  "Number of buffers implemented as BRAM: {count_bram}\n"
139  "Number of buffers implemented as registers only: {count_reg}\n"
140  "Total size of BRAM required in bits: {size_bram}\n"
141  "Total size of registers required: {size_reg}\n"
142  "\n"
143  ).format(
144  count=count,
145  size=total_size,
146  count_bram=count_bram,
147  count_reg=count_reg,
148  size_bram=size_bram,
149  size_reg=size_reg,
150  )
151  )
152  if verbosity > 0:
153  count = 0
154  for buff in buffer_rows:
155  buffsize = buff.get_size()
156  if buffsize <= minimum_bram_size:
157  size_reg = buffsize
158  size_bram = 0
159  else:
160  window_size = buff.window_width * buff.get_bit_width()
161  size_bram = buffsize - window_size
162  size_reg = window_size
163  print(
164  (
165  "-- {num} --\n"
166  "Buffer '{name}':\n"
167  "Input delay: {in_delay}\n"
168  "Output delay: {out_delay}\n"
169  "Buffer length: {length}\n"
170  "Bit width: {bit_width}\n"
171  "Window width: {window_width}\n"
172  "BRAM size in bits: {size_bram}\n"
173  "Register size in bits: {size_reg}\n"
174  "\n"
175  ).format(
176  num=count,
177  name=buff.name,
178  in_delay=buff.input_delay,
179  out_delay=buff.output_delay,
180  length=buff.length,
181  bit_width=buff.get_bit_width(),
182  window_width=buff.window_width,
183  size_bram=size_bram,
184  size_reg=size_reg,
185  )
186  )
187  count += 1

191 def generate_window_assignments(window_signals: list, code_dict: dict):
192  """! @brief Generate VHDL code to assign image rows to t_generic_window signals.
193  Creates a process for use in a 2D Window Pipeline subsystem.
194  Assigning t_generic_line signals to one or more t_generic_window signals."""
195  process_body = []
196  line_arrays = []
197  input_line_signals = set()
198  window_configs = set()
199 
200  # For all window ports
201  for ws in window_signals:
202  wp = ws.outgoing[0]
203  winter = wp.parent
204  line_arrays.append(
205  LINE_ARRAY_VARIABLE_DECLARATION.format(
206  window=ws.code_name,
207  width=winter.window.x,
208  height=winter.window.y,
209  bits=ws.data_width.get_bit_width(),
210  )
211  )
212  window_configs.add((winter.window.x, ws.data_width.get_bit_width()))
213  # Add comment with the window signal name
214  process_body.append(" -- Build window '{}'".format(ws.code_name))
215 
216  # 'incoming' stores the buffers that provide data for this window
217  for row_idx in range(len(winter.incoming)):
218  buff = winter.incoming[row_idx]
219 
220  # We need to scan all inputs
221  for in_port, window_connection, to_ports in zip(
222  buff.inputs,
223  buff.is_window_signal,
224  buff.to_window_ports,
225  ):
226  # Skip unrelated inputs (^= buffer slices)
227  if (not window_connection) or ((wp, row_idx) not in to_ports):
228  continue
229  line_signal = buff.line_signal.code_name
230  input_line_signals.add(line_signal)
231 
232  # Fill assignment template
233  process_body.append(
234  LINE_SIGNAL_TO_LINE_ARRAY.format(
235  row_idx=row_idx,
236  line_signal=line_signal,
237  start_bit_idx=in_port.start_index,
238  bit_width=in_port.bit_width,
239  width=winter.window.x,
240  window=ws.code_name,
241  )
242  )
243  process_body.append(
244  WINDOW_FROM_LINE_ARRAY.format(
245  window=ws.code_name,
246  width=winter.window.x,
247  bits=ws.data_width.get_bit_width(),
248  )
249  )
250  sens = list(sorted(input_line_signals))
251  code_dict["body"].append(
252  PROCESS_WITH_VARIABLES.format(
253  sens=", ".join(sens),
254  variables="\n".join(line_arrays),
255  in_process="\n".join(process_body),
256  )
257  )
258  for width, bits in window_configs:
259  code_dict["signals"].append(
260  LINE_ARRAY_TYPE_DECLARATION.format(width=width, bits=bits)
261  )
262  code_dict["signals"].append(
263  LINE_ARRAY_TO_WINDOW_FUNCTION.format(width=width, bits=bits)
264  )
265  code_dict["signals"].append(" ")

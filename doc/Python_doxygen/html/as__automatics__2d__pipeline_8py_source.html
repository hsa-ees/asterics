430  def _propagate_delay(self, source):
431  """! @brief Recursive method to propagate a strobe/pixel delay across the pipeline.
432  Uses the delay of Port 'source' as the input delay.
433  Passes the delay to all output ports of the module.
434  Calls itself for all output port connections.
435  Stops if moving outside of this module group (the pipeline)
436  or when reaching a module with equal or greater input delay."""
437  module = get_parent_module(source)
438  # Special case for modules looping back a window port
439  if module is get_parent_module(source.incoming):
440  set_delay(source, module.input_delay)
441  return None
442  LOG.debug("DP @ '%s' for '%s'", module.name, source.code_name)
443  self.pipeline_delay = max(get_delay(source), self.pipeline_delay)
444  # If we made it to outside of the pipeline, stop!
445  if module.parent is not self:
446  LOG.debug("STOP! Outside of pipeline.")
447  return None
448  try:
449  # For window modules: Update the input delay attribute if necessary
450  if module.input_delay < get_delay(source):
451  module.input_delay = get_delay(source)
452  module.__update_delay__()
453  LOG.debug("Updated module delay to " + str(module.delay))
454  else:
455  # We only need to update the delays of the module,
456  # if the input delay of the module is greater than
457  # the source delay or if it isn't (-1) (initial value)
458  LOG.debug("STOP! Module delay is greater than source delay.")
459  return None
460  except AttributeError:
461  # For regular modules: Use the single delay attribute
462  if get_delay(module) is not None:
463  if get_delay(module) < get_delay(source):
464  # Found an input port with a higher delay than previously
465  # Update the delay and all module connections
466  set_delay(module, get_delay(source))
467  LOG.debug("Updated module delay to " + str(module.delay))
468  else:
469  # We only need to update the delays of the module,
470  # if the input delay of the module is greater than
471  # the source delay or if it isn't (-1) (initial value)
472  LOG.debug(
473  "STOP! Module delay is greater/equal than source delay."
474  )
475  return None
476  else: # No delay set, this is the initial pass for this module
477  set_delay(module, get_delay(source))
478  LOG.debug("Updated module delay to " + module.delay)
479  # Need to handle this module! For all ports
480  for port in module.get_full_port_list(include_signals=False):
481  # We move from a module input to all of it's outputs
482  # Only handle non-interface ports that are outputs
483  if (
484  port.port_type in ("single", "interface")
485  and port.get_direction_normalized() == "out"
486  ):
487  LOG.debug("For port '%s'", port.code_name)
488  # Set the delay for this port
489  set_delay(port, get_delay(module))
490  # Handle all port connections
491  for target in port.outgoing:
492  LOG.debug("Found connection to '%s'", target.code_name)
493  # Update the delay for the connected port
494  if isinstance(target.parent, AsWindowInterface):
495  window = target.parent.window
496  set_delay(
497  target,
498  get_delay(module)
499  + window.get_delay(self.columns)
500  + 1,
501  )
502  else:
503  set_delay(target, get_delay(module))
504  # and run this method again for the port
505  self._propagate_delay(target)

507  def _add_delay_lines(self):
508  """! @brief Add delay lines where needed (automatically).
509  For all modules: For every relevant input that has a lower pixel delay
510  than the input with the highest pixel delay, add a delay line with a
511  length of the difference between the highest delay and the delay
512  of the port."""
513  for mod in self.modules:
514  # All inputs of the module
515  input_list = [p for p in mod.ports if p.direction == "in"]
516  if isinstance(mod, AsWindowModule): # Add window inputs
517  input_list.extend(
518  (wif.window_port for wif in mod.window_interfaces)
519  )
520  # Tuple: (Delay value, input port)
521  delays = []
522  # Compile the delay list; Leave out ports without a delay value
523  for inp in input_list:
524  try:
525  delays.append((inp.delay, inp))
526  except AttributeError:
527  pass
528  # If only one input port is present, we don't need any delay lines
529  if len(delays) <= 1:
530  continue
531  # Sort the delay list in descending order
532  delays = sorted(delays, key=lambda delay: delay[0], reverse=True)
533  high_delay = delays[0][0]
534 
535  # Are there any delays that differ from the greatest delay (idx 0)?
536  if any((delay[0] != high_delay for delay in delays)):
537  # We can skip the first delay, as it is the reference value
538  for delay in delays[1:]:
539  # If the delay values differ, add a delay line
540  if delay[0] < high_delay:
541  LOG.info(
542  (
543  "Delay line inserted for\nModule '%s' \n"
544  "Port '%s' \nWith delay %i\nInput delay %i"
545  "\nPort delay %i\n"
546  ),
547  mod.name,
548  delay[1].code_name,
549  high_delay - delay[0],
550  high_delay,
551  delay[0],
552  )
553  self._add_delay_line(
554  mod, delay[1], high_delay - delay[0]

561  def _add_delay_line(self, module: AsModule, port: Port, delay: int):
562  """! @brief Insert a delay line infront of 'port' of 'module'.
563  Connect the input of the delay line to the incoming port of 'port'
564  and the output to the input of 'port'
565  The bit width of the delay line is port's bitwidth
566  and the length is the provided 'delay'."""
567  source = port.incoming
568 
569  dl_name = (
570  get_parent_module(source).name
571  + "_"
572  + source.code_name
573  + "_buffer_line_0"
574  )
575  dl_num = 0
576  for buff in self.buffer_rows:
577  if (
578  buff.name == dl_name
579  and buff.inputs[0].port.incoming[0] is source
580  ):
581  if buff.length == delay:
582  self.__connect__(buff.outputs[0].port, port)
583  return None
584  else:
585  dl_name, dl_num = tuple(dl_name.rsplit("_", maxsplit=1))
586  dl_num = int(dl_num) + 1
587  dl_name += "_" + str(dl_num)
588  continue
589 
590  delay_line = AsPipelineRow(dl_name, delay - 1, 1, self)
591 
592  if get_parent_module(source) is self:
593  signame = source.code_name + "_delay_input_" + str(dl_num)
594  else:
595  signame = (
596  get_parent_module(source).name
597  + "_"
598  + source.code_name
599  + "_delay_input_"
600  + str(dl_num)
601  )
602 
603  in_signal = self.define_signal(
604  signame, source.data_type, source.data_width
605  )
606  set_delay(in_signal, get_delay(source))
607  source.glue_signal = in_signal
608  source.outgoing.append(in_signal)
609  in_signal.incoming.append(source)
610 
611  if not delay_line.add_input(in_signal):
612  raise AsConnectionError(
613  port, "Unable to add delay line because of previous errors!"
614  )
615  # Create an intermediate signal between the buffer and the target port
616  if get_parent_module(source) is self:
617  signame = source.code_name + "_delayed_" + str(dl_num)
618  else:
619  signame = (
620  get_parent_module(source).name
621  + "_"
622  + source.code_name
623  + "_delayed_"
624  + str(dl_num)
625  )
626  out_signal = self.define_signal(
627  signame, source.data_type, source.data_width
628  )
629  # Remove incoming port connection
630  if isinstance(port.parent, AsWindowInterface):
631  port.glue_signal = out_signal
632  else:
633  port.glue_signal = None
634  port.incoming = None
635  # and reconnect the port with the delayed signal
636  self.__connect__(out_signal, port)
637  # Add the signal as an output of the buffer line
638  delay_line.add_output(out_signal)
639  # Add the buffer to the pipeline
640  self.buffer_rows.append(delay_line)

644  def _merge_same_signal_buffers(self):
645  """! @brief Optimize buffers that delay the same signal by merging them.
646  This method looks for buffers that have the same input signal.
647  These buffers of differing delay length are then merged:
648  @verbatim
649  E.g. sig1 -> [BUFFER 1200] -> out1
650  sig1 -> [BUFFER 400 ] -> out2
651  Is merged to:
652  sig1 -> [BUFFER 400] -+-> [BUFFER 800] -> out1
653  +-> out2

663  except (IndexError, TypeError):

668  treated_signals.append(crt_signal)

677  continue

685  bufflist.append(comp_buff)
686  # --------- MERGE buffers -------------------

688  if len(bufflist) == 1:
689  continue

691  bufflist.sort(key=lambda buff: buff.length)
692  LOG.debug("Merging buffers: %s", str(bufflist))
693  # Merge the buffers
694  prev_buff = bufflist[0]
695  for buff in bufflist[1:]:
696

698  start_delay = prev_buff.input_delay

701  delay_diff = buff.output_delay - prev_buff.output_delay

712  sig_out.incoming.append(buff_input_signal)
713  else:
714  sig_out.incoming = buff_input_signal
715  # If the delay difference is zero, we can replace buff
716  # with prev_buff: Remove buff and all signals and connect the
717  # output glue signal of prev_buff to the target of buff
718  if delay_diff == 0:
719  # Mark buffer to be removed
720  removed_buffers.append(buff)
721  # Remove the connection for buff's input signal
722  # from the source signal
723  sig_to_remove.incoming[0].outgoing.remove(sig_to_remove)
724  # Remove the input glue signal for buff
725  self.remove_signal(sig_to_remove)
726  # Buff's output glue signal
727  old_source = buff.outputs[0].port
728  # Buff's target port
729  target = old_source.outgoing[0]
730  # Remove the target port's connections to the glue signal
731  target.incoming = None
732  target.glue_signal = None
733  # Remove the glue signal
734  self.remove_signal(old_source)

736  self.__connect__(
737  prev_buff.outputs[0].port,
738  target,
739  )

742  buff.input_delay = prev_buff.output_delay
743  # Change BUFFs length to the delay difference
744  buff.set_buffer_length(delay_diff)
745  # Move to next buffer
746  prev_buff = buff
747  for buff in removed_buffers:
748  # Remove buffers that were merged into other buffers
749  self.modules.remove(buff.module)
750  self.buffer_rows.remove(buff)

877  def _reshape_long_buffers(
878  self, min_length: int = -1, maximum_width: int = 7
879  ):
880  """! @brief Buffer optimization reshaping long and thin buffers.
881  This optimization turns long and thin buffers into wider but shorter
882  buffers, potentially using fewer BRAM resources.
883  @param min_length Any buffers shorter than this value are not optimized
884  This value is also used as a target length for the optimization
885  @param maximum_width Buffers wider than this value are not optimzed"""
886 
887  if min_length == -1:
888  min_length = int(self.minimum_bram_size * 2.5)
889  elif min_length < self.minimum_bram_size:
890  return None
891 
892  bufflist = [
893  b
894  for b in self.buffer_rows
895  if b.length > min_length
896  and b.get_bit_width() <= maximum_width
897  and not any(b.is_window_signal)
898  and len(b.inputs) == 1
899  ]
900 
901  for buff in bufflist:
902  iterations = int(buff.length / min_length)
903  new_length = int(buff.length / iterations)
904  left_over = buff.length - new_length * iterations
905 
906  buff.set_buffer_length(new_length)
907  bsig = buff.outputs[0].port
908  buff.remove_outputs()
909  for n in range(iterations - 1):
910  intersig = self.define_signal(
911  bsig.code_name + "_loopdelay_" + str(n),
912  bsig.data_type,
913  bsig.data_width,
914  )
915  set_delay(intersig, buff.input_delay + new_length * (n + 1))
916  buff.add_output(intersig)
917  buff.add_input(intersig, check_delay=False)
918  if left_over == 0:
919  buff.add_output(bsig)
920  else:
921  buff.output_delay = new_length * iterations
922  new_buff = AsPipelineRow(
923  buff.name + "_last_delay", left_over, 1, self
924  )
925  intersig = self.define_signal(
926  bsig.code_name + "_last_delay",
927  bsig.data_type,
928  bsig.data_width,
929  )
930  set_delay(intersig, buff.output_delay - left_over)
931  buff.add_output(intersig)
932  new_buff.add_input(intersig)
933  new_buff.add_output(bsig)
934  self.buffer_rows.append(new_buff)

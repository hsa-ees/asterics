82 class AsPipelineRow:
83  """! @brief Class representing an image row buffer of the
84  as_automatics_2d_pipeline::As2DWindowPipeline.
85  The pipeline stores image data in buffers each holding one or more
86  image rows storing the same number of pixels.
87  Each buffer has a section stored in registers, accessible simultaneously and
88  a BRAM part, where only the last pixel in the FIFO BRAM is readable.
89  These buffers are described by this AsPipelineRow class.
90  """

165  def add_window_interface_row(
166  self, window_in, row_index: int, input_port: Port = None
167  ) -> GenericSignal:
168  """! @brief Add a window interface row as a buffer target.
169  Adds itself as the buffer row into the WindowDef of the window port,
170  at row index 'row_index'. Generates an intermediate signal for data
171  transport between the output of the buffer component and assigning
172  the window signal, which is returned on success.
173  @param window_in The window interface to supply with data
174  @param row_index The row of the window interface to target
175  @param input_port The Port supplying the image data
176  @return intermediate signal of this buffer's delayed data output
177  """
178  try:
179  wport = window_in.window_port
180  window = window_in.window
181  except AttributeError:
182  raise AsConnectionError(
183  window_in,
184  "Invalid object passed to 'add_window_interface' for "
185  "buffer '{}': {}!".format(self.name, window_in),
186  )
187  source = None
188  if row_index == 0:
189  if wport.glue_signal:
190  source = wport.glue_signal
191  elif wport.incoming.glue_signal:
192  source = wport.incoming.glue_signal
193  else:
194  source = wport.incoming
195  else:
196  source = input_port
197  # Register the source with this buffer
198  if not self.add_input(
199  source,
200  is_window_signal=True,
201  to_window_port=wport,
202  window_row_idx=row_index,
203  ):
204  return None
205  window.rows[row_index] = self
206  signame = (
207  window_in.parent.name
208  + "_"
209  + wport.code_name
210  + "_row_"
211  + str(row_index)
212  )
213  im_sig = self.pipe.define_signal(
214  signame, data_type="std_logic_vector", data_width=wport.data_width
215  )
216  set_delay(im_sig, self.output_delay)
217  self.add_output(im_sig)
218  return im_sig

220  def add_input(
221  self,
222  data_input: Port,
223  is_window_signal: bool = False,
224  to_window_port=None,
225  window_row_idx: int = 0,
226  check_delay: bool = True,
227  ) -> bool:
228  """! @brief Add a Port as an input for this pipeline buffer row.
229  Extends this buffer by the port.
230  @param data_input The Port to add as an input of this buffer
231  @param is_window_signal Marks this part of the data buffer as a window signal
232  @param to_window_port The window port targeted by this part of the buffer
233  Must be set if is_window_signal is set to True
234  @param window_row_idx The row index this part of the buffer targets
235  Must be set if is_window_signal is set to True
236  @param check_delay Whether or not to make sure the input delay of the
237  Port added matches the other input Ports
238  @return True on success, else False
239  """
240  # Do we already have this data input?
241  if data_input in (inp.port for inp in self.inputs):
242  LOG.warning(
243  "Port input '%s' already in '%s'!", str(data_input), self.name
244  )
245  return False
246  # Does the data input have a delay value?
247  if get_delay(data_input) is None:
248  LOG.error(
249  "Port '%s' added to '%s' without a data delay value!",
250  str(data_input),
251  self.name,
252  )
253  return False
254  # If this is the first input, use it's delay as the input delay
255  if self.input_delay is None:
256  self.input_delay = get_delay(data_input)
257  self.output_delay = self.input_delay + self.length
258  elif check_delay and get_delay(data_input) != self.input_delay:
259  LOG.error(
260  "Port '%s' added to '%s' with differing input delays!",
261  str(data_input),
262  self.name,
263  )
264  return False
265  # Create and append input management object
266  new_input = AsPipelineDataInfo(data_input, self.bit_width_in)
267  if new_input.bit_width is None:
268  raise AsConnectionError(
269  data_input,
270  (
271  "Port added to buffer '{}' in '{}' "
272  "has unresolved data width [{}]!"
273  ).format(self.name, self.pipe.name, data_input.data_width),
274  )
275  self.inputs.append(new_input)
276  self.is_window_signal.append(is_window_signal)
277  if isinstance(to_window_port, list):
278  self.to_window_ports.append(to_window_port)
279  else:
280  self.to_window_ports.append([(to_window_port, window_row_idx)])
281 
282  # Update bit width and module generic
283  self.bit_width_in += new_input.bit_width
284  self.module.set_generic_value("DATA_WIDTH", self.bit_width_in)
285  return True

287  def add_output(self, data_output: Port, update_delay: bool = True) -> bool:
288  """! @brief Add a data output Port to this buffer.
289  The next unconnected output index is used for the connection.
290  @param data_output The Port to supply with data from this buffer
291  @param update_delay Whether or not to update the strobe delay of the Port
292  @return True on success, else False
293  """
294  if data_output in (out.port for out in self.outputs):
295  LOG.warning(
296  "Port output '%s' already in '%s'!", str(data_output), self.name
297  )
298  return False
299  if data_output.data_width.get_bit_width() != self.inputs[-1].bit_width:
300  LOG.error(
301  (
302  "Bit width of data output '%s' added to '%s' does not "
303  "match corresponding data input '%s': %i <-> %i"
304  ),
305  str(data_output),
306  self.name,
307  str(self.inputs[-1].port),
308  data_output.data_width.get_bit_width(),
309  self.inputs[-1].bit_width,
310  )
311  return False
312  new_output = AsPipelineDataInfo(data_output, self.bit_width_out)
313  if new_output.bit_width is None:
314  raise AsConnectionError(
315  data_output,
316  (
317  "Port added to buffer '{}' in '{}' "
318  "has unresolved data width [{}]!"
319  ).format(self.name, self.pipe.name, data_output.data_width),
320  )
321  self.outputs.append(new_output)
322 
323  if update_delay:
324  new_output.delay = self.output_delay
325  set_delay(new_output.port, self.output_delay)
326  self.bit_width_out += new_output.bit_width

335  def build_inout_vectors(self) -> GenericSignal:
336  # Get module ports
337  mod_in_port = self.module.get_port("buff_in")
338  mod_out_port = self.module.get_port("data_out")
339  mod_line_port = self.module.get_port("line_out")
340 
341  foreach(self.module.get_full_port_list(), resolve_generic)
342 
343  # Create intermediate vector signals
344  sig_name = self.name + "_data_"
345  self.in_signal = self.pipe.define_signal(
346  sig_name + "in",
347  data_type="std_logic_vector",
348  data_width=(self.bit_width_in - 1, "downto", 0),
349  )
350  self.out_signal = self.pipe.define_signal(
351  sig_name + "out",
352  data_type="std_logic_vector",
353  data_width=(self.bit_width_in - 1, "downto", 0),
354  )
355  # Connect them to the buffers input and output
356  self.pipe.chain.__connect__(self.in_signal, mod_in_port, top=self.pipe)
357  self.pipe.chain.__connect__(
358  mod_out_port, self.out_signal, top=self.pipe
359  )
360  # Define vector signal connections
361  for source in self.inputs:
362  self.in_signal.assign_to_this_vector(
363  source.port, source.start_index
364  )
365  for target in self.outputs:
366  self.out_signal.assign_from_this_vector(
367  target.port, target.start_index
368  )
369  # Create line port signal
370  self.line_signal = self.pipe.define_signal(
371  self.name + "_line_data",
372  data_type="t_generic_line",
373  data_width=(self.bit_width_in - 1, "downto", 0),
374  )
375  setattr(
376  self.line_signal,
377  "line_width",
378  Port.DataWidth(0, "to", self.window_width - 1),
379  )
380  # Connect to buffer module line port
381  self.pipe.__connect__(self.line_signal, mod_line_port)
382  return self.line_signal

384  def merge(self, prow, merge_outputs: bool = True):
385  """Merge AsPipelineRow 'prow' into this pipeline row."""
386  for idx in range(len(prow.inputs)):
387  window_ports = prow.to_window_ports[idx]
388  for wp, row_idx in window_ports:
389  if not wp:
390  continue
391  winter = wp.parent
392  winter.incoming.remove(prow)
393  winter.incoming.insert(row_idx, self)
394  winter.window.rows[row_idx] = self
395 
396  self.add_input(
397  prow.inputs[idx].port,
398  prow.is_window_signal[idx],
399  prow.to_window_ports[idx],
400  check_delay=False,
401  )
402  if merge_outputs:
403  for out in prow.outputs:
404  self.add_output(out.port, update_delay=False)

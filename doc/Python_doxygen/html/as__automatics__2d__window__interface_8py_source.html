81  def update_window(self) -> bool:
82  """! @brief Update and set the window size for this interface.
83  Also attempts to parse the variable size of the 't_generic_window'
84  datatype.
85  @return Whether the t_generic_window data type could be parsed."""
86  parse_successful = True
87  port = self.window_port
88 
89  # Find and assign generics to the port
90  # Put the data widths into a single list
91  widths = ittls.chain([port.data_width], port.window_config)
92  # Extract the generics and package the result into a list
93  gen_names = ittls.chain(*(extract_generics(w) for w in widths))
94  # Add found generics of the parent module to the port this interface
95  for name in gen_names:
96  gen = self.parent.get_generic(name, suppress_error=True)
97  if gen:
98  port.add_generic(gen)
99  self.add_generic(gen)
100 
101  port.window_config = [
102  resolve_data_width(width, port.generics)
103  for width in port.window_config
104  ]
105 
106  if not all((width.is_resolved() for width in port.window_config)):
107  LOG.warning(
108  (
109  "Could not determine window size for "
110  "interface '%s' of module '%s'! Unresolved "
111  "generics present in '%s'!"
112  ),
113  self.name,
114  self.parent.name,
115  str(port.window_config),
116  )
117  parse_successful = False
118  else:
119  self.window = self.data_widths_to_window(*port.window_config)
120 
121  return parse_successful

124  @staticmethod
125  def data_widths_to_window(
126  columns: Port.DataWidth, rows: Port.DataWidth
127  ) -> WindowDef:
128  if columns.sep == "to":
129  to_x = columns.b
130  else:
131  to_x = columns.a
132  if rows.sep == "to":
133  to_y = rows.b
134  else:
135  to_y = rows.a
136  # Only return a window object, if the indices are resolved (no generics!)
137  if any((not isinstance(idx, int) for idx in (to_x, to_y))):
138  return None
139  return WindowDef(to_x + 1, to_y + 1)

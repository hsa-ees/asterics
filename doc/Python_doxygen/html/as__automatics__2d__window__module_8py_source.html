205  def __assign_window_interfaces__(self):
206  """! @brief Picks out all window ports and bundles them to WindowInterfaces.
207  Must be run before regular interfaces are assigned."""
208 
209  to_remove = [] # Collecting ports assigned before
210  for port in self.entity_ports:
211  # If port is not part of a window interface, skip it
212  if not getattr(port, "window_config"):
213  continue
214  # Create a new window interface and add the port
215  inter = AsWindowInterface()
216  inter.name = port.code_name
217  inter.add_port(port)
218  inter.window_port = port
219  inter.assign_to(self)
220  inter.direction = port.direction
221 
222  # Register the window interface with the module
223  self.window_interfaces.append(inter)
224  to_remove.append(port)
225 
226  for port in to_remove:
227  self.entity_ports.remove(port)

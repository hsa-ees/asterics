69 def copytree(src: str, dst: str, clobber: bool = True):
70  """! @brief Copy a directory with all files and subdirectories.
71  Not using shutil.copytree, as it raises an error when the destination
72  folder already exists. This function uses shutil.copy to copy files."""
73  for item in os.listdir(src):
74  source = os.path.join(src, item)
75  dest = os.path.join(dst, item)
76  if os.path.isdir(source):
77  os.makedirs(dest, 0o755, exist_ok=True)
78  copytree(source, dest)
79  else:
80  try:
81  copy(source, dest, follow_symlinks=True)
82  except FileExistsError:
83  if clobber:
84  os.remove(dest)
85  copy(source, dest, follow_symlinks=True)
86  else:
87  pass

107 def get_dependencies(
108  chain: AsProcessingChain, module: AsModule, module_list: list
109 ):
110  """! @brief Add module dependencies of 'module' to 'module_list'.
111  Recursively determine the dependencies on other AsModules of 'module',
112  as stored in the attribute AsModule.dependencies.
113  Adds any new dependencies found to the parameter 'module_list'.
114  @param chain: The current AsProcessingChain. Used to retrieve AsModule templates.
115  @param module: The module to analyse the dependencies of.
116  @param module_list: Where to store additional dependencies.
117  """
118  # For all direct dependencies of 'module'
119  for dep in module.dependencies:
120  # Filter new dependencies
121  if dep not in module_list:
122  # Add to the list ...
123  module_list.append(dep)
124  # ... and determine further dependencies of the new dependency
125  dep_mod = chain.library.get_module_template(dep)
126  if dep_mod is None:
127  dep_mod = chain.library.get_module_template(
128  dep, window_module=True
129  )
130  if dep_mod is None:
131  LOG.error(
132  "Module '%s' not found! Required by '%s'.",
133  dep,
134  module.entity_name,
135  )
136  raise AsModuleError(
137  dep,
138  "Module not found!",
139  "Required by '{}'.".format(module.entity_name),
140  )
141  # Let's go down the rabbit hole! Recursion time!
142  get_dependencies(chain, dep_mod, module_list)

334 def write_asterics_h(chain: AsProcessingChain, output_path: str):
335  """! @brief Write the toplevel ASTERICS driver C header
336  The header contains include statements for all driver header files
337  and the definition of the register ranges and base addresses."""
338 
339  LOG.info("Generating ASTERICS main software driver header file...")
340  asterics_h_path = append_to_path(output_path, "/")
341  include_list = set()
342  include_str = ""
343  include_template = '#include "{}" \n'
344 
345  module_base_reg_template = "#define AS_MODULE_BASEREG_{modname} {offset}\n"
346  module_base_addr_template = (
347  "#define AS_MODULE_BASEADDR_{modname} ((uint32_t*)(ASTERICS_BASEADDR + "
348  "(AS_MODULE_BASEREG_{modname} * 4 * AS_REGISTERS_PER_MODULE)))\n"
349  )
350 
351  # Build a list of module additions to asterics.h
352  # Additions should be unique
353  # (two HW instances of an ASTERICS module use the same driver)
354  module_additions = set()
355  for module in chain.modules:
356  module_additions.update(module.get_software_additions())
357  # Format module additions: One line per additional string
358  module_additions = "\n".join(module_additions)
359  module_additions = (
360  "/************************** Module Defines and "
361  "Additions ***************************/\n\n"
362  ) + module_additions
363 
364  # Get list of modules
365  unique_modules = get_unique_modules(chain)
366 
367  # Collect driver header files that need to be included
368  def add_header(mod: AsModule):
369  for driver in mod.driver_files:
370  if driver.endswith(".h"):
371  include_list.add(driver.rsplit("/", maxsplit=1)[-1])
372 
373  for mod in list(ittls.chain(chain.modules, chain.pipelines)):
374  add_header(mod)
375  for entity in unique_modules:
376  mod = chain.library.get_module_template(entity)
377  if mod is not None:
378  add_header(mod)
379  # Build include list
380  include_str = "".join(
381  [include_template.format(i) for i in sorted(include_list)]
382  )
383 
384  # Register base definition list and register range order
385  reg_bases = ""
386  reg_addrs = ""
387  # Generate register definitions
388  for addr in chain.address_space:
389  # Get register interface object
390  regif = chain.address_space[addr]
391  # Build register interface module name
392  regif_modname = regif.parent.name
393  if regif.name_suffix:
394  regif_modname += regif.name_suffix
395  reg_bases += module_base_reg_template.format(
396  modname=regif_modname.upper(), offset=regif.regif_num
397  )
398  reg_addrs += module_base_addr_template.format(
399  modname=regif_modname.upper()
400  )
401 
402  try:
403  with open(asterics_h_path + ASTERICS_H_NAME, "w") as file:
404  # Write asterics.h
405  file.write(
406  ASTERICS_H_TEMPLATE.format(
407  header=ASTERICS_HEADER_SW.format(
408  filename=ASTERICS_H_NAME,
409  description=ASTERICS_H_DESCRIPTION,
410  ),
411  base_addr=chain.asterics_base_addr,
412  regs_per_mod=chain.max_regs_per_module,
413  module_driver_includes=include_str,
414  base_regs=reg_bases,
415  addr_map=reg_addrs,
416  module_additions=module_additions,
417  )
418  )
419 
420  except IOError as err:
421  print(
422  "Couldn't write {}: '{}'".format(
423  asterics_h_path + ASTERICS_H_NAME, err
424  )
425  )
426  raise AsFileError(
427  asterics_h_path + ASTERICS_H_NAME,
428  detail=str(err),
429  msg="Could not write to file",
430  )

433 def write_vivado_package_tcl(
434  chain: AsProcessingChain, output_path: str, additions_c1: str = ""
435 ) -> bool:
436  """! @brief Write two TCL script fragments sourced by the packaging TCL script.
437  This function generates Vivado-specific TCL commands!
438  The first fragment is sourced very early in the packaging script and
439  contains basic information like the project directory,
440  source file directory, etc.
441  The second fragment contains packaging commands setting up the
442  AXI interface configuration.
443  This function needs to access the toplevel modules in the current
444  processing chain to generate the TCL code.
445 
446  Parameters:
447  chain - AsProcessingChain: The current processing chain to build
448  output_path - String: Toplevel folder of the current project
449  Returns a boolean value: True on success, False otherwise"""
450 
451  # Class to manage the AXI interface information for the TCL packaging
452  # script
453  class TCL_AXI_Interface:
454  """'Private' class capsuling methods to write the TCL fragments
455  required for the packaging script for Xilinx targets."""
456 
457  def __init__(self, axi_type: str):
458  # Type of AXI interface ("AXI Slave" / "AXI Master")
459  self.axi_type = axi_type
460  self.refname = "" # Variable name of this interface
461  self.bus_if_name = "" # Name of this bus interface in Vivado TCL
462  self.clock_name = "_aclk" # Name of the associated clock
463  self.reset_name = "_aresetn" # Name of the associated reset
464  self.if_type = "" # AXI slave register address block name (Slave)
465  self.memory_range = 0 # Range of the address block (Slave)
466 
467  def update(self):
468  """Update the clock and reset names
469  after setting the bus interface name"""
470  self.clock_name = self.bus_if_name + self.clock_name
471  self.reset_name = self.bus_if_name + self.reset_name
472 
473  def get_tcl_bus_assoc_commands(self) -> str:
474  """Generate the bus interface association TCL commands"""
475  return BUS_IF_ASSOCIATION_TCL_TEMPLATE.format(
476  busif=self.bus_if_name,
477  clk=self.clock_name,
478  rst=self.reset_name,
479  ref=self.refname,
480  )
481 
482  def get_tcl_mem_commands(self) -> str:
483  """Generate the memory mapping TCL commands"""
484  if any((not self.if_type, not self.memory_range)):
485  LOG.debug(
486  (
487  "Generating TCL packaging script: Interface type "
488  "of slave memory range not set for '%s'!"
489  ),
490  self.bus_if_name,
491  )
492  return ""
493  # Else ->
494  return MEMORY_MAP_TCL_TEMPLATE.format(
495  ref=self.refname,
496  axi_type=self.if_type,
497  mem_range=self.memory_range,
498  busif=self.bus_if_name,
499  mem_map_ref=self.refname + "_mem_ref",
500  )
501 
502  LOG.info("Generating TCL packaging scripts...")
503  # Generate the necessary paths:
504  # IP-Core project directory
505  project_dir = append_to_path(os.path.realpath(output_path), "/")
506  # Subdirectory containing the HDL sources
507  hdl_dir = append_to_path(project_dir, "/hw/hdl/vhdl/")
508 
509  # Populate the fragment files with a generic file header
510  content1 = TCL_HEADER.format(
511  TCL1_NAME, "TCL fragment (1) part of the IP packaging TCL script"
512  )
513  content2 = TCL_HEADER.format(
514  TCL2_NAME, "TCL fragment (2) part of the IP packaging TCL script"
515  )
516  content3 = TCL_HEADER.format(
517  TCL3_NAME, "TCL fragment (2) part of the IP packaging TCL script"
518  )
519 
520  # Generate the commands for the first basic fragment
521  content1 += "\nset projdir " + project_dir
522  content1 += "\nset hdldir " + hdl_dir
523  if additions_c1:
524  content1 += "\n" + additions_c1 + "\n"
525 
526  # Possible AXI interface types
527  templates = ("AXI_Slave", "AXI_Master")
528 
529  for inter in chain.top.interfaces:
530  temp = None
531  if inter.type.lower() in ("axi_slave_external", "axi_master_external"):
532  slave = templates[0] in inter.type
533  if slave:
534  # For AXI slaves: Populate the TCL AXI class
535  temp = TCL_AXI_Interface(templates[0])
536  # And set parameters for the memory map (slave registers)
537  temp.memory_range = 65536
538  temp.if_type = "axi_lite"
539  else:
540  temp = TCL_AXI_Interface(templates[1])
541  temp.refname = (
542  minimize_name(inter.name_prefix)
543  .replace(temp.axi_type.lower(), "")
544  .strip("_")
545  )
546  temp.bus_if_name = inter.ports[0].code_name.rsplit("_", 1)[0]
547 
548  # Set the reference name and update the clock and reset names
549  temp.update()
550  # Generate the bus association commands for all interfaces
551  content2 += temp.get_tcl_bus_assoc_commands()
552  if slave:
553  # Only slaves have a memory map
554  content2 += temp.get_tcl_mem_commands()
555 
556  # Build interface instantiation strings for as_iic modules
557  iic_if_inst_str = ["\n"]
558  for module in chain.modules:
559  if module.entity_name == "as_iic":
560  if_name = module.name
561  top_if = chain.top.__get_interface_by_un_fuzzy__(
562  module.get_interface("in", if_type="iic_interface").unique_name
563  )
564  if top_if is None:
565  LOG.error(
566  (
567  "Was not able to determine port names for IIC "
568  "interface '%s' - IIC interface not found!"
569  ),
570  if_name,
571  )
572  mod_prefix = ""
573  else:
574  mod_prefix = minimize_name(
575  top_if.name_prefix, chain.NAME_FRAGMENTS_REMOVED_ON_TOPLEVEL
576  )
577  iic_if_inst_str.append(
578  "#Instantiate interface for {}\n".format(if_name)
579  )
580  iic_if_inst_str.append(
581  AS_IIC_MAP_TCL_TEMPLATE.format(
582  iic_signal_prefix=mod_prefix, iic_if_name=if_name
583  )
584  )
585  iic_if_inst_str.append(
586  "# END Instantiate interface for {}\n\n".format(if_name)
587  )
588  # Assemble the IIC interface string and add to the TCL fragment
589  content2 += "\n".join(iic_if_inst_str)
590 
591  tcl_ooc_remove_outsourced_template = "set_property is_enabled false -quiet [get_files -quiet -of_objects [get_filesets sources_1] {{{files}}}]\n"
592 
593  outsourced_files = []
594  ooc_modules = [
595  mod for mod in chain.modules if isinstance(mod, AsModuleWrapper)
596  ]
597  if ooc_modules:
598  tcl_ooc_commands = [
599  "# Create OOC blocks\n",
600  'puts "Generating Out-of-Context Synthesis Runs..."\n',
601  ]
602  else:
603  tcl_ooc_commands = []
604  count = 1
605  for mod in ooc_modules:
606  source_files = set()
607  modfilename = mod.name + ".vhd"
608  source_files.add(modfilename)
609  outsourced_files.append(modfilename)
610  source_files.update(mod.modules[0].files)
611  dep_mods = []
612  get_dependencies(chain, mod.modules[0], dep_mods)
613  if dep_mods:
614  for dmod in dep_mods:
615  modtemplate = chain.library.get_module_template(dmod)
616  source_files.update(modtemplate.files)
617  source_files = (
618  sf.rsplit("/", maxsplit=1)[-1] for sf in sorted(source_files)
619  )
620  tcl_ooc_commands.append(
621  TCL_OOC_TEMPLATE.format(
622  ent_name=mod.entity_name,
623  source_files=" ".join(source_files),
624  top_source=mod.name + ".vhd",
625  progress=str(count) + " of " + str(len(ooc_modules)),
626  )
627  )
628  count += 1
629  tcl_ooc_commands.append(
630  tcl_ooc_remove_outsourced_template.format(
631  files=" ".join(outsourced_files)
632  )
633  )
634 
635  content3 += "\n".join(tcl_ooc_commands)
636 
637  # Make sure all files are newline-terminated
638  content1 += "\n"
639  content2 += "\n"
640  content3 += "\n"
641 
642  for name, content in zip(
643  (TCL1_NAME, TCL2_NAME, TCL3_NAME), (content1, content2, content3)
644  ):
645  try:
646  with open(project_dir + name, "w") as file:
647  file.write(content)
648  except IOError as err:
649  LOG.error("Could not write '%s' in '%s'.", name, project_dir)
650  raise AsFileError(
651  project_dir + name, "Could not write file!", str(err)
652  )

655 def run_vivado_packaging(
656  chain: AsProcessingChain, output_path: str, tcl_additions: str = ""
657 ):
658  """! @brief Write the necessary TCL fragments and run the TCL packaging script.
659  Requires Vivado to be installed and in callable from the current terminal.
660  Parameters:
661  chain: The current processing chain.
662  output_path: The root of the output folder structure.
663  No return value."""
664  # Write the necessary tcl fragments

803 def gather_sw_files(
804  chain: AsProcessingChain,
805  output_folder: str,
806  use_symlinks: bool = True,
807  sep_dirs: bool = False,
808 ) -> bool:
809  """! @brief Copy or link to software files for an ASTERICS chain.
810  Collect all available software driver files in 'drivers' folders
811  of the module source directories in the output folder structure.
812  @param chain: The current AsProcessingChain. Source of the modules list.
813  @param output_folder: The root of the output folder structure.
814  @param use_symlinks: Whether or not to link (True) or copy (False) files
815  @param sep_dirs: Whether or not to generate separate directories per module driver
816  @return True on success, False otherwise."""
817 
818  LOG.info("Gathering ASTERICS module software driver source files...")
819  out_path = os.path.realpath(output_folder)
820  # We don't want the trailing slash if we're not using separate folders
821  out_path = append_to_path(out_path, "/", sep_dirs)
822  # Collect all module entity names
823  unique_modules = get_unique_modules(chain)
824  modules = list(ittls.chain(chain.modules, chain.pipelines))
825  handled_entities = []
826 
827  # Initialize driver list with asterics support package drivers
828  drivers = [
829  (
830  "as_support",
831  list(
832  asp_driver.format(asterics_root=asterics_home)
833  for asp_driver in ASP_FILES
834  ),
835  )
836  ]
837 
838  for mod in modules:
839  if mod.entity_name not in handled_entities:
840  drivers.append((mod.entity_name, mod.driver_files))
841  handled_entities.append(mod.entity_name)
842 
843  for entity in filter(
844  lambda ent: ent not in handled_entities, unique_modules
845  ):
846  module = chain.library.get_module_template(entity)
847  drivers.append((module.entity_name, module.driver_files))
848 
849  driver_dir = {}
850  for entity, paths in drivers:
851  if not paths:
852  continue
853  try:
854  driver_dir[entity].extend(paths)
855  except KeyError:
856  driver_dir[entity] = paths
857 
858  for entity in driver_dir:
859  driverlist = driver_dir[entity]
860  if sep_dirs:
861  # Build destination path: out + entity name (cut off trailing '/')
862  dest_path = append_to_path(out_path, entity, False)
863  else:
864  # If drivers should not be stored in separate directories
865  dest_path = out_path
866  # Linking / copying the driver files
867  # Get mode of operation
868  mode = "link" if use_symlinks else "copy"
869  copy_op = os.symlink if use_symlinks else copy
870 
871  for driverfile in driverlist:
872  filename = driverfile.rsplit("/", maxsplit=1)[-1]
873  dest_file = append_to_path(dest_path, filename, False)
874  if os.path.exists(dest_file):
875  os.unlink(dest_file)
876  try:
877  copy_op(driverfile, dest_file)
878  except IOError as err:
879  LOG.critical(
880  ("Could not %s driver file '%s' of module '%s'!" " - '%s"),
881  mode,
882  filename,
883  entity,
884  str(err),
885  )
886  return False
887  return True

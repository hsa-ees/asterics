84 def generate_adder_layout(max_stages, filter_weights):
85  elem_count = get_total_elements_for_filter(filter_weights)
86  elements_per_stage = [elem_count, elem_count // 4 + elem_count % 4]
87  elements_per_adder = [4]
88  adders_per_stage = [elem_count // 4]
89  stage_count = 1
90  while True:
91  cur_elems = elements_per_stage[stage_count]
92  if get_csa_adder_stages(cur_elems) <= max_stages:
93  elements_per_adder.append(cur_elems)
94  adders_per_stage.append(1)
95  elements_per_stage.append(1)
96  else:
97  elements_per_adder.append(elements_for_csa_with_stages(max_stages))
98  adders_per_stage.append(
99  cur_elems // elements_per_adder[stage_count]
100  )
101  elements_per_stage.append(
102  adders_per_stage[stage_count]

104  )
105  stage_count += 1
106  if elements_per_stage[-1] == 1:
107  break

169 def reduce_add_sub(weights: list, max_ones: int) -> list:
170  """Reduce the number of PoT factors required per weight up to 'max_ones'.
171  Returns the modified 'weights'."""
172  out = []
173  count = 0
174  for w in weights:
175  ones = ones_lut[abs(w)]
176  if ones > max_ones:
177  count += 1
178  pos = w
179  neg = w
180  while True:
181  pos += 1
182  neg -= 1
183  if ones_lut[abs(pos)] <= max_ones:
184  out.append(pos)
185  break
186  if ones_lut[abs(neg)] <= max_ones:
187  out.append(neg)
188  break
189  else:
190  out.append(w)
191  LOG.info("Transformed {} weights!".format(count))
192  return out

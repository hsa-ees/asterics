74  def register_error(self, err):
75  """! @brief Called when a new error object is generated."""
76  try:
77  self.error_count[err.type] += 1
78  except KeyError:
79  self.error_count[err.type] = 1
80  try:
81  self.error_severities[err.severity] += 1
82  except KeyError:
83  self.error_severities[err.severity] = 1
84  self.errors.append(err)

86  def clear_error(self, err):
87  try:
88  self.error_severities[err.severity] -= 1
89  self.error_count[err.type] -= 1
90  self.errors.remove(err)
91  except KeyError:
92  pass
93  except ValueError:
94  self.error_severities[err.severity] += 1
95  self.error_count[err.type] += 1

97  def has_errors(self, severity: str = "Error") -> bool:
98  """! @brief Returns whether errors have been encountered.
99  Use 'severity' to exclude errors below the passed severity."""
100  if not self.errors:
101  return False
102  if severity not in SEVERITIES:
103  raise ValueError("Invalid error severity '{}'".format(severity))
104  sev_id = next(
105  (
106  idx
107  for idx in sorted(SEVERITY_MAP.keys())
108  if SEVERITY_MAP[idx] == severity
109  )
110  )
111  for idx in range(sev_id, 0, -1):
112  try:
113  if self.error_severities[SEVERITY_MAP[idx]]:
114  return True
115  except KeyError:
116  pass
117  return False
118 
119  def has_specific_error(
120  self, err_type: str = "", severity: str = ""
121  ) -> bool:
122  """! @brief Returns whether specific errors have ocurred.
123  Allows checking whether errors of a specific type and/or severity
124  have occurred."""
125  if err_type and severity:
126  return bool(
127  [
128  err
129  for err in self.errors
130  if err.type == err_type and err.severity == severity
131  ]
132  )
133  elif err_type:
134  return True if self.get_error_count(err_type) > 0 else False
135  elif severity:
136  return (
137  True if self.get_error_severity_count(severity) > 0 else False
138  )
139  # Else ->
140  return bool(len(self.errors))
141 
142  def get_error_count(self, err_type: str = "") -> int:
143  if err_type:
144  try:
145  return self.error_count[err_type]
146  except KeyError:
147  if err_type in ERROR_TYPES:
148  return 0
149  else:
150  LOG.error(
151  "Invalid error type querying AsErrorManager! '%s'",
152  err_type,
153  )
154  return -1
155  # Else ->
156  return len(self.errors)
157 
158  def get_error_severity_count(self, severity: str) -> int:
159  try:
160  return self.error_severities[severity]
161  except KeyError:
162  if severity in SEVERITIES:
163  return 0
164  else:
165  LOG.error(
166  ("Invalid error severity querying AsErrorManager!" " '%s'"),
167  severity,
168  )
169  return -1
170 
171  def get_errors(self, err_type: str = "") -> list:
172  if err_type:
173  return [err for err in self.errors if err.type == err_type]
174  # -> Else
175  return self.errors

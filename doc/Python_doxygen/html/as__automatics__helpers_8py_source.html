78 def get_prefix_suffix(
79  port_name: str, code_name: str, ignored_keywords: Sequence[str]
80 ) -> [str, str]:
81  """! @brief Extract the prefix and suffix from a code name of a VHDL port.
82  removes the 'static' port name and filters out the
83  'ignored_keywords'. Assumes that the name fragments of the code name
84  are delimited with '_'."""
85  # Remove the port name
86  try:
87  prefix, suffix = code_name.split(port_name, maxsplit=1)
88  except ValueError:
89  temp = code_name.split(port_name, maxsplit=1)
90  namepos = code_name.find(port_name)
91  if namepos > 0:
92  prefix = temp[0]
93  suffix = ""
94  else:
95  prefix = ""
96  suffix = temp[0]
97 
98  # Remove excess '_' and split the rest of the name
99  prefix_list = prefix.strip("_").split("_")
100  suffix_list = suffix.strip("_").split("_")
101  if suffix_list:
102  suffix_list.reverse()
103  prefix = suffix = ""
104 
105  # Filter out ignored keywords and rebuild the prefix and suffix
106  for string in prefix_list:
107  if string not in ignored_keywords and string != "":
108  prefix = prefix + string + "_"
109  for string in suffix_list:
110  if string not in ignored_keywords and string != "":
111  suffix = "_" + string + suffix
112 
113  return [prefix, suffix]

152  built-in functions (we only want to resolve mathematic expressions)."""

179 def extract_generics(data_width) -> Sequence[str]:
180  """! @brief Extract generic names from the data width tuple or a string."""
181  if isinstance(data_width, str):
182  temp = data_width.split("downto")
183  dwstr = []
184  for tstr in temp:
185  dwstr.extend(tstr.split("to"))
186  dwstr = "".join(dwstr)
187  else:
188  if not data_width.sep:
189  return ""

191 
192  to_replace = " -+*/^()<>,"

231 def append_to_path(path: str, to_append: str, add_trailing_slash: bool = True):
232  """! @brief Take two path fragments and merge them.
233  This function makes sure that each directory is separated by a '/'
234  and returns the path with a '/' at the end
235  unless 'add_slash' is set to False."""
236  # Merge to one string
237  string = path + "/" + to_append + "/"
238  # Split it and filter out empty strings resulting from subsequent "/"es
239  strsplit = [s for s in string.split("/") if s != ""]
240  # Join the path separating each string by a single "/"
241  out = "/".join(strsplit) + ("/" if add_trailing_slash else "")
242  if string[0] == "/": # Add a leading "/" if the initial string had one
243  return "/" + out
244  return out

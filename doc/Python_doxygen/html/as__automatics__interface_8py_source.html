115  def duplicate(self):
116  """! @brief Generate a copy of this interface object reseting some values.
117  Generate a copy of this interface object and reset a few parameters
118  to default values. Returns a 'cleaned' version of this object to use."""
119 
120  # First, create a low-level copy of this interface
121  out = copy.copy(self)
122  # Reset connection data
123  out.incoming = []
124  out.outgoing = []
125 
126  # Copy all ports
127  out.ports = []
128  for port in self.ports:
129  out.ports.append(port.duplicate())
130 
131  # and Generics
132  out.generics = []
133  for gen in self.generics:
134  out.generics.append(copy.copy(gen))
135 
136  # Reset connection state, parent, connect_to
137  out.connected = False
138  out.parent = None
139  connect_to = getattr(self, "connect_to", None)
140  if connect_to:
141  setattr(out, "connect_to", connect_to)
142 
143  # Reset connection data and state of all ports
144  for port in out.ports:
145  port.glue_signal = None
146  port.connected = False
147  port.incoming = None
148  port.outgoing = []
149  port.assign_to(out)
150  return out

152  def add_port(self, port_obj: Port) -> bool:
153  """! @brief Add a port to this interface instance.
154  Performs some checks on whether the handed port object
155  may be added to this interface: Duplicate ports aren't allowed.
156  If the interface has a template, the port must be part of it
157  and have the matching direction."""
158  # Make sure we got a port object
159  if not isinstance(port_obj, Port):
160  LOG.error(
161  "'add_port' was passed the non-Port object '%s'.",
162  repr(port_obj),
163  )
164  return False
165 
166  # Check if a port with the same name is already assigned
167  if self.has_port(port_obj.name):
168  LOG.debug(
169  "Port '%s' already present in interface '%s'",
170  port_obj.name,
171  str(self),
172  )
173  return False
174 
175  # Checks for when a template is set
176  if self.template is not None:
177  # Check if this port is in the template
178  tport = self.template.get_port(port_obj.name, suppress_error=True)
179  add = bool(tport)
180  if not add:
181  LOG.debug(
182  "Port '%s' not found in template '%s'",
183  port_obj.name,
184  str(self.template),
185  )
186  return False
187 
188  if tport.data_type != port_obj.data_type:
189  LOG.debug("Data type mismatch for port '%s'.", port_obj.name)
190  return False
191  # Also make sure the port direction matches
192  add = tport.direction == port_obj.direction
193  if self.direction == "out":
194  # If this interface's direction is out,
195  # reverse the port direction (and set add = !add)
196  add = not add
197  if add:
198  port_obj.direction = tport.direction
199  if not add:
200  LOG.debug("Port direction mismatch!")
201  return False
202  port_obj.optional = tport.optional
203  port_obj.set_ruleset(list(tport.ruleset))
204  port_obj.in_entity = tport.in_entity
205 
206  # If the checks passed, add this port
207  LOG.debug(
208  "Add port '%s' to interface '%s'.", port_obj.code_name, str(self)
209  )
210  port_obj.assign_to(self)
211  port_obj.port_type = "interface"
212  self.ports.append(port_obj)
213  return True

413  def fit_and_add_port(self, port_obj: Port) -> bool:
414  """! @brief Adds a Port to this interface if it matches and is in the template.
415  Use if only the code_name of the port is available.
416  Finds the longest matching name fragment of the code_name with the
417  names of the ports of this interfaces template. This is used to set
418  the ports name (port.name). Also compares the pre- and suffixes
419  of the interface and port to add"""
420  # Make sure we got a port object
421  if not isinstance(port_obj, Port):
422  LOG.error(
423  "'fit_and_add_port' got the non-Port object '%s'.",
424  repr(port_obj),
425  )
426  return False
427 
428  # This method can only work if the template is set
429  if self.template is None:
430  LOG.error(
431  (
432  "'fit_and_add_port' called for port '%s' while "
433  "interface '%s' had no template assigned!"
434  ),
435  port_obj.code_name,
436  str(self),
437  )
438  return False
439 
440  # Check if the port is defined in this interfaces template
441  matchlist = [
442  tport.name
443  for tport in self.template.ports
444  if tport.name in port_obj.code_name
445  ]
446  if not matchlist:
447  LOG.debug("Port '%s' not found in template.", port_obj.code_name)
448  return False
449 
450  # Find the longest matching template port -> set the ports name to that
451  matchlist.sort(key=lambda name: len(name), reverse=True)
452  port_obj.name = matchlist[0]
453 
454  # Pre-/Suffix check
455  this_prefix, this_suffix = as_help.get_prefix_suffix(
456  port_obj.name, port_obj.code_name, Port.directions
457  )
458  first_port = not self.ports
459  if not first_port:
460  # If this is not the first port, the pre-/suffix need to match
461  if (this_suffix != self.name_suffix) or (
462  this_prefix != self.name_prefix
463  ):
464  LOG.debug(
465  "Port '%s' didn't fit, prefix/suffix mismatch!",
466  port_obj.code_name,
467  )
468  return False
469  else:
470  # If this is the first port, use its pre-/suffix for this regif
471  self.name_prefix = this_prefix
472  self.name_suffix = this_suffix
473 
474  LOG.debug(
475  "Trying to add '%s' to interface '%s'",
476  port_obj.code_name,
477  str(self),
478  )
479  # Try adding the port
480  if not self.add_port(port_obj):
481  return False
482  return True

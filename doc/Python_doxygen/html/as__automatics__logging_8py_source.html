52 def init_log(
53  logfilename: str = "as_automatics.log",
54  *,
55  loglevel_console="WARNING",
56  loglevel_file="INFO"
57 ):
58  """! @brief Initialize logging and return the logger.
59  Should only be run once per kernel."""
60  # Setup logging
61  logger = logging.getLogger("as_automatics")
62 
63  ll_console = getattr(logging, loglevel_console, logging.INFO)
64  ll_file = getattr(logging, loglevel_file, logging.INFO)
65 
66  logger.setLevel(min((ll_console, ll_file)))
67  logger.disabled = False
68 
69  # Make sure only the handlers we want are instantiated
70  # (Not sure if this is a particularly 'neat' implementation...)
71  if logger.hasHandlers():
72  # "Nuke" all references to old handlers
73  logger.handlers.clear()
74 
75  # Set format for log entries
76  formatter_logfile = logging.Formatter(
77  "%(asctime)s, %(module)s: %(levelname)s - %(message)s"
78  )
79  formatter_console = logging.Formatter(
80  "Automatics %(levelname)s: %(message)s"
81  )
82 
83  # Setup logfile log handler
84  logfile = logging.FileHandler(logfilename)
85  logfile.setLevel(ll_file)
86  logfile.setFormatter(formatter_logfile)
87  logger.addHandler(logfile)
88 
89  # Setup console log handler
90  logstream = logging.StreamHandler()
91  logstream.setLevel(ll_console)
92  logstream.setFormatter(formatter_console)
93  logger.addHandler(logstream)
94  return logger

272  """! @brief Generic interface and port search method.

274  @param direction The interface's direction. ! Does not apply for ports !
275  @param if_type The type of interface to search for. ! Does not apply for ports !

490  def get_interface(
491  self,
492  interface_name: str,
493  direction: str = "",
494  if_type: str = "",
495  *,
496  suppress_error: bool = False
497  ) -> Interface:
498  """! @brief Return an interface instance of name 'interface_name'.
499  This method supports filters in cases where multiple interfaces with
500  the same name exist. Throws an error if no matching interface is found.
501  Parameters:
502  interface_name: This is the name to search for. NOT the interface's type.
503  direction: The data direction of the interface to find.
504  if_type: The type of the interface. Eg.: as_stream
505  suppress_error: For internal use. Method will not throw errors.
506  Returns 'None' if no interface with the specified name exists."""
507  # If both direction and type are unspedified
508  if direction == "" and if_type == "":
509  found = next(
510  (
511  inter
512  for inter in self.interfaces
513  if str(inter) == interface_name
514  ),
515  None,
516  )
517  else: # Filtered search
518  # Filter by name
519  found = [
520  inter
521  for inter in self.interfaces
522  if str(inter) == interface_name
523  ]
524  # Filter by direction, if type unspecified
525  if found and if_type == "":
526  found = next(
527  (inter for inter in found if inter.direction == direction),
528  None,
529  )
530  # Filter by type, if direction unspecified
531  elif found and direction == "":
532  found = next(
533  (inter for inter in found if inter.type == if_type), None
534  )
535  # Filter by both type and direction
536  elif found:
537  found = next(
538  (
539  inter
540  for inter in found
541  if inter.type == if_type
542  and inter.direction == direction
543  ),
544  None,
545  )
546  # "Interface not found" error report
547  if not found and not suppress_error:
548  raise AsNameError(interface_name, self, "Could not find interface!")
549  return found

1084  def __name_interfaces__(self):
1085  """! @brief Names interfaces of this module.
1086  Names are chosen according to the following rules:
1087  If the interface has a unique prefix and/or suffix,
1088  only the prefix and/or suffix are used as the name.
1089  If multiple interfaces with no prefix and or suffix
1090  but different types are present, the type is added as a prefix.
1091  Interfaces of the same type and direction cannot exist witht he same
1092  prefix and suffix, as their VHDL names would be identical.
1093  If no prefix and suffix and only one interface maximum per direction
1094  is present, only the data direction is used as a name ("in", "out").
1095  Slave register interfaces always use the type in their name and are
1096  handled by a different function.
1097  """
1098 
1099  # Helper function to left-rotate a list by a fixed number of items
1100  def rotate(l: list, n: int) -> list:
1101  return l[n:] + l[:n]
1102 
1103  # "Worker" function for the naming function. Names a single interface
1104  # Uses a list of all remaining interfaces to make the decision
1105  # If the choice for the current interface is ambiguos, the interface is
1106  # not named, instead 'None' is returned and the interface list rotated
1107  # left by one, leaving the interface as the last to handle
1108  # All interfaces *should* always be able to be named deterministically.
1109  def handler(interfaces: list) -> list:
1110  crt_if = interfaces[0] # First interface in list will be named
1111  # Unique names through suffix and/or prefix
1112  # Both suffix and prefix
1113  if crt_if.name_prefix != "" and crt_if.name_suffix != "":
1114  crt_if.name = (
1115  crt_if.name_prefix.rsplit("_", maxsplit=1)[0]
1116  + "_"
1117  + crt_if.name_suffix.split("_", maxsplit=1)[-1]
1118  )
1119  return [crt_if]
1120  # Only prefix
1121  if crt_if.name_prefix != "":
1122  crt_if.name = crt_if.name_prefix.rsplit("_", maxsplit=1)[0]
1123  return [crt_if]
1124  # Only suffix
1125  if crt_if.name_suffix != "":
1126  crt_if.name = crt_if.name_suffix.split("_", maxsplit=1)[-1]
1127  return [crt_if]
1128 
1129  # Neither prefix nor suffix are present
1130  # Determine number of interfaces with the same type
1131  others = [
1132  inter for inter in interfaces if inter.type == crt_if.type
1133  ]
1134  if len(others) > 2: # More than two: Filter by direction
1135  others = [
1136  inter
1137  for inter in others
1138  if inter.direction == crt_if.direction
1139  ]
1140  if len(others) == 1: # Exactly one: name using direction
1141  crt_if.name = crt_if.direction
1142  return [crt_if]
1143  if len(others) == 2: # Exactly two:
1144  other_if = interfaces[
1145  1
1146  ] # Use the next interface for comparison
1147  # If direction differs: name both by their direction
1148  if crt_if.direction != other_if.direction:
1149  crt_if.name = crt_if.direction
1150  other_if.name = other_if.direction
1151  elif other_if.name_prefix != "" or other_if.name_suffix != "":
1152  crt_if.name = crt_if.direction
1153  other_if.name = other_if.name_prefix + other_if.name_suffix
1154  other_if.name = other_if.name.replace("__", "_").strip("_")
1155  other_if.name += "_" + other_if.direction
1156  else: # Add the type
1157  crt_if.name = crt_if.type + "_" + crt_if.direction
1158  other_if.name = other_if.type + "_" + other_if.direction
1159  return [crt_if, other_if]
1160  return None
1161 
1162  # Working copy of interface list
1163  interfaces = copy.copy(self.interfaces)
1164  while interfaces: # Run until all interfaces have been named
1165  to_remove = handler(interfaces) # Name zero to two interfaces
1166  if to_remove is None: # No interface named
1167  # Move current (first) interface to back of list
1168  interfaces = rotate(interfaces, 1)
1169  continue
1170  # Otherwise, remove the named interfaces from the list
1171  for inter in to_remove:
1172  interfaces.remove(inter)

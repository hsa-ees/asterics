153  def add_register(self, register_type: Register):
154  """! @brief Add a register to this module group.
155  Use asterics.Register.<type>!
156  This is automatically called when using 'assign_port_to_register' or
157  'assign_register_to_port' on an index that has no register associated.
158  @param register_type The type/data direction of the new register
159  """

200  def modify_register_type(
201  self, register_num: int, new_type: Register
202  ) -> bool:
203  """! @brief Modify the type/data direction of an existing register.
204  Does normally not have to be called directly, instead is indirectly
205  executed when assign_port_to_register or assign_register_to_port is
206  called.
207  @param register_num The offset of the register to modify
208  @param new_type The register type (asterics.Register.<type>) to change to
209  @return True on success, else False (register number does not exist)
210  """
211  if self.register_count <= register_num:
212  return False
213  old_type = self.register_config[register_num]
214  new_type = REGMGR_REGISTER_CONFIG_NAMES[new_type.name]
215 
216  if old_type != new_type:
217  self.regmod.assign_to_this_vector(None, register_num)
218  if new_type in ("AS_REG_NONE", "AS_REG_CONTROL"):
219  self.regmod.assign_to_this_vector(ZERO_SIGNAL, register_num)
220  elif new_type == "AS_REG_STATUS":
221  self.regmod.assign_to_this_vector(ONE_SIGNAL, register_num)
222  elif new_type == "AS_REG_BOTH":
223  self.regmod.assign_to_this_vector(DC_SIGNAL, register_num)
224  self.register_config[register_num] = new_type
225  self._update_register_params()
226  return True

398  def assign_port_to_register(
399  self, register_num: int, port: Port, to_bit_index: int
400  ) -> bool:
401  """! @brief Assign a Port as a part of a register as a data sink.
402  @param register_num Define the number of the register to assign to
403  @param port The Port to assign to the register
404  @param to_bit_index The bit offset to assign to.
405  register bit map: (31 downto 0)
406  """
407  if to_bit_index > 31 or to_bit_index < 0:
408  LOG.error("Index for register assignment is out of bounds!")
409  raise ValueError(to_bit_index, "Out of bounds!", port)
410  # If the register index does not have a register associated, add them
411  while self.register_count <= register_num:
412  if register_num == self.register_count:
413  self.add_register(Register.status)
414  else: # Empty registers for empty spots
415  self.add_register(Register.none)
416  # Update register type if necessary
417  if self.register_config[register_num] == "AS_REG_NONE":
418  self.modify_register_type(register_num, Register.status)
419  elif self.register_config[register_num] == "AS_REG_CONTROL":
420  self.modify_register_type(register_num, Register.both)
421 
422  # Get or create the register assignment signal
423  try:
424  reg_signal = self.reg_assign_map_status[register_num]
425  except KeyError:
426  reg_signal = self.define_signal(
427  "s_register_id{}_status".format(register_num),
428  data_type="std_logic_vector",
429  data_width=Port.DataWidth(31, "downto", 0),
430  )
431  self._assign_to_register(register_num, reg_signal)
432 
433  # Get or create the source signal
434  if isinstance(port, GenericSignal):
435  source = port
436  else:
437  self.__update_generics_list__()
438  port.glue_signal = self.define_signal_based_on_port(port)
439  self.chain.connect(port, port.glue_signal)
440  source = port.glue_signal
441  # Assign the source signal to the register assignment signal
442  reg_signal.assign_to_this_vector(source, to_bit_index)

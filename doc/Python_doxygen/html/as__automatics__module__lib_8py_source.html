221  def get_module_template(
222  self, module_name: str, repo_name: str = "", window_module: bool = None
223  ) -> AsModule:
224  """! @brief Return a module template from the AsModuleLibrary.
225  Search for and return a reference to the AsModule with the entity
226  name 'module_name' in the AsModuleLibrary.
227  Alternatively search only in module repository 'repo_name'.
228  @note Do NOT add the return value to an AsProcessingChain!
229  @note To get an AsModule to use in a chain use 'get_module_instance' instead!
230 
231  @param module_name: The module name to search for.
232  Modules are named after their toplevel VHDL entity.
233  @param repo_name: Optional. Search only in module repository 'repo_name'.
234  @param window_module: Optional. Search only among window modules (True),
235  regular modules (False), or both (omitted).
236  @return A reference to the matching module, 'None' if no match is found.
237  """
238  repo = None
239  module_name = module_name.lower()
240  # If no reponame was defined, check all repos
241  # Select the first repo containing the module
242  if repo_name == "":
243  for irepo in self.repos:
244  # Differentiate between regular and window modules
245  if window_module is None:
246  found = irepo.has_module_generic(module_name)
247  elif window_module:
248  found = irepo.has_window_module(module_name)
249  else:
250  found = irepo.has_module(module_name)
251  if found:
252  repo = irepo
253  break
254  else:
255  # If a specific repo is given, get that
256  repo = self.get_repo(repo_name)
257  if repo is None:
258  return None # Error if it doesn't exist
259  # Get the module
260  if window_module is None:
261  return repo.get_module_generic(module_name)
262  if window_module:
263  return repo.get_window_module(module_name)
264  return repo.get_module(module_name)

268  ) -> AsModule:
269  """! @brief Return a copy of a module template from the AsModuleLibrary.
270  Search for and return a copy to the AsModule with the entity
271  name 'module_name' in the AsModuleLibrary.
272  Alternatively search only in module repository 'repo_name'.
273  @note Only use this function if you want to modify the module object.
274  @note Else you might want to use 'get_module_template', for better performance

276  @param module_name: The module name to search for.

281  @return A copy to the matching module object, 'None' if no match found.
282  """
283  template = self.get_module_template(

390  @classmethod
391  def __get_module_scripts_in_dir__(cls, module_dir: str) -> Sequence[str]:
392  scripts = []
393  module_dir = os.path.realpath(module_dir)
394  try:
395  mod_folders = os.listdir(append_to_path(module_dir, ""))
396  except IOError:

405  if not os.path.isdir(folder_path):
406  continue

408  if not os.path.isdir(script_path):

411  if cls.__script_name_valid__(file):
412  scripts.append((folder_path, script_path + file))

415  @classmethod
416  def __get_modules_from_dir__(cls, module_dir: str) -> Sequence[AsModule]:
417  # Make sure the module path is valid syntactically and ends in a "/"
418  module_dir = append_to_path(module_dir, "/")
419  module_list = []
420  # Get all module initialization scripts
421  script_list = cls.__get_module_scripts_in_dir__(module_dir)
422 
423  for script in script_list:
424  module_folder = script[0]
425  script_path = script[1]
426  script_name = script_path.rsplit("/", maxsplit=1)[-1]
427 
428  # For each valid script: run the contained function
429  # 'get_module_instance'
430  LOG.debug("Modlib importing script '%s' ...", script_name)
431  # Get Python module spec
432  spec = importutil.spec_from_file_location(script_name, script_path)
433  # Get Python module and run / load it
434  imported_script = importutil.module_from_spec(spec)
435  spec.loader.exec_module(imported_script)
436  LOG.debug(
437  "Modlib calls 'get_module_inst' of script '%s'", script_name
438  )
439  # Execute the function "get_module_instance"
440  module_inst = imported_script.get_module_instance(module_folder)
441  LOG.debug(
442  "Modlib received '%s' from script '%s'",
443  str(module_inst),
444  script_name,
445  )
446  # If the output is an AsModule, add it to the library
447  if isinstance(module_inst, AsModule):
448  # Add the module source dir, making sure it
449  module_inst.module_dir = module_folder
450  module_list.append(module_inst)
451  # Discover driver files for this module:
452  # If this module already has files manually assigned,
453  # don't scan default location
454  if not module_inst.driver_files:
455  # Find files in default location
456  module_inst.driver_files = get_software_drivers_from_dir(
457  append_to_path(module_folder, cls.DRIVER_FOLDER)
458  )
459  else:
460  # If files are manually assigned, normalize path format
461  module_inst.driver_files = [
462  os.path.realpath(df) for df in module_inst.driver_files
463  ]
464  return module_list

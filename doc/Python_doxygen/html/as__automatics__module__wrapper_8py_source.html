59 class AsModuleWrapper(AsModuleGroup):
60  """! @brief Class encapsulating an AsModule to manage generating an
61  out-of-context Vivado run for it.
62  This feature will generate a managed wrapper VHDL component containing only
63  the interface for the wrapped ASTERICS module and instantiates it within the
64  wrapper entity.
65  This is necessary, as only standard std_logic data types are supported within
66  interfaces and only component instantation is supported for out-of-context runs.
67  @warning This is feature only compatible with Xilinx Vivado.
68  This feature is not fully tested and considered an alpha feature.
69  Using this will not allow Automatics to generate an IP-Core that
70  can be used directly within block designs! The resulting IP-Core must first
71  be opened manually as a Vivado project and synthesized before it can be used
72  in other projects!"""

128  def convert_window_port_to_vector(self) -> list:
129  """! @brief Convert as_window to only use standard data types.
130  If this module wrapper contains a window module, the window port's
131  data type must be converted from t_generic_window to std_logic_vector.
132  Generates the conversion code and adds the necessary signals within
133  the wrapper VHDL file (conversion from vector to window type) and
134  converts the port data type."""
135  # If we wrap a window module, we need to convert t
136  if not isinstance(self.modules[0], AsWindowModule):
137  return None
138  mod = self.modules[0]
139  window_ports = []
140  conversion_template = (
141  "f_convert_vector_to_generic_window({vector_name}, {x}, {y})"
142  )
143  # Xilinx Vivado OOC Synthesis can't deal with custom data types
144  # For all window interfaces, take the window port using the
145  # t_generic_window data type and convert it to a std_logic_vector
146  # Within this wrapper, take the vectorized port and convert it back
147  # to a t_generic_window port to pass to the filter module.
148  # The conversion from window to vector is done in the 2D Pipeline class
149  for winter in getattr(self, "window_interfaces"):
150  wport = winter.window_port
151  window_dims = winter.window
152  window_elements = window_dims.x * window_dims.y
153  new_vector_width = (
154  window_elements * wport.data_width.get_bit_width()
155  )
156  wport.data_type = "std_logic_vector"
157  wport.data_width = Port.DataWidth(new_vector_width - 1, "downto", 0)
158  mport = mod.get_port(wport.code_name, suppress_error=True)
159  sig = self.define_signal(
160  mport.code_name + "_reconv",
161  data_type=mport.data_type,
162  data_width=mport.data_width,
163  fixed_value=conversion_template.format(
164  vector_name=wport.code_name,
165  x=window_dims.x,
166  y=window_dims.y,
167  ),
168  )
169  sig.window_config = wport.window_config
170  wport.window_config = None
171  mport.incoming = sig
172  window_ports.append(wport)
173  return window_ports

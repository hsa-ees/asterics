134  def get_hash(self) -> str:
135  """! @brief Provide a SHA256 hash based on the configuration of this processing chain.
136  Can be used to identify this ASTERICS processing chain."""
137  hashgen = sha256()
138  string = [self.parent.version]
139  for mod in self.modules:
140  string.append(mod.name + mod.entity_name)
141  for gen in mod.generics:
142  string.append(gen.code_name + str(gen.get_value()))
143  for modg in self.module_groups:
144  for gen in mod.generics:
145  string.append(gen.code_name + str(gen.get_value()))
146  for sig in modg.signals:
147  string.append(sig.code_name)
148  for mod in self.top.modules:
149  string.append(mod.name + mod.entity_name)
150  string = "".join(string)
151  hashgen.update(string.encode())
152  return hashgen.hexdigest()

393  self,
394  entity_name: str,
395  module_name: str = "",

399  ) -> AsModule:
400  """! @brief Add an ASTERICS module to this processing chain.

402  as_automatics_module_lib::AsModuleLibrary to this instance

404  If none is provided, a consecutive number is appended to the
405  standard module name (the name of the VHDL entity).
406  @param entity_name The entity name of the module to add

408  @param repo_name The repository to search for the module in (optional)
409  @return A reference to the newly added module.
410  """
411  LOG.debug(

413  entity_name,
414  module_name,

417  # TODO: Test support for module groups once that feature is implemented
418  if group is None:
419  group = self.as_main

422  entity_name = entity_name.lower()
423  # Get the reference for the module to add from the library
424  module = self.library.get_module_instance(entity_name, repo_name)

486  def auto_instantiate(self) -> Sequence[AsModule]:
487  """! @brief Add modules to the processing chain defined by interfaces.
488  Check all interfaces of all modules for the 'instantiate_in_top'
489  attribute. Adds the modules referenced by this attribute to the
490  system toplevel module group."""
491 
492  out = []
493  if self.auto_inst_done:
494  return out
495  self.auto_inst_done = True
496 
497  # For all modules
498  for module in ittls.chain(self.modules, self.pipelines, [self.as_main]):
499  # Check each interface
500  for inter in module.interfaces:
501  # Is 'instantiate_in_top' set?
502  if inter.instantiate_in_top is None:
503  continue
504  # Get the referenced module group:
505  gmod_name = inter.instantiate_in_top[1]
506  if gmod_name == "":
507  gmod = self.top
508  else:
509  gmod = self.get_module_group(gmod_name)
510  if gmod is None:
511  err = "Could not find module group '{}'!".format(
512  gmod_name
513  )
514  LOG.error(err)
515  raise AsModuleError(err)
516  inst_mod = inter.instantiate_in_top[0]
517 
518  # If yes: Add the referenced module to the system toplevel
519  ret = self.add_module(
520  inst_mod, "{}_{}".format(module.name, inst_mod), group=gmod
521  )
522  # Error message if the module wasn't found
523  if ret is None:
524  LOG.error(
525  (
526  "Could not find module '%s' set as "
527  "automatic toplevel instantiation for "
528  "interface '%s' of module '%s'!"
529  ),
530  inst_mod,
531  inter.name,
532  module.name,
533  )
534  continue

536  ret.auto_inst_config(ret, module)
537  LOG.debug("Ran autoconfig function for '%s'", repr(ret))
538  except AttributeError:
539  pass # OK if function doesn't exist -> no special config
540  out.append(ret)
541  LOG.info(
542  (
543  "Automatically instantiated module '%s' to '%s' "
544  "from interface '%s' of module '%s'"

547  gmod.name,
548  inter.unique_name,
549  module.name,
550  )

552  if iinter.to_external:

556

576  # For module groups below "as_main" and their modules:
577  # Delegate connections to a Module Group if it has a connect method
578  try:
579  if (
580  isinstance(source_parent, AsModuleGroup)

584  elif (
585  isinstance(sink_parent, AsModuleGroup)
586  and sink_parent.modlevel > 1
587  ):

589  elif (
590  isinstance(source_parent.parent, AsModuleGroup)
591  and source_parent.modlevel > 2

595  isinstance(sink_parent.parent, AsModuleGroup)
596  and sink_parent.modlevel > 2
597  ):
598  sink_parent.parent.connect(source, sink)

602  self.user_cons.append((source, sink, top))

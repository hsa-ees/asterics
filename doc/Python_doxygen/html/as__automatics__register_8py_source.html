221  def __decode_slvreg_table__(self) -> bool:
222  """! @brief Decodes the value of the config Constant object.
223  Assumes that the value is in the format:
224  '('{'"XX"',}','' '['others => "XX"']')'
225  Where the 'X's may be either '0' or '1'.
226  Additionally, instead of "XX", the constants "AS_REG_X" may be used,
227  where the 'X' can be any of: "NONE", "STATUS", "CONTROL" and "BOTH".
228  """
229  self.config_applied = False
230  # Do we even have a config Constant assigned?
231  if self.config is None:
232  return False
233  # Split the value into single register defines
234  # (the comma delimited values).
235  str_arr = self.config.value.split(",")
236  # If the result is an empty list (evaluates to False), error!
237  if not bool(str_arr):
238  LOG.error(
239  (
240  "The register interface configuration constant "
241  "'%s' could not be parsed!"
242  ),
243  self.config.code_name,
244  )
245  return False
246 
247  self.register_table = []
248  for raw_string in str_arr:
249  # For every string: First clean the string
250  item = raw_string.strip(' ()",;\n')
251  try:
252  # try to use the string as a key to the constant register
253  # mapping directory. This "translates" the VHDL-source-string
254  current_reg = self.REG_DIR_MATCH[item]
255  except KeyError:
256  # If that didn't work (string is not a direct register descr.)
257  if "OTHERS" in item:
258  # If it is an "others" directive, extract the register
259  # description string
260  temp = item.strip(' "()').split("=>")
261  # If the current string does not contain "=>", error!
262  if len(temp) < 2:
263  LOG.error(
264  (
265  "The 'others' directive '%s' in the config "
266  "constant '%s' is malformed!"
267  ),
268  raw_string,
269  str(self.config),
270  )
271  continue
272  # Else we can clean the result and match the register type
273  temp[1] = temp[1].strip(' "()')
274  try:
275  others_reg = self.REG_DIR_MATCH[temp[1]]
276  except KeyError:
277  # If that also doesn't work, error!
278  LOG.warning(
279  (
280  "Unrecognized register config: "
281  "'%s' in module '%s'"
282  ),
283  item,
284  str(self.parent),
285  )
286  return False
287  # Else, "fill up" the remaining empty slots of the register
288  # type list
289  LOG.debug(
290  "'others' found, with register type '%s'.", others_reg
291  )
292  reg_count = self.ports[0].data_width.b + 1
293  while len(self.register_table) < reg_count:
294  self.register_table.append(others_reg)
295  break
296  elif "0 =>" in item or "0=>" in item:
297  temp = item.strip(' "()').split("=>")
298  if len(temp) < 2:
299  LOG.error(
300  (
301  "The register configuration constant "
302  "assignemnt is malformed! - '%s'"
303  ),
304  raw_string,
305  )
306  continue
307  temp = temp[-1].strip('" ()')
308  try:
309  current_reg = self.REG_DIR_MATCH[temp]
310  except KeyError:
311  LOG.error(
312  (
313  "Unrecognized register config "
314  "assignment: '%s'"
315  ),
316  temp,
317  )
318  return False
319 
320  else:
321  LOG.error(
322  (
323  "Unrecognized register config: "
324  "'%s' in module '%s'"
325  ),
326  item,
327  str(self.parent),
328  )
329  return False
330  # If the match succeeded, add this registers type to the list
331  self.register_table.append(current_reg)
332 
333  self.config_applied = True
334  # Update the register type count (count: HW->SW, SW->HW, HW<=>SW)
335  self.__count_register_types__()
336  self.get_generic("REG_COUNT").value = self.reg_count
337  LOG.debug(
338  "Register configuration decoded as: '%s'", str(self.register_table)
339  )
340  return True

342  def __count_register_types__(self):
343  """! @brief This method is used to update the attributes 'bidirectional_regs',
344  'hwtosw_regs' and 'swtohw_regs'."""
345  if not self.config_applied:
346  self.__decode_slvreg_table__()
347  self.bidirectional_regs = 0
348  self.hwtosw_regs = 0
349  self.swtohw_regs = 0
350  self.inactive_regs = 0
351  for reg in self.register_table:
352  if reg == "HW <=> SW":
353  self.bidirectional_regs += 1
354  elif reg == "HW -> SW":
355  self.hwtosw_regs += 1
356  elif reg == "HW <- SW":
357  self.swtohw_regs += 1
358  elif reg == "None":
359  self.inactive_regs += 1
360  self.reg_count = (
361  self.bidirectional_regs
362  + self.hwtosw_regs
363  + self.swtohw_regs
364  + self.inactive_regs
365  )

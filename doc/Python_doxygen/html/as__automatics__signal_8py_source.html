138  def __vector_assignment__(self, other, from_bit_index: int, direction: str):
139  if "vector" not in self.data_type:
140  raise AsConnectionError(
141  self,
142  (
143  "Cannot partially assign to/from signal '{}'"
144  " with data type '{}'."
145  ).format(self, self.data_type),
146  )
147  # Removing an assignment
148  if other is None:
149  if direction == "in":
150  vector_map = self.vector_map_incoming
151  con_list = self.incoming
152  else:
153  vector_map = self.vector_map_outgoing
154  con_list = self.outgoing
155  # Get the connecting signal
156  try:
157  other = self.vector_map_incoming[from_bit_index]
158  except KeyError:
159  return False
160  # Remove the references in the vector map and connection list
161  con_list.remove(other)
162  vector_map.pop(from_bit_index)
163  return True
164 
165  if other.data_type not in (
166  "std_logic",
167  "std_logic_vector",
168  "bit",
169  "bit_vector",
170  "std_ulogic",
171  "std_ulogic_vector",
172  ):
173  raise AsConnectionError(
174  self,
175  (
176  "Cannot partially assign {} with data type '{}' "
177  "to/from signal '{}'."
178  ).format(other, other.data_type, self),
179  )
180  omod = get_parent_module(other)
181  if (not isinstance(other, GenericSignal)) and (
182  omod.modlevel > get_parent_module(self).modlevel
183  ):
184  if other.glue_signal:
185  other = other.glue_signal
186  else:
187  signame = omod.name + "_" + other.code_name + "_signal"
188  sig = omod.parent.get_signal(signame)
189  if sig is None:
190  sig = omod.parent.define_signal(
191  signame, other.data_type, other.data_width
192  )
193  sig.generics = copy.copy(other.generics)
194  omod.chain.__connect__(sig, other, top=omod.parent)
195  other = sig
196 
197  # Make the connection
198  if direction == "in":
199  other.outgoing.append(self)
200  self.incoming.append(other)
201  self.vector_map_incoming[from_bit_index] = other
202  # Add a glue signal if not present
203  if not isinstance(other, GenericSignal):
204  if other.glue_signal is None:
205  other.glue_signal = GlueSignal.generate_glue_signal(
206  other, other, self
207  )
208  else:
209  other.incoming.append(self)
210  self.outgoing.append(other)
211  self.vector_map_outgoing[from_bit_index] = other

266  def __check_vector_assignments__(self):
267  """! @brief Checks the plausibility of the vector assignments.
268  If assignments are overlapping or out of bounds, an AsConnectionError
269  is raised."""
270  if not self.data_width.is_resolved():
271  raise AsConnectionError(
272  self,
273  (
274  "Cannot partially assign to/from signal '{}' "
275  "with unresolved data width '{}' (generics present)!"
276  ).format(self, self.data_width),
277  )
278  bitwidth = self.data_width.get_bit_width()
279  for vector_map in (self.vector_map_incoming, self.vector_map_outgoing):
280  assignment_map = set()
281  # Check plausibility of vector assignment (out of bounds, overlap, etc.)
282  for assignment_id in vector_map:
283  # Is bit position out of bounds?
284  if assignment_id > bitwidth:
285  raise AsConnectionError(
286  self,
287  (
288  "Vector assignment to/from signal '{}' at "
289  "bit index {} is out of bounds for signal type: {}"
290  ).format(
291  self,
292  assignment_id,
293  as_help.get_printable_datatype(self),
294  ),
295  )
296  # Get source port/signal
297  signal = vector_map[assignment_id]
298  # Is its data width resolved?
299  if not signal.data_width.is_resolved():
300  raise AsConnectionError(
301  signal,
302  (
303  "Cannot assign {} to/from vector signal {}. "
304  "Data width '{}' is unresolved!"
305  ).format(
306  signal, self, as_help.get_printable_datatype(signal)
307  ),
308  )
309 
310  # Get bit width and a tuple with all bit positions to assign to
311  signal_bit_width = signal.data_width.get_bit_width()
312  new_map = tuple(
313  range(assignment_id, assignment_id + signal_bit_width)
314  )
315  # Does the new assignment overlap with previous assignments?
316  if assignment_map.intersection_update(new_map):
317  raise AsConnectionError(
318  signal,
319  (
320  "Vector assignment to/from signal '{}' from '{}' "
321  "overlaps with previous assignments! List: {}"
322  ).format(self, signal, vector_map),
323  )

347  @classmethod
348  def generate_glue_signal(
349  cls, port: Port, con_from: Port, con_to: Port
350  ) -> Port:
351  """! @brief Generates and returns a GlueSignal port for the given port.
352  First checks a few things using the provided ports (con_to, con_from),
353  to determine if a glue signal is required."""
354  from_parent = get_parent_module(con_from)
355  to_parent = get_parent_module(con_to)
356  from_mdlvl = getattr(from_parent, "modlevel", 0)
357  to_mdlvl = getattr(to_parent, "modlevel", 0)
358  if from_mdlvl + 1 == to_mdlvl:
359  return cls.generate_glue_signal(con_to, con_to, con_from)
360  if from_mdlvl == to_mdlvl + 1:
361  try:
362  con_sig = to_parent.get_signal(con_to.code_name)
363  except AttributeError:
364  con_sig = None
365  if con_sig is None:
366  con_sig = con_to
367  return con_sig
368 
369  if from_mdlvl != to_mdlvl:
370  return None
371  # If the connection source is already a signal
372  if isinstance(con_from, GenericSignal):
373  return con_from
374 
375  # For ports: Format and generate the glue signal name
376  if con_from.port_type == "interface":
377  signame = "{}_{}{}".format(
378  get_parent_module(port).name,
379  port.parent.name_prefix,
380  port.code_name,
381  )
382  else:
383  signame = "{}_{}".format(
384  get_parent_module(port).name, port.code_name
385  )
386  # Create the glue signal object
387  signal = cls(
388  name=signame, data_width=port.data_width, data_type=port.data_type
389  )
390  # Add delay value (for 2D Window Pipeline systems) to glue signal
391  try:
392  setattr(signal, "delay", con_from.delay)
393  except AttributeError:
394  pass
395  signal.assign_to(get_parent_module(port))
396  signal.incoming = con_from
397  signal.outgoing.append(con_to)
398  return signal

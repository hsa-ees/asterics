448  def as_main_dynamic_code(chain, code_dict: dict):
449  # code_dict format: lists "signals" and "body"
450 
451  # Generate a hexadecimal string storing the ASTERICS base address:
452  # Calculate the width in hexadecimal characters of the base address
453  length = int(
454  2 ** ceil(log2(chain.asterics_base_addr.bit_length())) / 4
455  )
456  base_addr_str = 'X"{addr:0{length}X}"'.format(
457  length=length, addr=chain.asterics_base_addr
458  )
459  # Generate some required signals and constants for register management:
460  code_dict["signals"].extend(
461  [
462  "\n -- Register interface constants and signals:",
463  " constant c_asterics_base_addr : unsigned({} downto 0) := {};".format(
464  length * 4 - 1, base_addr_str
465  ),
466  " constant c_slave_reg_addr_width : integer := {};".format(
467  chain.mod_addr_width + chain.reg_addr_width
468  ),
469  " constant c_module_addr_width : integer := {};".format(
470  chain.mod_addr_width
471  ),
472  " constant c_reg_addr_width : integer := {};".format(
473  chain.reg_addr_width
474  ),
475  " constant c_reg_if_count : integer := {};".format(
476  sum(
477  [
478  len(mod.register_ifs)
479  for mod in ittls.chain(
480  chain.modules, chain.pipelines
481  )
482  ]
483  )
484  ),
485  (" signal read_module_addr : integer;"),
486  (
487  " signal sw_address : std_logic_vector"
488  "(c_slave_reg_addr_width - 1 downto 0);"
489  ),
490  (
491  " signal mod_read_data_arr : slv_reg_data"
492  "(0 to c_reg_if_count - 1);"
493  ),
494  ]
495  )
496 
497  # Generate the register manager instantiations:
498  for addr in chain.address_space: # For each address space
499  # Grab the register manager reference
500  regif = chain.address_space[addr]
501  code_dict["signals"].append(
502  " -- Module register base address: 16#{:8X}#".format(
503  regif.base_address
504  )
505  )
506  code_dict["signals"].append(
507  " constant c_{}_regif_num{} : integer := {};".format(
508  regif.parent.name, regif.name_suffix, regif.regif_num
509  )
510  )
511 
512  self.dynamic_code_generators.append(as_main_dynamic_code)

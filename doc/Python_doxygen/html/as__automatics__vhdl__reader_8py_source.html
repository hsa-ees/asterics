116  @classmethod
117  def __valide_line__(cls, file_obj, description: str, lenght: int ) -> (str, int):
118  i = 0
119  while i < lenght:
120  i += 1
121  line = file_obj.readline()
122  if cls.__isDirtyLine__(line) or "@brief" in line or "@file" in line or "@addtogroup" in line or "@defgroup" in line:
123  return description.strip(), i
124  else:
125  if "--!" in line:
126  description += " " + line[line.find("--!") + 3 :].strip()
127  else:

313  self.arch_name,
314  file_obj.name,
315  )

318  # of the VHDL file starts. That doesn't need analysis, used as a
319  # stop condition for this VHDLReader
320  elif ret == "begin":
321  if not in_entity and in_arch and not in_component:
322  self.file_analyzed = True
323  LOG.debug(
324  (

326  " rest of '%s'."
327  ),
328  file_obj.name,
329  )
330  # On keyword 'port', call port anaylsis method when in entity
331  elif ret == "port":

334  else:

336  # Analogous to 'port'

340  else:

342  # Analogous to 'port', only call the method when in architecture
343  elif ret == "constant":

356  "'architecture' in '%s'"

358  file_obj.name,
359  )

361  elif ret == "end":
362  if "entity" in self.line or self.entity_name in self.line:
363  in_entity = False
364  LOG.debug("Found 'end' of entity '%s'", self.entity_name)
365 
366  elif "architecture" in self.line or self.arch_name in self.line:
367  in_arch = False
368  self.file_analyzed = True

674  def __get_data_width__(self, parens: str) -> str:
675  if "," in parens: # For data widths with multiple dimensions
676  data_width = []
677  for width in parens.split(","):
678  data_width.append(self.__get_data_width__(width))
679  return [width for width in data_width if width is not None]
680  # Determine the "direction" of the data width assignment
681  dt_pos = parens.find(" downto ")
682  to_pos = parens.find(" to ")
683  if dt_pos > -1:
684  spliton = "downto"
685  elif to_pos > -1:
686  spliton = "to"
687  else:
688  return None
689 
690  values = parens.split(spliton, 1)

351  def _instantiate_module(self, module: AsModule) -> str:
352  """! @brief Generate VHDL code as a list of strings to instantiate 'module'.
353  Handles generic assignment and port mapping."""
354 
355  # -> Generating the generic map <-
356  gen_str = []
357  if module.generics:
358  gen_str.append(" generic map(")
359  for tgen in module.generics:
360  # If the generic was set by the user in the generator,
361  # use that value
362  if isinstance(tgen.value, Generic):
363  ret = tgen.value.code_name
364  else:
365  ret = tgen.get_value()
366  # Skip generics that use the default value
367  if ret == tgen.default_value:
368  continue
369  gen_str.append(" {} => {},".format(tgen.code_name, ret))
370  if len(gen_str) > 1:
371  # Remove the last comma "," and add a closing bracket
372  gen_str[-1] = gen_str[-1].strip(",")
373  gen_str.append(" )\n")
374  gen_str = "\n".join(gen_str)
375  else:
376  gen_str = ""
377 
378  # -> Generating the port map <-
379  port_str = [" port map("]
380  full_port_list = module.get_full_port_list(include_signals=False)
381  # For every port of this module:
382  for port in full_port_list:
383  # Target of the port map
384  target = None
385 
386  # Determine the format for this ports port map line
387  if full_port_list.index(port) < len(full_port_list) - 1:
388  templ_str = " {} => {},"
389  else:
390  templ_str = " {} => {}\n );"
391 
392  # -> Has glue signal <-
393  # Port mapping target is the port's glue signal
394  if port.glue_signal:
395  glue = port.glue_signal
396  try:
397  target = glue if isinstance(glue, str) else glue.code_name
398  except AttributeError:
399  raise AsConnectionError(
400  port,
401  "Unsuitable object assigned as glue signal of port! "
402  "'{}: {}'".format(type(glue), str(glue)),
403  )
404  # If this glue signal should be included in the signal list
405  if isinstance(glue, GlueSignal) and glue.is_signal:
406  # Add the glue signal to the signal list
407  # Assemble vhdl signal declaration string
408  glue_signal_str = " signal {} : {};".format(
409  glue.code_name, as_help.get_printable_datatype(glue)
410  )
411  # Make sure the same glue signal is not declared twice
412  if glue_signal_str not in self.signal_list:
413  self.signal_list.append(glue_signal_str)
414 
415  else: # -> No glue signal present <-
416  # Port mapping target is one of the connected ports,
417  # depending on port direction
418  target = (
419  port.incoming
420  if port.get_direction_normalized() == "in"
421  else port.outgoing[0]
422  if port.outgoing
423  else None
424  )
425  # If the target is a Port object: use the code_name as target
426  if isinstance(target, Port):
427  target = target.code_name
428  # For strings: use the string as is (eg. for static values)
429  elif isinstance(target, str):
430  target = target
431  else:
432  target = None
433 
434  if not target: # -> No target <-
435  # If the target is 'None': get neutral value
436  if port.get_direction_normalized() == "in":
437  target = port.get_neutral_value()
438  else:
439  target = "open"
440  # And warn the user of an unconnected port
441  if port.optional:
442  LOG.debug(
443  (
444  "Optional port '%s' of module '%s' was "
445  "left unconnected, automatically set to [%s]!"
446  ),
447  port.code_name,
448  get_parent_module(port).name,
449  target,
450  )
451  else:
452  LOG.info(
453  (
454  "Port '%s' of module '%s' was left "
455  "unconnected, automatically set to [%s]!"
456  ),
457  port.code_name,
458  get_parent_module(port).name,
459  target,
460  )
461  # Insert the values in the format string and add to the return list
462  port_str.append(templ_str.format(port.code_name, target))
463  port_str = "\n".join(port_str)
464 
465  # --> OUT
466  entity_keyword_str = (
467  "" if isinstance(module, AsModuleWrapper) else " entity"
468  )
469  out_str = as_static.MODULE_INSTANTIATION_TEMPLATE.format(
470  module_name=module.name,
471  entity_name=module.entity_name,
472  entity_keyword=entity_keyword_str,
473  port_map=port_str,
474  generic_map=gen_str,
475  )
476  return out_str

513  library_use_str += lib_use_template_entities.format(entity)
514

516  entity_name=module_group.name,
517  filename=filename,
518  longdesc=description,
519  briefdesc=description,
520  )

524  as_static.LIBS_IEEE,

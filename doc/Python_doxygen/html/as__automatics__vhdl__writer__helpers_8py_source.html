62 def generate_vector_assignments(signal: GenericSignal) -> list:
63  def get_target_width(a: int, sep: str, b: int = None):
64  if sep == "to":
65  return "{} to {}".format(a, b)
66  elif sep == "downto":
67  return "{} downto {}".format(b, a)
68  else:
69  return str(a)
70 
71  out = ["\n -- Assign to signal {}:".format(signal.code_name)]
72  assignment_template = " {signal}({target}) <= {source};"
73 
74  # Iterate over the vector map starting from the lowest bit
75  for idx in sorted(signal.vector_map_incoming.keys()):
76  source = signal.vector_map_incoming[idx]
77  if source is signal:
78  return []
79  source_bit_width = source.data_width.get_bit_width()
80  sep = source.data_width.sep
81  # Generate the assignment string for this source
82  target = get_target_width(idx, sep, idx + source_bit_width - 1)
83 
84  out.append(
85  assignment_template.format(
86  signal=signal.code_name,
87  target=target,
88  source=source.code_name,
89  )
90  )
91  from_idx = idx + source_bit_width
92  if from_idx < signal.data_width.get_bit_width() - 1:
93  target = get_target_width(
94  from_idx,
95  signal.data_width.sep,
96  signal.data_width.get_bit_width() - 1,
97  )
98  out.append(
99  assignment_template.format(
100  signal=signal.code_name,
101  target=target,
102  source="(others => '0')",
103  )
104  )
105  out.append("")
106  return out

109 def generate_from_vector_assignment_strings(signal: GenericSignal) -> list:
110  """! @brief Generate a string assigning to vector signal 'signal'
111  from its vector map dictionary."""
112  # Comment and empty line for readability
113  out = ["\n -- Assign from signal {}:".format(signal.code_name)]
114  # Iterate over the vector map starting from the lowest bit
115  for idx in sorted(signal.vector_map_outgoing.keys()):
116  target = signal.vector_map_outgoing[idx]
117  sep = target.data_width.sep
118  # Get the target signal
119  if isinstance(target, GenericSignal):
120  pass
121  elif isinstance(target, Port):
122  if target.glue_signal:
123  target = target.glue_signal
124  else:
125  raise AsConnectionError(
126  target,
127  "Port '{}' does not have a glue signal assigned!".format(
128  target
129  ),
130  )
131  target_bit_width = target.data_width.get_bit_width()
132  # Generate the assignment string for this target
133  width = ""
134  if sep == "to":
135  width = "{} to {}".format(idx, idx + target_bit_width - 1)
136  elif sep == "downto":
137  width = "{} downto {}".format(idx + target_bit_width - 1, idx)
138  else:
139  width = str(idx)
140  out.append(
141  " {} <= {}({});".format(target.code_name, signal.code_name, width)
142  )
143  # Leave an empty line after assignment for readability
144  out.append("")
145  return out

148 def bundle_signals(bundle_list: list, bundle_type: str) -> list:
149  """! @brief Generate VHDL code to bundle the signals in the bundle list.
150  The type of bundling (and / or) is determined by 'bundle_type'.
151  Returns a list of VHDL statements."""
152  if not bundle_list:
153  return None # No action on empty list
154  out = []
155  crt_name = ""
156  crt_stmt = ""
157  local_list = copy.copy(bundle_list)
158  local_list.sort(key=lambda sig: sig.code_name)
159 
160  # While there are unprocessed ports in the bundle list
161  while local_list:
162  # Filter by the name of the first port item
163  crt_name = local_list[0].code_name
164  # Initialize the statement
165  crt_stmt = "{} <= ".format(crt_name)
166  # Add all ports with the same name to the statement
167  for nport in (
168  port for port in bundle_list if port.code_name == crt_name
169  ):
170  # Find the connection of the current port:
171  target = nport.glue_signal
172  # Use the connections sink and the bundle type to append
173  crt_stmt += "{} {} ".format(target.code_name, bundle_type)
174  local_list.remove(nport) # And remove them from the local list
175  # Append the statement to the return list
176  # The '[:-4]' removes the last superfluous bundle operand.
177  out.append(" {};".format(crt_stmt[:-4].strip()))
178  return out

181 def convert_generic_entity_list(generics: Sequence[Generic]):
182  """! @brief Convert a list of Generic objects to the string representation
183  in the format for a VHDL entity."""
184  out = []
185  for gen in generics:
186  # Grab the default value
187  gval = gen.default_value
188  # Add quotes if it is a string (not boolean or numeric)
189  if not (gval == "True" or gval == "False" or str(gval).isnumeric()):
190  gval = '"{}"'.format(gval)
191  # Set the format string (last generic "closes" the generic section)
192  if generics.index(gen) == len(generics) - 1:
193  format_str = " {} : {} := {}\n );"
194  else:
195  format_str = " {} : {} := {};"
196  # Insert the generic name, data type and default value
197  # and add the result to the return list
198  out.append(format_str.format(gen.code_name, gen.data_type, gval))
199  return out

202 def convert_port_entity_list(ports: Sequence[Port]):
203  """! @brief Convert a list of port objects to the string representation
204  in the format for a VHDL entity."""
205  out = []
206  in_entity = []
207  for port in ports:
208  # Skip specifically excluded ports
209  if not port.in_entity:
210  continue
211  # Skip ports of excluded interfaces
212  if port.port_type == "interface":
213  if (
214  port.parent.to_external or port.parent.instantiate_in_top
215  ) and port.parent.in_entity:
216  # For external interfaces
217  pass
218  elif port.parent.parent.modlevel > 2 and port.parent.in_entity:
219  # For interfaces of module groups within as_main
220  pass
221  # inlcude only external type single ports
222  elif port.port_type in ("external", "single"):
223  pass
224  else:
225  continue
226  in_entity.append(port)
227 
228  for port in in_entity:
229  # Generate string parts of the entity port declaration
230  port_data = get_printable_datatype(port)
231  port_dir = port.get_direction_normalized()
232  # Generate and add string for entity
233  if in_entity.index(port) == len(in_entity) - 1:
234  out.append(
235  " {} : {} {}\n );".format(
236  port.get_print_name(), port_dir, port_data
237  )
238  )
239  else:
240  out.append(
241  " {} : {} {};".format(
242  port.get_print_name(), port_dir, port_data
243  )
244  )
245  return out

160  # Rules for ports:
161  # Ports need to be assigned rules.
162  # These rules tell Automatics what to do with the ports when it
163  # comes time to build a system. When the generator integrates a module
164  # into a system it will match up interfaces of modules and their ports.
165  # Depending on the condition, which ports are available, Automatics
166  # needs to choose which rules to follow.
167 
168  # Valid rule conditions:
169  # Conditions for comparing two ports of interfaces:
170  # - "any_present" (this is always true)
171  # - "both_present"
172  # - "sink_missing"
173  # Conditions for ports not part of an interface:
174  # - "single_port"
175  # - "external_port"
176 
177  # Valid rule-actions:
178  # "connect":
179  # Only applicable on the condition "both_present".
180  # Connects the two ports to each other
181  # "make_external":
182  # Only applicable on the condition "single_port" or "external_port".
183  # The generator will connect the port through to the toplevel
184  # of the ASTERICS IP-Core.
185  # "error"
186  # Applicable for any condition.
187  # Raises an error message and stops as_automatics.
188  # "warning"
189  # Applicable for any condition.
190  # Prints a warning message, continuing the process.
191  # "note"
192  # Applicable for any condition.
193  # Prints a n information message to the console output and build log.
194  # "set_value(<value>)"
195  # Applicable to the conditions "single_port",
196  # "sink_missing", "any_present".
197  # Sets the port to the user defined <value>.
198  # This can be a VHDL signal name, VHDL keyword,
199  # number or data representation (eg. '1', X"F00BA5", open, reg_03).
200  # The "value" given will be directly copied into the VHDL source code.
201  # "bundle_and"
202  # Applicable to all conditions.
203  # Similar to "make_external", but bundles all ports with the same
204  # rule and name of all ASTERICS modules in this system.
205  # The signals are bundled together using a single big AND gate.
206  # "bundle_or"
207  # This rule behaves exactly like "bundle_and", except that for
208  # this rule the signals are bundled using a big OR gate.
209  # "fallback_port(<port name>)"
210  # Applicable to "sink_missing",
211  # depending on the ports data direction.
212  # This action defines another port the port may be connected to, if its
213  # counterpart is missing.
214  # "fallback_signal(<signal name>)"
215  # Applicable to "sink_missing".
216  # This action defines a signal that the port may be connected to, if its
217  # counterpart is missing.
218  # "none"
219  # The default rule action: Do nothing.

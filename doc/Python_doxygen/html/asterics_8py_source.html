215 def new_2d_window_pipeline(
216  image_width: int,
217  image_height: int = 480,
218  name: str = "",
219  force_synchronous_pipeline: bool = None,
220 ) -> As2DWindowPipeline:
221  """! @brief Provide a new As2DWindowPipeline object.
222  This allows you to specify and build a new ASTERICS 2D Window Pipeline for
223  processing image data using convolution and other two dimensional filters.
224  You need to provide the dimensions of the image data you want to process,
225  as the pipeline size and resource usage is highly dependend on the size of
226  the image data.
227  @param image_width: The number of horizontal pixels per image row
228  @param image_height: The number of vertical pixels per image column (optional)
229  @param name: The name of the pipeline (optional)
230  @param force_synchronous_pipeline: Wether the pipeline should be implemented in
231  a synchronous design, i.e. all modules will utilize the same
232  "Strobe" (data valid) signal, including the pipeline's output.
233  Leave as "None" if you're not sure, to let Automatics choose.
234  @return A new As2DWindowPipeline object."""
235  if not Auto.current_chain:
236  LOG.error(
237  (
238  "Before creating a new 2D Window Pipeline, you first have to"
239  " create an ASTERICS processing chain using"
240  " 'asterics.new_chain()'"
241  )
242  )
243  raise as_err.AsTextError(
244  "",
245  msg="2D Window Pipeline construction requires a processing chain.",
246  detail=(
247  "Call 'asterics.new_chain()' before "
248  "'asterics.new_2d_window_pipeline()'."
249  ),
250  severity="Critical",
251  )
252  if not name:
253  name = "as_window_pipe_{}".format(len(Auto.windowpipes))
254  pipe = As2DWindowPipeline(
255  image_width, image_height, name, chain=Auto.current_chain
256  )
257  if force_synchronous_pipeline is not None:
258  pipe.set_pipeline_synchronous(force_synchronous_pipeline)
259  Auto.windowpipes.append(pipe)
260  return pipe

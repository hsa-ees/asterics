534  def autofill(self):
535  """fills in the module list table according to the text entered in the search field"""
536  searchText = self.completer.text().lower()
537  self.portList.hide()
538  modNames = []
539  for modName in self.module_names_automatics:
540  rowData = self.module_names_automatics[modName]
541  if (self.showOther.isChecked() and rowData["mod_show"]) or (
542  self.showAutomatics.isChecked() and not rowData["mod_show"]
543  ):
544  continue
545 
546  modName = modName.lower()
547  if searchText == modName:
548  if self.modlist.rowCount() != 1:
549  modNames.append(modName)
550  continue
551  else:
552  self.modlist.selectRow(0)
553  selectedRow = self.modlist.selectedItems()[0]
554  self.modlist.scrollToItem(selectedRow)
555  return
556  elif (
557  searchText in modName
558  or searchText in rowData["mod_status"].lower()
559  or searchText in rowData["dev_status"].lower()
560  or searchText in rowData["module_category"].lower()
561  ):

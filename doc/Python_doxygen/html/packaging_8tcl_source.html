37 # findFiles
38 # basedir - the directory to start looking in
39 # pattern - A pattern, as defined by the glob command, that the files must match
40 proc findFiles { basedir pattern } {
41 
42  # Fix the directory name, this ensures the directory name is in the
43  # native format for the platform and contains a final directory seperator
44  set basedir [string trimright [file join [file normalize $basedir] { }]]
45  set fileList {}
46 
47  # Look in the current directory for matching files, -type {f r}
48  # means ony readable normal files are looked at, -nocomplain stops
49  # an error being thrown if the returned list is empty
50  foreach fileName [glob -nocomplain -type {f r} -path $basedir $pattern] {
51  lappend fileList $fileName
52  }
53 
54  # Now look for any sub direcories in the current directory
55  foreach dirName [glob -nocomplain -type {d r} -path $basedir *] {
56  # Recusively call the routine on the sub directory and append any
57  # new files to the results
58  set subDirList [findFiles $dirName $pattern]
59  if { [llength $subDirList] > 0 } {
60  foreach subDirFile $subDirList {
61  lappend fileList $subDirFile
62  }
63  }
64  }
65  return $fileList
66  }

59 entity OutputSERDES is
60  Generic (
61  kParallelWidth : natural := 10); -- number of parallel bits
62  Port (
63  PixelClk : in STD_LOGIC; --TMDS clock x1 (CLKDIV)
64  SerialClk : in STD_LOGIC; --TMDS clock x5 (CLK)
65 
66  --Encoded serial data
67  sDataOut_p : out STD_LOGIC;
68  sDataOut_n : out STD_LOGIC;
69 
70  --Encoded parallel data (raw)
71  pDataOut : in STD_LOGIC_VECTOR (kParallelWidth-1 downto 0);
72 
73  aRst : in STD_LOGIC);
74 end OutputSERDES;
75 
76 architecture Behavioral of OutputSERDES is
77 
78 signal sDataOut, ocascade1, ocascade2 : std_logic;
79 signal pDataOut_q : std_logic_vector(13 downto 0);
80 begin
81 
82 -- Differential output buffer for TMDS I/O standard
83 OutputBuffer: OBUFDS
84  generic map (
85  IOSTANDARD => "TMDS_33")
86  port map (
87  O => sDataOut_p,
88  OB => sDataOut_n,
89  I => sDataOut);
90 
91 -- Serializer, 10:1 (5:1 DDR), master-slave cascaded
92 SerializerMaster: OSERDESE2
93  generic map (
94  DATA_RATE_OQ => "DDR",
95  DATA_RATE_TQ => "SDR",
96  DATA_WIDTH => kParallelWidth,
97  TRISTATE_WIDTH => 1,
98  TBYTE_CTL => "FALSE",
99  TBYTE_SRC => "FALSE",
100  SERDES_MODE => "MASTER")
101  port map (
102  OFB => open, -- 1-bit output: Feedback path for data
103  OQ => sDataOut, -- 1-bit output: Data path output
104  -- SHIFTOUT1 / SHIFTOUT2: 1-bit (each) output: Data output expansion (1-bit each)
105  SHIFTOUT1 => open,
106  SHIFTOUT2 => open,
107  TBYTEOUT => open, -- 1-bit output: Byte group tristate
108  TFB => open, -- 1-bit output: 3-state control
109  TQ => open, -- 1-bit output: 3-state control
110  CLK => SerialClk, -- 1-bit input: High speed clock
111  CLKDIV => PixelClk, -- 1-bit input: Divided clock
112  -- D1 - D8: 1-bit (each) input: Parallel data inputs (1-bit each)
113  D1 => pDataOut_q(13),
114  D2 => pDataOut_q(12),
115  D3 => pDataOut_q(11),
116  D4 => pDataOut_q(10),
117  D5 => pDataOut_q(9),
118  D6 => pDataOut_q(8),
119  D7 => pDataOut_q(7),
120  D8 => pDataOut_q(6),
121  OCE => '1', -- 1-bit input: Output data clock enable
122  RST => aRst, -- 1-bit input: Reset
123  -- SHIFTIN1 / SHIFTIN2: 1-bit (each) input: Data input expansion (1-bit each)
124  SHIFTIN1 => ocascade1,
125  SHIFTIN2 => ocascade2,
126  -- T1 - T4: 1-bit (each) input: Parallel 3-state inputs
127  T1 => '0',
128  T2 => '0',
129  T3 => '0',
130  T4 => '0',
131  TBYTEIN => '0', -- 1-bit input: Byte group tristate
132  TCE => '0' -- 1-bit input: 3-state clock enable
133  );
134 
135 SerializerSlave: OSERDESE2
136  generic map (
137  DATA_RATE_OQ => "DDR",
138  DATA_RATE_TQ => "SDR",
139  DATA_WIDTH => kParallelWidth,
140  TRISTATE_WIDTH => 1,
141  TBYTE_CTL => "FALSE",
142  TBYTE_SRC => "FALSE",
143  SERDES_MODE => "SLAVE")
144  port map (
145  OFB => open, -- 1-bit output: Feedback path for data
146  OQ => open, -- 1-bit output: Data path output
147  -- SHIFTOUT1 / SHIFTOUT2: 1-bit (each) output: Data output expansion (1-bit each)
148  SHIFTOUT1 => ocascade1,
149  SHIFTOUT2 => ocascade2,
150  TBYTEOUT => open, -- 1-bit output: Byte group tristate
151  TFB => open, -- 1-bit output: 3-state control
152  TQ => open, -- 1-bit output: 3-state control
153  CLK => SerialClk, -- 1-bit input: High speed clock
154  CLKDIV => PixelClk, -- 1-bit input: Divided clock
155  -- D1 - D8: 1-bit (each) input: Parallel data inputs (1-bit each)
156  D1 => '0',
157  D2 => '0',
158  D3 => pDataOut_q(5),
159  D4 => pDataOut_q(4),
160  D5 => pDataOut_q(3),
161  D6 => pDataOut_q(2),
162  D7 => pDataOut_q(1),
163  D8 => pDataOut_q(0),
164  OCE => '1', -- 1-bit input: Output data clock enable
165  RST => aRst, -- 1-bit input: Reset
166  -- SHIFTIN1 / SHIFTIN2: 1-bit (each) input: Data input expansion (1-bit each)
167  SHIFTIN1 => '0',
168  SHIFTIN2 => '0',
169  -- T1 - T4: 1-bit (each) input: Parallel 3-state inputs
170  T1 => '0',
171  T2 => '0',
172  T3 => '0',
173  T4 => '0',
174  TBYTEIN => '0', -- 1-bit input: Byte group tristate
175  TCE => '0' -- 1-bit input: 3-state clock enable
176  );
177 
178 -------------------------------------------------------------
179 -- Concatenate the serdes inputs together. Keep the timesliced
180 -- bits together, and placing the earliest bits on the right
181 -- ie, if data comes in 0, 1, 2, 3, 4, 5, 6, 7, ...
182 -- the output will be 3210, 7654, ...
183 -------------------------------------------------------------
184 SliceOSERDES_q: for slice_count in 0 to kParallelWidth-1 generate begin
185  --DVI sends least significant bit first
186  --OSERDESE2 sends D1 bit first
187  pDataOut_q(14-slice_count-1) <= pDataOut(slice_count);
188 end generate SliceOSERDES_q;
189 
190 end Behavioral;

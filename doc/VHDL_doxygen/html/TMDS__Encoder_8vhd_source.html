40 -- Purpose:
41 -- This module implements the encoding algorithm outlined in the
42 -- DVI 1.0 specifications and instantiates the serializer block. The 8-bit data
43 -- and 3 control signals are encoded and transmitted over the data channel.
44 -- The sDataOut_p/n ports must connect to top-level ports.
45 --

63 entity TMDS_Encoder is
64  Port (
65  PixelClk : in std_logic; --Recovered TMDS clock x1 (CLKDIV)
66  SerialClk : in std_logic; --Recovered TMDS clock x5 (CLK)
67  aRst : in std_logic; --asynchronous reset; must be reset when PixelClk/SerialClk is not within spec
68 
69  --Encoded parallel data
70  pDataOutRaw : out std_logic_vector(9 downto 0);
71 
72  --Unencoded parallel data
73  pDataOut : in std_logic_vector(7 downto 0);
74  pC0 : in std_logic;
75  pC1 : in std_logic;
76  pVde : in std_logic
77  );
78 end TMDS_Encoder;

80 architecture Behavioral of TMDS_Encoder is
81 
82 signal pDataOut_1 : std_logic_vector(7 downto 0);
83 signal q_m_1, q_m_xor_1, q_m_xnor_1, q_m_2: std_logic_vector(8 downto 0);
84 signal control_token_2, q_out_2: std_logic_vector(9 downto 0);
85 signal n1d_1, n1q_m_2, n0q_m_2, n1q_m_1 : unsigned(3 downto 0); --range 0-8
86 signal dc_bias_2, cnt_t_3, cnt_t_2 : signed(4 downto 0) := "00000"; --range -8 - +8 + sign
87 signal pC0_1, pC1_1, pVde_1, pC0_2, pC1_2, pVde_2 : std_logic;
88 signal cond_not_balanced_2, cond_balanced_2 : std_logic;
89 
90 function sum_bits(u : std_logic_vector) return unsigned is
91  variable sum : unsigned(3 downto 0);
92  begin
93  assert u'length < 16 report "sum_bits error";
94  sum := to_unsigned(0,4);
95  for i in u'range loop
96  sum := sum + unsigned(u(i downto i));
97  end loop;
98  return sum;
99  end sum_bits;
100 
101 begin
102 
103 ----------------------------------------------------------------------------------
104 -- DVI 1.0 Specs Figure 3-5
105 -- Pipeline stage 1, minimise transitions
106 ----------------------------------------------------------------------------------
107 Stage1: process(PixelClk)
108 begin
109  if Rising_Edge(PixelClk) then
110  pVde_1 <= pVde;
111 
112  n1d_1 <= sum_bits(pDataOut(7 downto 0));
113  pDataOut_1 <= pDataOut; --insert data into the pipeline;
114  pC0_1 <= pC0; --insert control into the pipeline;
115  pC1_1 <= pC1;
116  end if;
117 end process Stage1;
118 
119 ----------------------------------------------------------------------------------
120 -- Choose one of the two encoding options based on n1d_1
121 ----------------------------------------------------------------------------------
122 q_m_xor_1(0) <= pDataOut_1(0);
123 encode1: for i in 1 to 7 generate
124  q_m_xor_1(i) <= q_m_xor_1(i-1) xor pDataOut_1(i);
125 end generate encode1;
126 q_m_xor_1(8) <= '1';
127 
128 q_m_xnor_1(0) <= pDataOut_1(0);
129 encode2: for i in 1 to 7 generate
130  q_m_xnor_1(i) <= q_m_xnor_1(i-1) xnor pDataOut_1(i);
131 end generate encode2;
132 q_m_xnor_1(8) <= '0';
133 
134 q_m_1 <= q_m_xnor_1 when n1d_1 > 4 or (n1d_1 = 4 and pDataOut_1(0) = '0') else
135  q_m_xor_1;
136 
137 n1q_m_1 <= sum_bits(q_m_1(7 downto 0));
138 
139 ----------------------------------------------------------------------------------
140 -- Pipeline stage 2, balance DC
141 ----------------------------------------------------------------------------------
142 Stage2: process(PixelClk)
143 begin
144  if Rising_Edge(PixelClk) then
145  n1q_m_2 <= n1q_m_1;
146  n0q_m_2 <= 8 - n1q_m_1;
147  q_m_2 <= q_m_1;
148  pC0_2 <= pC0_1;
149  pC1_2 <= pC1_1;
150  pVde_2 <= pVde_1;
151  end if;
152 end process Stage2;
153 
154 cond_balanced_2 <= '1' when cnt_t_3 = 0 or n1q_m_2 = 4 else -- DC balanced output
155  '0';
156 cond_not_balanced_2 <= '1' when (cnt_t_3 > 0 and n1q_m_2 > 4) or -- too many 1's
157  (cnt_t_3 < 0 and n1q_m_2 < 4) else -- too many 0's
158  '0';
159 
160 control_token_2 <= kCtlTkn0 when pC1_2 = '0' and pC0_2 = '0' else
161  kCtlTkn1 when pC1_2 = '0' and pC0_2 = '1' else
162  kCtlTkn2 when pC1_2 = '1' and pC0_2 = '0' else
163  kCtlTkn3;
164 
165 q_out_2 <= control_token_2 when pVde_2 = '0' else --control period
166  not q_m_2(8) & q_m_2(8) & not q_m_2(7 downto 0) when cond_balanced_2 = '1' and q_m_2(8) = '0' else
167  not q_m_2(8) & q_m_2(8) & q_m_2(7 downto 0) when cond_balanced_2 = '1' and q_m_2(8) = '1' else
168  '1' & q_m_2(8) & not q_m_2(7 downto 0) when cond_not_balanced_2 = '1' else
169  '0' & q_m_2(8) & q_m_2(7 downto 0); --DC balanced
170 
171 dc_bias_2 <= signed('0' & n0q_m_2) - signed('0' & n1q_m_2);
172 
173 cnt_t_2 <= to_signed(0, cnt_t_2'length) when pVde_2 = '0' else --control period
174  cnt_t_3 + dc_bias_2 when cond_balanced_2 = '1' and q_m_2(8) = '0' else
175  cnt_t_3 - dc_bias_2 when cond_balanced_2 = '1' and q_m_2(8) = '1' else
176  cnt_t_3 + signed('0' & q_m_2(8 downto 8) & '0') + dc_bias_2 when cond_not_balanced_2 = '1' else
177  cnt_t_3 - signed('0' & not q_m_2(8 downto 8) & '0') - dc_bias_2;
178 
179 ----------------------------------------------------------------------------------
180 -- Pipeline stage 3, registered output
181 ----------------------------------------------------------------------------------
182 Stage3: process(PixelClk)
183 begin
184  if Rising_Edge(PixelClk) then
185  cnt_t_3 <= cnt_t_2;
186  pDataOutRaw <= q_out_2; --encoded, ready to be serialized
187  end if;
188 end process Stage3;
189 
190 end Behavioral;

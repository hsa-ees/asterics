566  function f_generate_csa_tree_input_bit_widths(constant bit_widths : in t_integer_array; constant stage_count : integer) return t_bit_width_mapping is
567  constant total_elements : integer := bit_widths'length;
568  variable bit_mapping : t_bit_width_mapping(0 to stage_count, 0 to total_elements - 1) := (others => (others => 0));
569  variable elements_this_stage : integer;
570  variable elements_propagate : integer;
571  variable adders_this_stage : integer;
572  variable current_elements : t_integer_array(0 to 2);
573  variable largest_bit_width : integer;
574  begin
575  -- First stage: data inputs
576  for N in bit_widths'range loop
577  bit_mapping(0, N) := bit_widths(N);
578  end loop;
579  if total_elements < 2 then
580  return bit_mapping;
581  end if;
582  -- Compute control values for first stage
583  elements_this_stage := total_elements;
584  elements_propagate := elements_this_stage mod 3;
585  adders_this_stage := elements_this_stage / 3;
586 
587  -- Stages 1 to S - 1:
588  for S in 1 to stage_count - 1 loop
589  -- For every Carry Safe Adder:
590  for N in 0 to adders_this_stage - 1 loop
591  -- Get input bit widths
592  current_elements := (
593  0 => bit_mapping(S - 1, N * 3 + 0),
594  1 => bit_mapping(S - 1, N * 3 + 1),
595  2 => bit_mapping(S - 1, N * 3 + 2)
596  );
597  -- Get largest input bit width
598  largest_bit_width := f_get_largest_integer(current_elements, 0, 2);
599  -- Sum output will have same output bit width (input of next stage)
600  bit_mapping(S, N * 2 + 0) := largest_bit_width;
601  -- Carry output will have largest input + 1 bit width
602  bit_mapping(S, N * 2 + 1) := largest_bit_width + 1;
603  end loop;
604  -- For element propagations:
605  for N in 0 to elements_propagate - 1 loop
606  -- Carry bit width to next stage
607  bit_mapping(S, adders_this_stage * 2 + N) := bit_mapping(S - 1, adders_this_stage * 3 + N);
608  end loop;
609  -- Update values for next stage
610  elements_propagate := elements_this_stage mod 3;
611  adders_this_stage := elements_this_stage / 3;
612  elements_this_stage := (adders_this_stage * 2) + elements_propagate;
613  end loop;
614 
615 
616  -- Last stage (Carry ripple adder):
617  current_elements := (
618  0 => bit_mapping(stage_count - 1, 0),
619  1 => bit_mapping(stage_count - 1, 1),
620  2 => 0
621  );
622  largest_bit_width := f_get_largest_integer(current_elements, 0, 1);
623  bit_mapping(stage_count, 0) := largest_bit_width + 1;
624  return bit_mapping;
625  end function;

903  if is_negative = '1' then

905  else

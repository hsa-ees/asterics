47 entity as_cnn_quantizer is
48  generic(
49  DIN_WIDTH : integer := 8;
50  DOUT_WIDTH : integer := 8;

52  QUANT_N_SHIFT_BITWIDTH : integer := 4;

77 architecture RTL of as_cnn_quantizer is
78  -- Register stage signals
79  signal data_with_bias : signed(55 downto 0);
80  signal mzero_register : signed(23 downto 0);
81  signal data_with_mzero : signed(79 downto 0);
82  signal data_with_mzero_s1 : signed(data_with_mzero'range);
83  signal data_with_mzero_s2 : signed(data_with_mzero'range);
84  signal data_with_mzero_s3 : signed(data_with_mzero'range);
85  signal data_with_mzero_s4 : signed(data_with_mzero'range);
86  signal data_with_mzero_s5 : signed(data_with_mzero'range);
87 begin
88 
89  p_bias_add : process(clk)
90  variable sum : signed(31 downto 0);
91  variable integer_result : integer;
92  begin
93  if rising_edge(clk) then
94  if reset = '1' then
95  data_with_bias <= (others => '0');
96  mzero_register <= (others => '0');
97  else
98  if strobe_in = '1' then
99  -- Store raw convolution result in 32 bits for processing
100  sum := resize(signed(data_in), 32);
101  -- Apply BIAS
102  sum := sum + bias_value;
103  data_with_bias(55 downto 24) <= sum; -- !register stage!
104  data_with_bias(23 downto 0) <= (others => '0');
105  -- Hold M0 value (Reduces signal delay to multiplier)
106  mzero_register <= signed(quant_m_zero); -- !register stage!
107  end if;
108  end if;
109  end if;
110  end process;
111 
112  p_quantization_multiplier_stage_0 : process(clk)
113  begin
114  if rising_edge(clk) then
115  if strobe_in = '1' then
116  data_with_mzero_s5 <= data_with_mzero_s4; -- !register stage!
117  data_with_mzero_s4 <= data_with_mzero_s3; -- !register stage!
118  data_with_mzero_s3 <= data_with_mzero_s2; -- !register stage!
119  data_with_mzero_s2 <= data_with_mzero_s1; -- !register stage!
120  data_with_mzero_s1 <= data_with_mzero; -- !register stage!
121  -- Multiply with quantization scaling factor M0
122  data_with_mzero <= data_with_bias * mzero_register; -- !register stage!
123  end if;
124  end if;
125  end process;
126 
127  p_activation : process(quant_n_shift, data_with_mzero_s5)
128  constant clamping_value_p: integer := (2 ** (DOUT_WIDTH - 1)) - 1;
129  constant clamping_value_n: integer := (2 ** (DOUT_WIDTH - 1)) * (-1);
130  constant c_quant_offset: signed(31 downto 0) := to_signed(QUANTIZATION_OFFSET, 32);
131 
132  variable scaled_result_signed : signed(31 downto 0);
133  variable data_shifted : signed(31 downto 0) := (others => '0');
134  variable data_quantized : signed(31 downto 0) := (others => '0');
135  variable data_saturated : signed(DOUT_WIDTH - 1 downto 0) := (others => '0');
136  variable data_activated : signed(DOUT_WIDTH - 1 downto 0) := (others => '0');
137  begin
138  scaled_result_signed := data_with_mzero_s5(78 downto 47);
139  -- Rounding right_shift by n (c_n)
140  -- Equivalent to multiplication by 2^(-c_n) + data_with_mzero(c_n - 1)
141  data_shifted := f_rounding_right_shift_non_reducing(scaled_result_signed, to_integer(quant_n_shift));
142  -- Apply offset defined by QUANTIZATION_OFFSET
143  data_quantized := data_shifted + c_quant_offset;
144  -- Clamp values to value domain of signed DOUT_WIDTH bits
145  if to_integer(data_quantized) > clamping_value_p then
146  data_saturated := to_signed(clamping_value_p, DOUT_WIDTH);
147  elsif to_integer(data_quantized) < clamping_value_n then
148  data_saturated := to_signed(clamping_value_n, DOUT_WIDTH);
149  else
150  data_saturated := data_quantized(DOUT_WIDTH - 1 downto 0);
151  end if;
152  -- Apply activation function
153  if ACTIVATION_FUNCTION = "relu" then
154  data_activated := to_signed(f_relu(to_integer(data_saturated)), DOUT_WIDTH);
155  elsif ACTIVATION_FUNCTION = "none" then
156  data_activated := data_saturated;
157  end if;
158  -- Output register stage
159  data_out <= std_logic_vector(data_activated);
160  end process;

162 end architecture;

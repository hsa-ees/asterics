280  stall_out_int <= '1' when unsigned(runtime_control(c_ctrl_stall_active_lb to c_ctrl_stall_active_ub)) > 0 or stall_int = '1' else '0';

288  -- Stall internal execution when both external stall is high and the next cycle would produce an output
289  -- Otherwise, if external stall is high and further internal execution does not produce results, it is safe to continue processing
290  stall_int <= '1' when ((stall_out = '1') and (runtime_control(runtime_control'length - 1) = '1')) else '0';
291 
292  -- strobe_in functions as the first bit of runtime_control
293  runtime_control <= strobe_in & runtime_control_int;
294  runtime_control_next <= runtime_control(0 to runtime_control'length - 2);

297  p_conv_control : process(clk) is
298  variable v_next_runtime_control : std_logic_vector(runtime_control'range);
299  constant c_start: unsigned(FILTER_COUNT - 1 downto 0) := shift_left(to_unsigned(1, FILTER_COUNT), FILTER_COUNT - 1);
300  begin
301  if rising_edge(clk) then
302  if reset = '1' then
303  strobe_out <= '0';
304  runtime_control_int <= (others => '0');
305  else
306  -- Halt when internal stall is high
307  if stall_int = '0' then
308  -- Shift runtime_control every clock cycle
309  runtime_control_int <= runtime_control_next;
310 
311  -- Update multiplexer control index signals
312  -- The mutliplexer indices are generated using small counters
313  if (unsigned(runtime_control_next(c_ctrl_adder_in_mux_lb - 1 to c_ctrl_adder_in_mux_ub - 1)) = c_start)
314  or (unsigned(runtime_control_next(c_ctrl_adder_in_mux_lb - 1 to c_ctrl_adder_in_mux_ub - 1)) = 0) then
315  adder_mux_ctrl_index <= (others => '0');
316  elsif unsigned(runtime_control_next(c_ctrl_adder_in_mux_lb - 1 to c_ctrl_adder_in_mux_ub - 1)) > 0 then
317  adder_mux_ctrl_index <= adder_mux_ctrl_index + 1;
318  end if;
319 
320  if (unsigned(runtime_control_next(c_ctrl_quant_bias_mzero_lb - 1 to c_ctrl_quant_bias_mzero_ub - 1)) = c_start)
321  or (unsigned(runtime_control_next(c_ctrl_quant_bias_mzero_lb - 1 to c_ctrl_quant_bias_mzero_ub - 1)) = 0) then
322  quant_bias_mzero_ctrl_index <= (others => '0');
323  elsif unsigned(runtime_control_next(c_ctrl_quant_bias_mzero_lb - 1 to c_ctrl_quant_bias_mzero_ub - 1)) > 0 then
324  quant_bias_mzero_ctrl_index <= quant_bias_mzero_ctrl_index + 1;
325  end if;
326 
327  if (unsigned(runtime_control_next(c_ctrl_quant_nshift_lb - 1 to c_ctrl_quant_nshift_ub - 1)) = c_start)
328  or (unsigned(runtime_control_next(c_ctrl_quant_nshift_lb - 1 to c_ctrl_quant_nshift_ub - 1)) = 0) then
329  quant_nshift_ctrl_index <= (others => '0');
330  elsif unsigned(runtime_control_next(c_ctrl_quant_nshift_lb - 1 to c_ctrl_quant_nshift_ub - 1)) > 0 then
331  quant_nshift_ctrl_index <= quant_nshift_ctrl_index + 1;
332  end if;
333 
334  strobe_out <= runtime_control_int(runtime_control_int'length - 1);
335  end if;
336  end if;
337  end if;
338  end process;

349  and stall_int = '0' else '0';

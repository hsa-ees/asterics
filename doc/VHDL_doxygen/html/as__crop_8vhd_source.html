94  -- Slave register configuration:
95  -- Allows for "dynamic" configuration of slave registers
96  -- Possible values and what they mean:
97  -- "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
98  -- "01": From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
99  -- "10": From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
100  -- "11": Combined Read/Write register. Data transport in both directions.
101  -- When both sides attempt to write simultaniously, only the HW gets to write.
102  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
103  constant slave_register_configuration : slv_reg_config_table(0 to 3) :=
104  ("11","10","10","01");

147  status_control_update_logic: process(control, control_reset, slv_ctrl_reg, state, reset)
148  variable var_control_new : std_logic_vector(15 downto 0);
149  begin
150  -- int_reset control and status register modify bit

152 
153  -- Clear control bits of the register on module reset
154  if reset = '1' then
155  control_new <= (others => '0');
156  else
157  control_new <= control;
158  end if;

160  -- Apply control_reset bit mask
161  var_control_new := control and (not control_reset);
162 
163  -- If either control or state was modified by hardware, set modify bit

165  control_new <= var_control_new;
166  slv_reg_modify(0) <= '1';
167  end if;
168  if state /= slv_ctrl_reg(0)(15 downto 0) then
169  slv_reg_modify(0) <= '1';
170  end if;
171  end process status_control_update_logic;

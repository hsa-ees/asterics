110  function f_filterGetMax(constant filter : in t_filter) return natural is
111  variable max : integer := 0;
112  begin
113  for i in 0 to c_filterWidth-1 loop
114  for j in 0 to c_filterWidth-1 loop
115  if abs(filter(i,j)) > max then
116  max := abs(filter(i,j));
117  end if;
118  end loop;
119  end loop;
120  return max;
121  end f_filterGetMax;
122 
123  function f_filterGetSumAbs(constant filter : in t_filter) return natural is
124  variable sum : integer := 0;
125  begin
126  for i in 0 to c_filterWidth-1 loop
127  for j in 0 to c_filterWidth-1 loop
128  sum := sum + abs(filter(i,j));
129  end loop;
130  end loop;
131  return sum;
132  end f_filterGetSumAbs;
133 
134  function f_filterGetSum(constant filter : in t_filter) return integer is
135  variable sum : integer := 0;
136  begin
137  for i in 0 to c_filterWidth-1 loop
138  for j in 0 to c_filterWidth-1 loop
139  sum := sum + filter(i,j);
140  end loop;
141  end loop;
142  return sum;
143  end f_filterGetSum;

52 entity as_generic_filter_module is
53  generic(
54  DIN_WIDTH : integer := 8;
55  FILTER_BIT_WIDTH : integer := 8;
56  DOUT_WIDTH : integer := 8;
57  WINDOW_X : integer := 3;
58  WINDOW_Y : integer := 3;
59  COMPUTATION_DATA_WIDTH_ADD : integer := 0;
60  NORMALIZE_TO_HALF : boolean := false;
61  OUTPUT_SIGNED : boolean := false
62  );
63  port(
64  clk : in std_logic;
65  reset : in std_logic;
66 
67  filter_values : in t_generic_window(0 to WINDOW_Y - 1, 0 to WINDOW_X - 1, FILTER_BIT_WIDTH - 1 downto 0); -- Interpreted as signed factors. Use powers of two for best (fastest) results!
68 
69  strobe : in std_logic;
70  window : in t_generic_window(0 to WINDOW_Y - 1, 0 to WINDOW_X - 1, DIN_WIDTH - 1 downto 0);
71  data_out : out std_logic_vector(DOUT_WIDTH - 1 downto 0)
72  );
73 end entity as_generic_filter_module;

77 architecture RTL of as_generic_filter_module is
78 
79  constant c_comp_bits_add : integer := COMPUTATION_DATA_WIDTH_ADD;
80  -- Data width for computation:
81  -- input width + filter weight (c_comp_bits_add) + delta to output if output wider
82  constant c_comp_data_width : integer := DIN_WIDTH + c_comp_bits_add + f_relu(DOUT_WIDTH - DIN_WIDTH);
83  -- # of bits to downscale:
84  -- c_comp_bits_add + delta to output if input wider
85  constant c_downscale_bits : integer := c_comp_bits_add + f_relu(DIN_WIDTH - DOUT_WIDTH);
86 
87  type t_sums is array(natural range<>) of signed(c_comp_data_width downto 0);
88 
89  signal applied_value_lines : t_sums(0 to WINDOW_Y - 1); -- signed values
90  --signal debug_pix : t_sums(0 to 8);
91  --signal debug_filter : t_sums(0 to 8);
92  --signal debug_res : t_sums(0 to 8);
93 begin
94 
95 
96  -- First pipeline stage
97  apply_filter_values : process(clk) is
98  variable factor : integer;
99  variable f_sign : std_logic;
100  variable pixel : unsigned(DIN_WIDTH - 1 downto 0);
101  variable result_int : integer;
102  variable result_vector : std_logic_vector(c_comp_data_width downto 0);
103  variable line_acc : t_sums(0 to WINDOW_X - 1);
104  variable acc : signed(c_comp_data_width downto 0);
105  begin
106  if rising_edge(clk) then
107  if reset = '1' then
108  applied_value_lines <= (others => (others => '0'));
109  else
110  if strobe = '1' then
111  for Y in 0 to WINDOW_Y - 1 loop
112  acc := (others => '0');
113  for X in 0 to WINDOW_X - 1 loop
114  -- Extract values
115  factor := to_integer(signed(f_get_vector_of_generic_window(filter_values, X, Y)));
116  pixel := unsigned(f_get_vector_of_generic_window(window, X, Y));
117  -- Multiply
118  result_int := to_integer(pixel) * factor;
119 
120  -- Debug:
121  --debug_pix(Y * 3 + X) <= to_signed(to_integer(pixel), c_comp_data_width + 1);
122  --debug_filter(Y * 3 + X) <= to_signed(factor, c_comp_data_width + 1);
123  --debug_res(Y * 3 + X) <= to_signed(result_int, c_comp_data_width + 1);
124 
125  -- Store result
126  line_acc(X) := to_signed(result_int, c_comp_data_width + 1);
127  end loop;
128  -- Accumulate horizontally
129  for N in line_acc'range(1) loop
130  acc := acc + line_acc(N);
131  end loop;
132  -- Writeback
133  applied_value_lines(Y) <= acc;
134  end loop;
135  end if;
136  end if;
137  end if;
138  end process;
139 
140  -- Second pipeline stage
141  sum_and_normalize : process(clk) is
142  constant half_output: signed(c_comp_data_width downto 0) := (0 => '1', others => '0');
143  variable acc : signed(c_comp_data_width downto 0);
144  variable result : std_logic_vector(c_comp_data_width downto 0);
145  begin
146  if rising_edge(clk) then
147  if reset = '1' then
148  data_out <= (others => '0');
149  else
150  if strobe = '1' then
151  acc := (others => '0');
152  -- Sum up intermediate results
153  for N in applied_value_lines'range(1) loop
154  acc := acc + applied_value_lines(N);
155  end loop;
156 
157  -- Add MAX_VALUE / 2 (if configured)
158  if NORMALIZE_TO_HALF = true then
159  acc := acc + shift_left(half_output, c_comp_data_width - 2);
160  end if;
161  -- Downscale and convert to unsigned (if configured)
162  if OUTPUT_SIGNED then
163  result := std_logic_vector(shift_right(acc, c_downscale_bits + 1));
164  else
165  acc(c_comp_data_width) := '0';
166  result := std_logic_vector(shift_right(acc, c_downscale_bits));
167  end if;
168  -- Output (registered)
169  data_out <= result(data_out'range);
170  end if;
171  end if;
172  end if;
173  end process;
174 
175 end architecture;

91  -- Slave register configuration:
92  -- Allows for "dynamic" configuration of slave registers
93  -- Possible values and what they mean:
94  -- "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
95  -- "01": From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
96  -- "10": From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
97  -- "11": Combined Read/Write register. Data transport in both directions.
98  -- When both sides attempt to write simultaniously, only the HW gets to write.
99  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
100 constant slave_register_configuration : slv_reg_config_table(0 to 2) :=
101  ("11","10","10");
102

138  vsync_gen : process(clk)
139  begin
140  if rising_edge(clk) then
141  if reset = '1' then
142  r_vsync <= '0';
143  r_frame_size_count <= to_signed(0, r_frame_size_count'length);
144  elsif r_enable = '1' then
145  if enable_sync_signals = '1' and STROBE_IN = '1' then
146  if r_frame_size_count - 1 < 0 then
147  r_vsync <= '1';
148  r_frame_size_count <= signed(reg_frame_size_config) - PIXELS_PER_STROBE;
149  else
150  r_vsync <= '0';
151  r_frame_size_count <= r_frame_size_count - to_signed(PIXELS_PER_STROBE, MEM_ADDRESS_BIT_WIDTH);
152  end if;
153  else
154  r_vsync <= '0';
155  end if;
156  end if;
157  end if;
158  end process;

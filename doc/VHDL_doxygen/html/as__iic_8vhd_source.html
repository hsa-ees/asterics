93  -- Slave register configuration:
94  -- Allows for "dynamic" configuration of slave registers
95  -- Possible values and what they mean:
96  -- "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
97  -- "01": From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
98  -- "10": From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
99  -- "11": Combined Read/Write register. Data transport in both directions.
100  -- When both sides attempt to write simultaniously, only the HW gets to write.
101  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
102  constant slave_register_configuration : slv_reg_config_table(0 to 1) :=
103  ("11","11");

302  scl_gen_logic: process(clk)
303  begin
304  if rising_edge(clk) then
305  -- Clear and increment logic for the scl counter register
306  if clear_sclcntr = '1' then
307  scl_counter <= (others => '0');
308  inc_lbclkcr <= '0';
309  elsif (scl_counter >= scl_div_local) and (enable_sclcntr = '1') then
310  scl_counter <= (others => '0');
311  inc_lbclkcr <= '1';
312  elsif enable_sclcntr = '1' then
313  scl_counter <= scl_counter_plusone;
314  inc_lbclkcr <= '0';
315  else scl_counter <= scl_counter;
316  inc_lbclkcr <= '0';
317  end if;
318 
319  -- Set local scl_div signal
320  scl_div_local <= scl_div_data_tx(SCL_DIV_REGISTER_WIDTH - 1 downto 0);
321 
322  -- Clear and increment logic for the local bus clock counter register
323  if clear_lbclkcr = '1' then
324  local_bus_clkcr <= "11";
325  elsif inc_lbclkcr = '1' then
326  local_bus_clkcr <= local_bus_clkcr_plusone;
327  else
328  local_bus_clkcr <= local_bus_clkcr;
329  end if;
330 
331  -- Synchronous state transitions
332  scl_curstate <= scl_nextstate;
333  end if;
334  end process;

342  scl_stalled <= '0';

348  else

350

352  when s_low =>
353  set_scl_low <= '1';
354  enable_sclcntr_scl <= '1';
355 
356  if lbclk = '1' then scl_nextstate <= s_high; end if;
357 
358  -- scl is high
359  when s_high =>
360  enable_sclcntr_scl <= '1';
361 
362  if lbclk = '0' then scl_nextstate <= s_low;
363  elsif lbclk = '1' and scl_in = '0' then scl_nextstate <= s_stalled; end if;
364 
365  -- scl is pulled low by the slave: clock stretching!
366  when s_stalled =>
367  scl_stalled <= '1';
368 
369  if scl_in = '1' then scl_nextstate <= s_high; end if;
370 
371  end case;
372  end if;
373  end process;

376  main_iic_logic : process(clk)
377  begin
378  if rising_edge(clk) then
379  -- Operations on the local data register
380  if clear_ldata = '1' then
381  ldata_next <= (others => '0');
382  elsif cp_data_to_ldata = '1' then
383  ldata_next <= data_in;
384  elsif set_ldata_lsb = '1' then
385  ldata_next(7 downto 1) <= local_data_register(7 downto 1);
386  ldata_next(0) <= sda_in;
387  elsif left_shift_ldata = '1' then
388  ldata_next(7 downto 1) <= local_data_register(6 downto 0);
389  ldata_next(0) <= '0';
390  else
391  ldata_next <= local_data_register;
392  end if;
393 
394  -- Clear and increment logic for the bit counter register
395  if clear_bitcntr = '1' then bit_counter <= "0001";
396  elsif inc_bitcntr = '1' then bit_counter <= bit_counter_plusone; end if;
397 
398  -- Data register driver
399  if cp_ldata_to_data = '1' then data_out <= local_data_register;
400  else data_out <= data_out; end if;
401 
402  -- Acknowledgement register logic
403  if clear_ackreg = '1' then slave_ackreg <= '1';
404  elsif slave_ack = '0' or slave_ackreg = '0' then slave_ackreg <= '0';
405  else slave_ackreg <= '1'; end if;
406 
407  -- Synchronous state transitions
408  iic_curstate <= iic_nextstate;
409  end if;
410  end process;

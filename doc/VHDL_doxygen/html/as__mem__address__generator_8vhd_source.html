51 entity AS_MEM_ADDRESS_GENERATOR is
52 generic (
53  ADDRESS_BIT_WIDTH : integer := 32; -- Data and bus address bit width
54  MEMORY_BIT_WIDTH : integer := 64;
55  BURST_LENGTH_WIDTH : integer := 12; -- Bit width of burst length (bus dependant)
56  SUPPORT_VARIABLE_BURST_LENGTH : boolean := false; -- Disable variable burst length (may only be set true if used with as_memreader)
57  SUPPORT_MULTIPLE_SECTIONS : boolean := false -- Disable having multiple sections (must not be true if used with as_memio)
58 );
59 port (
60  clk : in std_logic;
61  reset : in std_logic;
62 
63  go : in std_logic; -- Activate address generator
64  cnt_en : in std_logic; -- Increment current address by ADDRESS_BIT_WIDTH/8
65  ready : out std_logic; -- Signal others being ready for next operation
66 
67 -- Ports for configuration
68  section_addr : in std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Start address of memory data transaction
69  section_offset : in std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Offset in byte between to sections (start to start; don't care if SUPPORT_MULTIPLE_SECTIONS is set to false)
70  section_size : in std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Number of bytes to be transmitted (integer multiple of c_number_of_bytes_per_word)
71  section_count : in std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Number of sections (don't care if SUPPORT_MULTIPLE_SECTIONS is set to false)
72  max_burst_length : in std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Maximal burst length to be used (actual burst length might be smaller if SUPPORT_VARIABLE_BURST_LENGTH was set to true)
73 
74 -- Information for next bus access
75  memory_address : out std_logic_vector(ADDRESS_BIT_WIDTH-1 downto 0); -- Memory start address for next transaction
76  burst_length : out std_logic_vector(BURST_LENGTH_WIDTH-1 downto 0); -- Burst length for bus access (all zero if burst_enable = 0)
77  burst_enable : out std_logic -- Determines if a burst is possible (single-beat otherwise)
78 );
79 end AS_MEM_ADDRESS_GENERATOR;

83 architecture RTL of AS_MEM_ADDRESS_GENERATOR is
84 
85  -- Determine number of bytes of a word
86  constant c_number_of_bytes_per_word : natural := MEMORY_BIT_WIDTH/8;
87 
88  -- address generator statemachine signals
89  type address_gen_sm_state_t is (s_reset, s_idle, s_setup, s_cnt_active);
90  signal address_gen_sm_next_state, address_gen_sm_current_state : address_gen_sm_state_t;
91  signal s_counter_active : std_logic;
92 
93  -- signals for address calculation
94  signal s_address_calc_setup : std_logic;
95  signal r_current_address : unsigned(ADDRESS_BIT_WIDTH-1 downto 0);
96 
97  -- signals for total number of bytes left for read requests
98  signal s_byte_count_done : std_logic;
99  signal r_bytes_left : signed(ADDRESS_BIT_WIDTH-1 downto 0);
100 
101  -- signals for section calculation
102  signal s_section_count_done : std_logic;
103  signal r_current_section_count : signed(ADDRESS_BIT_WIDTH-1 downto 0);
104  signal r_current_section_offset : unsigned(ADDRESS_BIT_WIDTH-1 downto 0);
105 
106  -- signals for burst determination
107  signal s_actual_burst_length : std_logic_vector(BURST_LENGTH_WIDTH-1 downto 0);
108  signal s_burst_possible : std_logic;
109 
110 begin
111 
115  memory_address <= std_logic_vector(r_current_address);
116  burst_length <= s_actual_burst_length;
117  burst_enable <= s_burst_possible;
118 
121  address_gen_statemachine : process(address_gen_sm_current_state, go, s_section_count_done, reset)
122  begin
123 
124  address_gen_sm_next_state <= address_gen_sm_current_state;
125 
127  ready <= '0';
128  s_address_calc_setup <= '0';
129  s_counter_active <= '0';
130 
131  case address_gen_sm_current_state is
132 
135  when s_reset =>
136  address_gen_sm_next_state <= s_idle;
137 
140  when s_idle =>
141  ready <= '1';
142  if go = '1' then
143  address_gen_sm_next_state <= s_setup;
144  end if;
145 
149  when s_setup =>
150  ready <= '1';
151  s_address_calc_setup <= '1';
152  address_gen_sm_next_state <= s_cnt_active;
153 
157  when s_cnt_active =>
158  s_counter_active <= '1';
159  if s_section_count_done = '1' then
160  address_gen_sm_next_state <= s_idle;
161  end if;
162 
163  when others => address_gen_sm_next_state <= s_reset;
164  end case;
165  end process;
166 
168  process(clk)
169  begin
170  if rising_edge(clk) then
171  if reset = '1' then
172  address_gen_sm_current_state <= s_reset;
173  else
174  address_gen_sm_current_state <= address_gen_sm_next_state;
175  end if;
176  end if;
177  end process;

186  mem_address_calc : process(clk)
187  begin
188  if rising_edge(clk) then
189  if reset = '1' then
190  r_current_address <= to_unsigned(0, r_current_address'length);
191  else
192  -- Copy start address into shadow register for internal operation
193  if s_address_calc_setup = '1' then
194  r_current_address <= unsigned(section_addr);
195  -- Increment the current address by the number of bytes of a word (4 or 8) if the bus sends a data valid signal
196  elsif cnt_en = '1' and s_counter_active = '1' then
197  r_current_address <= r_current_address + to_unsigned(c_number_of_bytes_per_word, ADDRESS_BIT_WIDTH);
198  -- Increment the current address by the section size if previous section is done
199  elsif s_byte_count_done = '1' and s_counter_active = '1' and SUPPORT_MULTIPLE_SECTIONS = true then
200  r_current_address <= unsigned(section_addr) + r_current_section_offset;
201  end if;
202  end if;
203  end if;
204  end process;
205 
208  byte_counter : process(clk)
209  begin
210  if rising_edge(clk) then
211  if reset = '1' then
212  s_byte_count_done <= '0';
213  r_bytes_left <= to_signed(0, r_bytes_left'length);
214  else
215  -- Copy section_size configuration into shadow register for internal operation.
216  if s_address_calc_setup = '1' then
217  s_byte_count_done <= '0';
218  r_bytes_left <= signed(section_size);
219  -- Start calculation of bytes remaining for determining bus access type.
220  elsif s_counter_active = '1' then
221  -- Check if there are bytes left to be requested, otherwise signal 'done'.
222  if (r_bytes_left-1) < 0 then
223  s_byte_count_done <= '1';
224  r_bytes_left <= signed(section_size); -- reload r_bytes_left for next section
225  -- Decrement the number of remaining bytes by the number of bytes of a word, if the memory
226  -- bus sends a valid signal for the current data.
227  elsif cnt_en = '1' then
228  s_byte_count_done <= '0';
229  r_bytes_left <= r_bytes_left - to_signed(c_number_of_bytes_per_word, ADDRESS_BIT_WIDTH);
230  -- Set default to byte_counter not being done yet.
231  else
232  s_byte_count_done <= '0';
233  end if;
234  end if;
235  end if;
236  end if;
237  end process;
238 
243  section_counter : process(clk)
244  begin
245  if rising_edge(clk) then
246  if reset = '1' then
247  s_section_count_done <= '0';
248  r_current_section_count <= to_signed(0, r_current_section_count'length);
249  r_current_section_offset <= to_unsigned(0, r_current_section_offset'length);
250  else
251  -- Copy section_count and section_offset configuration into shadow registers for internal operation.
252  if s_address_calc_setup = '1' then
253  s_section_count_done <= '0';
254  if SUPPORT_MULTIPLE_SECTIONS = true then
255  r_current_section_count <= signed(section_count);
256  r_current_section_offset <= unsigned(section_offset);
257  else
258  r_current_section_count <= to_signed(1, r_current_section_count'length);
259  r_current_section_offset <= to_unsigned(0, r_current_section_offset'length);
260  end if;
261  -- Calculate the remaining number of sections and current section offset.
262  elsif s_counter_active = '1' then
263  -- Check if all sections have been processed
264  if (r_current_section_count-1) < 0 then
265  s_section_count_done <= '1';
266  -- Update offset to section start address and the number of remaining sections
267  elsif s_byte_count_done = '1' then
268  s_section_count_done <= '0';
269  r_current_section_count <= r_current_section_count - 1;
270  r_current_section_offset <= r_current_section_offset + unsigned(section_offset);
271  else
272  s_section_count_done <= '0';
273  end if;
274  end if;
275  end if;
276  end if;
277  end process;
278 
285  burst_calc : process(clk)
286  begin
287  if rising_edge(clk) then
288  if reset = '1' then
289  s_actual_burst_length <= (others => '0');
290  s_burst_possible <= '0';
291  else
292  -- Check if there are only enough bytes for a single beat left
293  if unsigned(r_bytes_left) < (to_unsigned(c_number_of_bytes_per_word, ADDRESS_BIT_WIDTH) + 1) then
294  s_actual_burst_length <= (others => '0');
295  s_burst_possible <= '0';
296  -- Check if configured max burst length is not sufficient for a burst (has to be at least two times the number of bytes per word)
297  elsif unsigned(max_burst_length) < (to_unsigned(c_number_of_bytes_per_word*2, ADDRESS_BIT_WIDTH)) then
298  s_actual_burst_length <= (others => '0');
299  s_burst_possible <= '0';
300  -- Determine burst length if variable burst length is configured
301  elsif SUPPORT_VARIABLE_BURST_LENGTH = true then
302  -- If the number of remaining bytes is less than max burst length, use remaining bytes as burst length
303  if unsigned(r_bytes_left) < unsigned(max_burst_length) then
304  s_actual_burst_length <= std_logic_vector(r_bytes_left(BURST_LENGTH_WIDTH-1 downto 0));
305  s_burst_possible <= '1';
306  -- Otherwise, use max burst length
307  else
308  s_actual_burst_length <= max_burst_length(BURST_LENGTH_WIDTH-1 downto 0);
309  s_burst_possible <= '1';
310  end if;
311  -- Otherwise, allow burst only when the number of remaining bytes is greater or equal to max burst length
312  else
313  if unsigned(r_bytes_left) < unsigned(max_burst_length) then
314  s_actual_burst_length <= (others => '0');
315  s_burst_possible <= '0';
316  else
317  s_actual_burst_length <= max_burst_length(BURST_LENGTH_WIDTH-1 downto 0);
318  s_burst_possible <= '1';
319  end if;
320  end if;
321  end if;
322  end if;
323  end process;
324 
325 end architecture;

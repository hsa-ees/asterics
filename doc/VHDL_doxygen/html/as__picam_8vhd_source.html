102  -- Slave register configuration:
103  -- Allows for "dynamic" configuration of slave registers
104  -- Possible values and what they mean:
105  -- "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
106  -- "01": From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
107  -- "10": From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
108  -- "11": Combined Read/Write register. Data transport in both directions.
109  -- When both sides attempt to write simultaniously, only the HW gets to write.
110  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
111  constant slave_register_configuration : slv_reg_config_table(0 to 0) := (0 => "11");
112  -- State and control signals
113  signal control : std_logic_vector(15 downto 0);
114  signal state : std_logic_vector(15 downto 0);
115  signal control_new : std_logic_vector(15 downto 0);
116  signal control_reset : std_logic_vector(15 downto 0);
117 
118  constant c_data_counter_bits : natural := integer(ceil(log2(real(RES_X*RES_Y))));
119  constant c_frame_width : signed(31 downto 0) := to_signed(RES_X, 32);

147 -- Handle the control_reset signal
148 -- and set the modify bit for the status and control register, as necessary
149 status_control_update_logic: process(control, control_reset, slv_ctrl_reg, state, reset)
150  variable var_control_new : std_logic_vector(15 downto 0);
151 begin
152  -- Reset control and status register modify bit
153  slv_reg_modify(0) <= '0';
154 
155  -- Clear control bits of the register on module reset
156  if reset = '1' then
157  control_new <= (others => '0');
158  else
159  control_new <= control;
160  end if;
161 
162  -- Apply control_reset bit mask
163  var_control_new := control and (not control_reset);
164 
165  -- If either control or state was modified by hardware, set modify bit
166  if control /= var_control_new then
167  control_new <= var_control_new;
168  slv_reg_modify(0) <= '1';
169  end if;
170  if state /= slv_ctrl_reg(0)(15 downto 0) then
171  slv_reg_modify(0) <= '1';
172  end if;
173 end process status_control_update_logic;

177 run_once <= control(2);

180 control_reset(2) <= run_once_reset;

189 frame_done_process : process(Clk)
190  variable p_data_counter : unsigned(c_data_counter_bits-1 downto 0);
191 begin
192  if (Clk'event and Clk = '1') then
193 
194  run_last <= run;
195 
196  if ( (Reset = '1') or
197  -- reset counter for subsequent frames (free run mode)
198  ( (s_axis_tvalid = '1') and ( s_axis_tuser = '1') ) or
199  -- reset counter for first frame, especially important for single shot mode
200  -- (else the old counter value would persist until a new frame arrives)
201  ( ( run_last = '0' ) and ( run = '1' ) ) ) then
202  p_data_counter := to_unsigned(1, p_data_counter'length);
203  s_state_reg_frame_done <= '0';
204  s_state_reg_frame_done_last <= '0';
205 
206  elsif ( s_axis_tvalid = '1') then
207  p_data_counter := p_data_counter + 1;
208  end if;

211  s_state_reg_frame_done <= '1';

255 -- Convert 32 bit ABGR image data to 8 bit grayscale
256 grayscale_conv : process (clk)
257 variable p_red_result: unsigned (7 downto 0);
258 variable p_green_result: unsigned (7 downto 0);
259 variable p_blue_result: unsigned (7 downto 0);
260 variable p_result: unsigned (7 downto 0);
261 begin
262 if rising_edge (clk) then
263  if reset = '1' then
264  r_data_out <= (others => '0');
265  else
266  p_result := (others => '0');
267  p_blue_result := (others => '0');
268  p_green_result := (others => '0');
269  p_red_result := (others => '0');
270  p_blue_result := (unsigned(s_axis_tdata(23 downto 16)) srl 4) + (unsigned(s_axis_tdata(7 downto 0)) srl 5);
271  p_green_result := (unsigned(s_axis_tdata(15 downto 8)) srl 1) + (unsigned(s_axis_tdata(15 downto 8)) srl 4);
272  p_red_result := (unsigned(s_axis_tdata(7 downto 0)) srl 2) + (unsigned(s_axis_tdata(23 downto 16)) srl 5);
273  p_result := p_red_result + p_green_result + p_blue_result;
274  r_data_out <= std_logic_vector(p_result);
275  end if;
276 end if;
277 end process;

279 -- Generate HSYNC signal at the beginning of every line

281 hsync_gen : process(clk)

284  if reset = '1' then
285  s_hsync_out <= '0';

289  if (s_axis_tvalid = '1') then

291  s_hsync_out <= '1';
292  r_x_resolution_count <= c_frame_width - 1;
293  else
294  s_hsync_out <= '0';
295  r_x_resolution_count <= r_x_resolution_count - to_signed(1, 32);
296  end if;
297  else
298  s_hsync_out <= '0';
299  end if;
300  else
301  s_hsync_out <= '0';
302  end if;
303  end if;
304 end process;

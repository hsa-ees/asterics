56 entity as_regmgr is

108  port(

111  -- Possible values (from HW view):
112  -- "00" -> No register,
113  -- "01" -> Write-Only (HW -> SW),

126  constant c_reg_addr_width : integer := REG_ADDR_WIDTH - MODULE_ADDR_WIDTH;
127 
128  -- Collection of the data_in_enable inputs of the registers
129  signal arr_sw_data_in_ena : std_logic_vector(0 to REG_COUNT - 1);
130 
131  -- sw_data_out <= register contents
132  -- Collection of data outputs from all registers of this module
133  signal arr_sw_data_out : slv_reg_data(0 to REG_COUNT - 1);
134 
135  signal reg_modify_arr : std_logic_vector(0 to REG_COUNT - 1);
136  signal reg_content : slv_reg_data(0 to REG_COUNT - 1);

138 begin
139 
140  -- Handle requests:
141  reg_address_decode: process(sw_address, sw_data_in_ena, arr_sw_data_out, sw_data_out_ena, slv_reg_config)
142  variable module_address : integer;
143  variable register_number : integer;
144  begin
145 
146  -- Set all enable signals to '0'
147  arr_sw_data_in_ena <= (others => '0');
148  -- Clear data output to software
149  sw_data_out <= (others => '0');
150 
151  -- Split address into module address and register address:
152  module_address := to_integer(unsigned(
153  sw_address(REG_ADDR_WIDTH - 1 downto
154  c_reg_addr_width)));
155  register_number := to_integer(unsigned(
156  sw_address(c_reg_addr_width - 1 downto 0)));
157  -- On modify request:
158  if sw_data_in_ena = '1' then
159 
160  -- Check module address - is this the addressed module?
161  if module_address = MODULE_BASEADDR and register_number < REG_COUNT then
162 
163  -- Make sure the addressed register is modifiable by software
164  if slv_reg_config(register_number)(1) = '1' then
165  -- Set data in enable for the addressed register
166  arr_sw_data_in_ena(register_number) <= '1';
167  else
168  -- For simulation purposes: Complain on "failed" write by software
169  report("Out of bounds write detected for register number: " & integer'image(register_number)
170  & " of module with address: " & integer'image(module_address));
171  end if;
172  end if;
173 
174  -- On data request by software
175  elsif sw_data_out_ena = '1' then
176 
177  -- Check module address - is this the addressed module?
178  if module_address = MODULE_BASEADDR and register_number < REG_COUNT then
179 
180  -- Make sure the addressed register is readable by software
181  if slv_reg_config(register_number)(0) = '1' then
182  -- Set the sw_data_out port to the addressed registers data output
183  sw_data_out <= arr_sw_data_out(register_number);
184  else
185  -- For simulation purposes: Complain on "failed" read by software
186  report("Out of bounds read detected for register number: " & integer'image(register_number)
187  & " of module with address: " & integer'image(module_address));
188  end if;
189  end if;
190  end if;
191  end process reg_address_decode;
192 
193  slv_ctrl_reg <= reg_content;
194 
195  -- Register generator:
196 
197  -- Auto modify for bidirectional registers:
198 
199  -- Automatically set the hardware write enable bit when
200  -- the value from hardware changes (ie differs from the value
201  -- sent to hardware ^= the value stored in the register)
202  p_auto_mod_enable: process(slv_status_reg, reg_content, arr_sw_data_in_ena, slv_reg_config, slv_reg_modify) is
203  begin
204  for N in 0 to REG_COUNT - 1 loop
205  if ((AUTO_HW_REG_MODIFY_BIT = True) and (slv_reg_config(N) = AS_REG_BOTH)) then
206  if slv_status_reg(N) /= reg_content(N) then
207  reg_modify_arr(N) <= '1';
208  elsif arr_sw_data_in_ena(N) = '1' then
209  reg_modify_arr(N) <= '0';
210  else
211  reg_modify_arr(N) <= '0';
212  end if;
213  else
214  reg_modify_arr(N) <= slv_reg_modify(N);
215  end if;
216  end loop;
217  end process;
218 
219  -- Generate the maximum register number according to the register configuration table
220  -- Registers configured as "00" should be optimized out during optimization phases of synthesis
221  module_registers: for N in 0 to REG_COUNT - 1 generate
222 
223  generic_register_slice: as_generic_regslice
224  generic map(
225  REG_DATA_WIDTH => REG_DATA_WIDTH
226  )
227  port map(
228  clk => clk,
229  rst_n => reset_n,
230  register_behaviour => slv_reg_config(N),
231  sw_data_in => sw_data_in,
232  sw_data_out => arr_sw_data_out(N),
233  sw_byte_mask => sw_byte_mask,
234  sw_data_in_ena => arr_sw_data_in_ena(N),
235  hw_data_in => slv_status_reg(N),
236  hw_data_out => reg_content(N),
237  hw_modify => reg_modify_arr(N));
238 
239  end generate;

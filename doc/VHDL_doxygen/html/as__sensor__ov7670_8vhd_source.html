106  -- Slave register configuration:
107  -- Allows for "dynamic" configuration of slave registers
108  -- Possible values and what they mean:
109  -- "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
110  -- "01": HW -> SW, Status register only. Data transport from hardware to software. HW can only write, SW can only read.
111  -- "10": HW <- SW, Control register only. Data transport from software to hardware. HW can only read, SW can only write.
112  -- "11": HW <=> SW, Combined Read/Write register. Data transport in both directions. ! Higher FPGA resource utilization !
113  -- When both sides attempt to write simultaniously, only the HW gets to write.
114  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
115  constant slave_register_configuration : slv_reg_config_table(0 to 1) := ("11","10");

175  status_control_update_logic: process(control, control_reset, slv_ctrl_reg, state, reset)
176  variable var_control_new : std_logic_vector(15 downto 0);
177  begin
178  -- Reset control and status register modify bit
179  slv_reg_modify(0) <= '0';
180 
181  -- Clear control bits of the register on module reset
182  if reset = '1' then
183  control_new <= (others => '0');
184  else

188  -- Apply control_reset bit mask
189  var_control_new := control and (not control_reset);

192  if control /= var_control_new then
193  control_new <= var_control_new;
194  slv_reg_modify(0) <= '1';
195  end if;

197  slv_reg_modify(0) <= '1';

328 DATA_COUNTER_PROCESS: process(Clk)
329  variable data_counter : unsigned(DATA_COUNTER_BITS-1 downto 0);
330 begin
331  if (Clk'event and Clk = '1') then
332 
333  run_last <= run;
334 
335  if ( (Reset = '1') or
336  (softreset = '1') or
337  ( (s_stobe_out = '1') and ( s_vsync_out = '1') ) or -- reset counter for subsequent frames (free run mode)
338  ( ( run_last = '0' ) and ( run = '1' ) ) ) then -- reset counter for first frame, especially important for single shot mode (else the old counter value would persist until a new frame arrives)
339  data_counter := to_unsigned(1, data_counter'length);
340  state_reg_frame_done <= '0';
341  state_reg_frame_done_last <= '0';
342 
343  elsif ( s_stobe_out = '1' ) then
344  data_counter := data_counter + 1;
345  end if;
346 
347  if ( data_counter = to_unsigned(RES_X*RES_Y, data_counter'length) ) then
348  state_reg_frame_done <= '1';
349  end if;
350 
351  state_reg_frame_done_last <= state_reg_frame_done;
352  end if;
353 end process;

419 -- ==============================================================================
420 -- Generate VSYNC and HSYNC signals (1 clock cycle delay to data, see above)
421 -- ==============================================================================
422 GENERATE_SYNCS : process(Sensor_PixClk)
423 variable v_FRAME_DETECT : std_logic;
424 begin
425  if ( Sensor_PixClk'event and Sensor_PixClk = '1' ) then
426  if ( Reset_Sensor_PixClk = '1' ) then
427  Ready_Sensor_PixClk <= '0';
428 
429  FRAME_DETECT <= '0';
430  FRAME_DETECT_last <= '0';
431 
432  s_Sensor_Frame_Valid_last <= '1';
433  s_Sensor_Line_Valid_last <= '1';
434 
435  VSYNC_gen <= '0';
436  HSYNC_gen <= '0';
437 
438  else
439  VSYNC_gen <= '0';
440  HSYNC_gen <= '0';
441 
442  -- drop every second arriving value (pixel format is Y-Cr-Y-Cb-... and we just want Y values, Cr and Cb have to be dropped)
443  s_drop <= not(s_drop);
444 
445  -- Save recent Frame/Line_Valid for next cycle to detect a change (rising/falling edge):
446  s_Sensor_Frame_Valid_last <= s_Sensor_Frame_Valid;
447  s_Sensor_Line_Valid_last <= s_Sensor_Line_Valid;
448 
449 
450  -- Recognize the frame start (which may happen not at the same clock cycle of line change):
451  if ((s_Sensor_Frame_Valid_last = '0') and (s_Sensor_Frame_Valid = '1')) then
452  v_FRAME_DETECT := not(FRAME_DETECT);
453  -- Store frame detection information:
454  FRAME_DETECT <= v_FRAME_DETECT;
455  end if;
456 
457  -- Detect a line change (-> HSYNC generated, optionally genrate VSYNC if also frame valid changes or changed beforehand)
458  if (s_Sensor_Line_Valid_last = '0' and s_Sensor_Line_Valid = '1') then
459 
460  HSYNC_gen <= '1';
461  s_drop <= '0'; -- never drop first pixel as it is accompanied by sync information. If pixels arrive in unpleasant order, change behaviour in the sensor config (I2C sensor config).
462 
463  -- Generate VSYNC if there's a frame change signalled in the recent cycle or beforehand:
464  if (((v_FRAME_DETECT xor FRAME_DETECT_last) or (FRAME_DETECT xor FRAME_DETECT_last)) = '1') then
465  VSYNC_gen <= '1';
466  Ready_Sensor_PixClk <= '1'; -- we are ready when the first frame comes in
467  end if;
468 
469  -- Handle frame change evaluation (VS happened before HS):
470  FRAME_DETECT_last <= v_FRAME_DETECT;
471 
472  end if;
473 
474  end if;
475  end if;
476 end process;

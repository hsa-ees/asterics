56 entity as_sim_file_reader is
57  generic(
58  SIM_FILE_NAME : string := "./tb/images/data_in.dat";
59  REGISTER_BIT_WIDTH : integer := 32; -- default bit width of slave registers (for configuration)
60  DOUT_WIDTH : integer := 32; -- default bit width of data (has to be equal to MEMORY_DATA_WIDTH)
61  MEMORY_DATA_WIDTH : integer := 32; -- default bit width of bus connections
62  MEM_ADDRESS_BIT_WIDTH : integer := 32; -- default bit width of memory address
63  BURST_LENGTH_BIT_WIDTH : integer := 12; -- default bit width of burst length setting (bus dependant)
64  FIFO_FILL_BIT_WIDTH : integer := 10; -- default bit width for fifo fill level
65  MAX_PLATFORM_BURST_LENGTH : integer := 256; -- max. supported burst length in byte by platform
66  SUPPORT_MULTIPLE_SECTIONS : boolean := true; -- has to be false if used for as_memio
67  SUPPORT_VARIABLE_BURST_LENGTH : boolean := true; -- Support smaller burst lengths if possible.
68  ENABLE_BIG_ENDIAN : boolean := true -- Enables big endian data interpretation
69  );
70  port(
71  clk : in std_logic;
72  reset : in std_logic;
73  ready : out std_logic;
74 
75  strobe_out : out std_logic;
76  data_out : out std_logic_vector(DOUT_WIDTH-1 downto 0);
77  stall_in : in std_logic;
78 
79  control : in std_logic_vector(15 downto 0);
80  control_reset : out std_logic_vector(15 downto 0);
81  state : out std_logic_vector(15 downto 0);
82 
83  reg_section_addr : in std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0);
84  reg_section_offset : in std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0); -- not implemented yet
85  reg_section_size : in std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0);
86  reg_section_count : in std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0); -- not implemented yet
87  reg_max_burst_length : in std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0); -- not needed in simulation
88  reg_current_hw_addr : out std_logic_vector(REGISTER_BIT_WIDTH-1 downto 0)
89  );
90 end as_sim_file_reader;

92 architecture SIM of as_sim_file_reader is
93 
94  constant c_number_of_mem_fields : natural := DOUT_WIDTH/c_memory_bit_width;
95  constant init_val : std_logic_vector(7 downto 0) := "00000000";
96 
97  shared variable mem_read: sim_ram;
98 
99  type t_state is (s_idle, s_read, s_done);
100  signal r_state : t_state;
101 
102  signal r_cur_addr : natural;
103  signal s_data_count : integer;
104 
105  signal r_reg_section_addr : integer := 0;
106  signal r_reg_section_size : integer := 0;
107 
108  signal file_start_addr : integer := 0;
109  signal read_whole_file : integer := 0;
110 
111  signal s_copy_input : std_logic;
112 
113 begin
114 
115  reg_current_hw_addr <= std_logic_vector(to_unsigned(r_cur_addr,reg_current_hw_addr'length));
116 
117  control_reset <= (others=>'0');
118 
119 
120  p_read_file : process(clk, reset)
121 
122  variable success : std_logic;
123  variable v_addr : integer;
124  variable v_data : std_logic_vector(c_memory_bit_width-1 downto 0);
125 
126  begin
127  if rising_edge(clk) then
128  if reset = '1' or control(0) = '1' then
129  STROBE_OUT <= '0';
130  DATA_OUT <= (others => '0');
131  r_cur_addr <= 0;
132  r_state <= s_idle;
133  state(0) <= '0'; -- DONE
134  state(1) <= '1'; -- BUSY
135  state(2) <= '0'; -- ERROR
136  s_copy_input <= '0';
137  -- Read in whole file during reset, both values have to be zero -> Results in reading whole file
138  report "mem_read.p_read_from_disk start..." severity note;
139  mem_read.p_read_from_disk(SIM_FILE_NAME, file_start_addr, read_whole_file);
140  report "mem_read.p_read_from_disk finished." severity note;
141  else
142  -- Default
143  s_copy_input <= '0';
144  STROBE_OUT <= '0';
145  DATA_OUT <= (others => '0');
146  state(0) <= '0'; -- DONE
147  state(1) <= '0'; -- BUSY
148  state(2) <= '0'; -- ERROR
149  case r_state is
150  when s_idle =>
151  state(0) <= '1'; -- DONE
152  state(1) <= '0'; -- BUSY
153  state(2) <= '0'; -- ERROR
154  r_reg_section_addr <= to_integer(unsigned(reg_section_addr));
155  r_reg_section_size <= to_integer(unsigned(reg_section_size));
156  s_data_count <= 0;
157  if control(1) = '1' then
158  r_state <= s_read;
159  end if;
160 
161  when s_read =>
162  if STALL_IN = '0' then
163  state(0) <= '0'; -- DONE
164  state(1) <= '1'; -- BUSY
165  state(2) <= '0'; -- ERROR
166  if s_data_count = (r_reg_section_size) then -- - c_number_of_mem_fields) then
167  r_state <= s_done;
168  else
169  STROBE_OUT <= '1';
170 
171  for i in c_number_of_mem_fields - 1 downto 0 loop
172  v_addr := r_reg_section_addr + s_data_count + i;
173  mem_read.p_read_data(v_addr, v_data, success);
174  if ENABLE_BIG_ENDIAN then
175  DATA_OUT(DOUT_WIDTH-c_memory_bit_width*i-1 downto DOUT_WIDTH-(c_memory_bit_width+c_memory_bit_width*i)) <= v_data;
176  else
177  DATA_OUT(c_memory_bit_width*(i+1)-1 downto c_memory_bit_width*i) <= v_data;
178  end if;
179  end loop;
180  s_data_count <= s_data_count + c_number_of_mem_fields;
181  end if;
182  end if;
183 
184  when s_done =>
185  state(0) <= '1'; -- DONE
186  state(1) <= '1'; -- BUSY
187  state(2) <= '0'; -- ERROR
188  STROBE_OUT <= '0';
189  r_cur_addr <= 0;
190  DATA_OUT <= (others => '0');
191 
192  r_state <= s_idle;
193  end case;
194  end if;
195  end if;
196 
197  end process p_read_file;
198 
199 end architecture;

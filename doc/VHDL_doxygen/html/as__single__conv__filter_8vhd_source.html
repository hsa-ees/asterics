102  -- Slave register configuration:
103  -- Allows for "dynamic" configuration of slave registers
104  -- Possible values and what they mean:
105  -- AS_REG_NONE / "00": Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
106  -- AS_REG_STATUS / "01": From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
107  -- AS_REG_CONTROL / "10": From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
108  -- AS_REG_BOTH / "11": Combined Read/Write register. Data transport in both directions.
109  -- When both sides attempt to write simultaniously, only the HW gets to write.
110  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.
111  constant slave_register_configuration : slv_reg_config_table(0 to 0) := (0 => AS_REG_BOTH);
112

143  slv_status_reg(0)(17 downto 16) <= status_reg_int;
144  status_reg_int <= (0 => ready_internal,
145  1 => stall_in);
146  slv_reg_modify(0) <= reg_update_enable;
147 
148  -- Update the status register
149  reg_update : process(clk) is
150  begin
151  if rising_edge(clk) then
152  if reset = '1' then
153  slv_status_reg(0)(31 downto 18) <= (others => '0');
154  slv_status_reg(0)(15 downto 0) <= (others => '0');
155  reg_update_enable <= '1';
156  else
157  reg_update_enable <= '0';
158  if flush_in = '1' and ready_internal = '0' then
159  slv_status_reg(0)(0) <= '0';
160  reg_update_enable <= '1';
161  end if;
162  if slv_ctrl_reg(0)(17 downto 16) /= status_reg_int then
163  reg_update_enable <= '1';
164  end if;
165  end if;
166  end if;
167  end process;

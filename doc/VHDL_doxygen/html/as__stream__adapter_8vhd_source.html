103  -- Slave register configuration:
104  -- Allows for "dynamic" configuration of slave registers
105  -- Possible values and what they mean:
106  -- "00", AS_REG_NONE: Register "off". Register will not be available and won't be implemented -> No hardware resource utilization.
107  -- "01", AS_REG_STATUS: From HW view: Status register. Data transport from hardware to software. HW can only write, SW can only read.
108  -- "10", AS_REG_CONTROL: From HW view: Control register. Data transport from software to hardware. HW can only read, SW can only write.
109  -- "11", AS_REG_BOTH: Combined Read/Write register. Data transport in both directions.
110  -- When both sides attempt to write simultaniously, only the HW gets to write.
111  -- These registers use both the slv_ctrl_reg and slv_status_reg ports for communication.

410  p_buffer_io : process(clk) is
411  variable next_in_buf_idx : integer;
412  variable next_out_buf_idx : integer;
413  variable v_load_data : std_logic_vector(DIN_WIDTH - 1 downto 0);
414 
415  variable v_stall_buffer_providing : std_logic;
416  variable v_stall_buffer_storing : std_logic;
417  begin
418  if rising_edge(clk) then
419  if reset_int = '1' then
420  data_buffer <= (others => '0');
421  in_buf_idx <= (others => '0');
422  out_buf_idx <= (others => '0');
423  hsync_buf_idx <= to_unsigned(c_sync_off, c_idx_sync_width);
424  vsync_buf_idx <= to_unsigned(c_sync_off, c_idx_sync_width);
425 
426  s_stall_buffer_state <= sbs_empty;
427  stall_buffer_data_reg <= (others => '0');
428  stall_buffer_sync_reg <= (others => '0');
429  v_stall_buffer_providing := '0';
430  v_stall_buffer_storing := '0';
431  sync_error <= '0';
432  else
433 
434  s_stall_buffer_state <= s_stall_buffer_state;
435  stall_buffer_data_reg <= stall_buffer_data_reg;
436  stall_buffer_sync_reg <= stall_buffer_sync_reg;
437  v_stall_buffer_providing := '0';
438  v_stall_buffer_storing := '0';
439  sync_error <= '0';
440 
441  case s_stall_buffer_state is
442 
443  -- The stall buffer is empty
444  -- Stores a data word if a strobe is received while we are sending a stall signal
445  -- Then move to either sbs_full, if STALL_BUFFER_SIZE == 1 else sbs_has_data
446  when sbs_empty =>
447  if stall_in_int = '1' and strobe_in = '1' and buff_empty_int = '0' then
448  v_stall_buffer_storing := '1';
449 
450  s_stall_buffer_state <= sbs_full;
451  end if;
452  -- The stall buffer is full
453  when sbs_full =>
454  -- Behaviour for: New data arriving
455  if strobe_in = '1' then
456  -- If we cannot output while the buffer is full, raise a sync_error!
457  if buff_full = '1' then
458  sync_error <= '1';
459  else -- Output data from buffer ...
460  v_stall_buffer_providing := '1';
461  v_stall_buffer_storing := '1';
462  end if;
463 
464  -- Behaviour for: No new data, output data to pipeline
465  elsif buff_full = '0' and strobe_in = '0' then
466  v_stall_buffer_providing := '1';
467  s_stall_buffer_state <= sbs_empty;
468  end if;
469 
470  end case;
471 
472 
473  -- Data buffer management
474  if v_stall_buffer_storing = '1' then
475  -- Store new data, increment stored data count
476  stall_buffer_data_reg <= data_in;
477  stall_buffer_sync_reg(0) <= hsync_in;
478  stall_buffer_sync_reg(1) <= vsync_in;
479  end if;
480 
481  stall_buffer_providing <= v_stall_buffer_providing;
482 
483 
484  -- !Outside of if clause to provide up-to-date values within the output if clause!
485  next_in_buf_idx := to_integer(in_buf_idx);
486 
487  -- On data input:
488  if (strobe_in = '1' and stall_in_int = '0') or v_stall_buffer_providing = '1' or (strobe_in = '1' and buff_empty_int = '1') then
489  if v_stall_buffer_providing = '1' then
490  v_load_data := stall_buffer_data_reg;
491  else
492  v_load_data := data_in;
493  end if;
494  -- Store the incoming data word in the data buffer
495  data_buffer(
496  to_integer(in_buf_idx) + DIN_WIDTH - 1
497  downto to_integer(in_buf_idx)
498  ) <= v_load_data;
499 
500  -- Calculate the location of the next incoming data word and update the register.
501  -- Note: 'DIN_WIDTH' is guaranteed to evenly divide 'c_buffer_width'
502  next_in_buf_idx := (to_integer(in_buf_idx) + DIN_WIDTH) mod c_buffer_width;
503  in_buf_idx <= to_unsigned(next_in_buf_idx, c_idx_width);
504 
505  -- For sync signals: Store the 'in buffer idx' when the signal ocurred
506  if ((hsync_in = '1')
507  or (v_stall_buffer_providing = '1'
508  and stall_buffer_sync_reg(0) = '1')
509  ) then
510 
511  if next_in_buf_idx = 0 then
512  hsync_buf_idx <= to_unsigned(c_buffer_width - 1, c_idx_sync_width);
513  else
514  hsync_buf_idx <= to_unsigned(next_in_buf_idx, c_idx_sync_width);
515  end if;
516  end if;
517 
518  if ((vsync_in = '1')
519  or (v_stall_buffer_providing = '1'
520  and stall_buffer_sync_reg(1) = '1')
521  ) then
522  if next_in_buf_idx = 0 then
523  vsync_buf_idx <= to_unsigned(c_buffer_width - 1, c_idx_sync_width);
524  else
525  vsync_buf_idx <= to_unsigned(next_in_buf_idx, c_idx_sync_width);
526  end if;
527  end if;
528  end if;
529 
530  strobe_out_int <= '0';
531  hsync_out <= '0';
532  vsync_out <= '0';
533 
534  -- Output when enough data is stored for output or the buffer is full
535  -- Also listen to the stall signal, only output if it is low!
536  if (next_in_buf_idx >= to_integer(out_buf_idx) + DOUT_WIDTH or buff_full = '1' or v_stall_buffer_providing = '1')
537  and stall_out = '0' then
538  -- Every time we have enough data stored to output a data word, send 'strobe_out_int' for one cycle
539  strobe_out_int <= '1';
540 
541  -- Calculate the next 'out buffer idx' and update the register
542  -- Note: 'DOUT_WIDTH' is guaranteed to evenly divide 'c_buffer_width'
543  next_out_buf_idx := (to_integer(out_buf_idx) + DOUT_WIDTH) mod c_buffer_width;
544  out_buf_idx <= to_unsigned(next_out_buf_idx, c_idx_width);
545 
546  -- For sync signals: When we reach a 'out buffer idx' higher or equal to the stored
547  -- 'in buffer idx' of when we got the sync signal, send the signal on the output
548  -- Edge case: The next 'out buffer idx' is 0, i.e. the register rolls over
549  -- before it reaches a higher or equal value to the stored 'in buffer idx'.
550  -- In that case, also send the sync signal.
551 
552  -- For 'hsync_out':
553  if next_out_buf_idx >= hsync_buf_idx
554  or (next_out_buf_idx = 0
555  and buff_full = '1'
556  and hsync_buf_idx < c_sync_off
557  ) then
558  hsync_out <= '1';
559  -- Reset the stored buffer index to the maximum buffer width (to disable signal output)
560  hsync_buf_idx <= to_unsigned(c_sync_off, c_idx_sync_width);
561  end if;
562 
563  -- For 'vsync_out':
564  if next_out_buf_idx >= vsync_buf_idx
565  or (next_out_buf_idx = 0
566  and buff_full = '1'
567  and vsync_buf_idx < c_sync_off
568  ) then
569  vsync_out <= '1';
570  -- Reset the stored buffer index to the maximum buffer width (to disable signal output)
571  vsync_buf_idx <= to_unsigned(c_sync_off, c_idx_sync_width);
572  end if;
573 
574  end if;
575  end if;
576  end if;
577  end process;
578 
579  -- Data output:
580  -- As the 'out buffer idx' lags behind by one clock cycle,
581  -- it will be updated by the time strobe is generated
582  -- Therefore it defines the end of the data word to output:
583  -- (out_buf_idx - 1 downto out_buf_idx - DOUT_WIDTH)
584  -- instead of (out_buf_idx + DOUT_WIDTH - 1 downto out_buf_idx), as indexing for data input works, as it is handled BEFORE the idx update.
585  -- Edge case: When the buffer rolls over, we need to output the last possible data word at the high end of the buffer.
586  -- As 'out buffer idx' is zero in that case, the before indexing scheme does not work, hence the extra case.
587  data_out_int <= (others => '0') when reset_int = '1'
588  else data_buffer(to_integer(out_buf_idx) - 1 downto to_integer(out_buf_idx) - DOUT_WIDTH) when out_buf_idx > 0
589  else data_buffer(c_buffer_width - 1 downto c_buffer_width - DOUT_WIDTH); -- For rolled over buffer
590  end generate;

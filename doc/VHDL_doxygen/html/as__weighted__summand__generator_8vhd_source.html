50 entity as_weighted_summand_generator is
51  generic(
52  -- Input pixel bit width
53  DIN_WIDTH : integer := 8;
54  -- Number of input channels
55  CHANNEL_COUNT : integer := 1;
56  -- Side length of the square kernel
57  KERNEL_SIZE : integer := 3;
58  -- Kernel values in data format: image channel, height, width
59  KERNEL_VALUES : t_generic_filter;
60  -- Total number of summands expected
61  SUMMAND_COUNT : integer := 148;
62  -- Bit width of summands in order: image channel, height, width, weight factors (small -> large)
63  SUMMAND_BIT_WIDTHS : t_integer_array;
64  LARGEST_SUMMAND_BIT_WIDTH : integer
65  );
66  port (
67  -- Current image window to convolve
68  window_in : in t_generic_window(0 to KERNEL_SIZE - 1, 0 to KERNEL_SIZE - 1, (DIN_WIDTH * CHANNEL_COUNT) - 1 downto 0);
69  -- Summands from convolution with the CURRENT image window (no clock delay!)
70  data_out : out std_logic_vector(SUMMAND_COUNT * LARGEST_SUMMAND_BIT_WIDTH - 1 downto 0)
71  );
72 end as_weighted_summand_generator;

76 architecture RTL of as_weighted_summand_generator is
77 
78  constant c_weights_per_channel : integer := KERNEL_SIZE * KERNEL_SIZE;
79  constant c_total_weights_in_kernel : integer := c_weights_per_channel * CHANNEL_COUNT;
80  -- Array of weight factors (powers of two) per weight of the convolution kernel
81  constant c_elements_per_weight : t_integer_array(0 to c_total_weights_in_kernel - 1)
82  := f_get_element_count_for_weights(KERNEL_VALUES);
83  -- Output array
84  type t_signed_array is array(0 to SUMMAND_COUNT - 1) of signed(LARGEST_SUMMAND_BIT_WIDTH - 1 downto 0);
85  signal unsorted_elements : t_signed_array;
86 
87 begin
88 
89  -- For all channels
90  gen_produce_summands : for N in 0 to CHANNEL_COUNT - 1 generate
91  -- Input window (single channel)
92  signal crt_window_N : t_generic_window(0 to KERNEL_SIZE - 1,
93  0 to KERNEL_SIZE - 1,
94  DIN_WIDTH - 1 downto 0) := (others => (others => (others => '0')));
95 
96  begin
97  -- Cut the input window along the channel dimension (extract a single channel)
98  crt_window_N <= f_cut_vectors_of_generic_window(window_in, N * DIN_WIDTH, DIN_WIDTH);
99 
100  -- For each row of pixels in each channel
101  gen_produce_summands_N : for Y in 0 to KERNEL_SIZE - 1 generate
102 
103  -- For each pixel in each row
104  gen_produce_summands_NY : for X in 0 to KERNEL_SIZE - 1 generate
105 
106  -- Number of the current weight (if they were in a 1D array)
107  constant c_current_index_NYX : integer := (N * (c_weights_per_channel)) + (Y * KERNEL_SIZE) + X;
108  -- Integer value of the current weight
109  constant c_weight_value_NYX : integer := KERNEL_VALUES(N, (Y * KERNEL_SIZE) + X);
110  -- Number of summands generated from this weight (number of unique power of two factors that it can be split into)
111  constant c_result_element_count_NYX : integer := c_elements_per_weight(c_current_index_NYX);
112  -- Array of unique powers of two this weight is expressed as
113  constant c_weight_shift_factors_NYX : t_integer_array(0 to c_result_element_count_NYX - 1)
114  := f_get_weight_shift_factors(c_weight_value_NYX);
115  -- Start index of the first summand generated from this weight in the output array
116  constant c_result_element_index_NYX : integer
117  := f_get_int_array_part_sum(c_elements_per_weight, 0, c_current_index_NYX - 1);
118  -- Input pixel from the image for this weight
119  signal input_value_NYX : std_logic_vector(DIN_WIDTH - 1 downto 0) := (others => '0');
120 
121  begin
122  -- Retrieve the pixel from the input window
123  input_value_NYX <= f_get_vector_of_generic_window(crt_window_N, X, Y);
124 
125  -- For all power of two factors of this weight
126  gen_shifted_values_NYX : for Z in 0 to c_result_element_count_NYX - 1 generate
127 
128  -- Index in the output array for the summand generated from this factor
129  constant c_shifted_value_index_NYXZ : integer := c_result_element_index_NYX + Z;
130  -- Integer value of the current factor
131  constant c_shift_factor_NYXZ : integer := c_weight_shift_factors_NYX(Z);
132  -- Expected maximum bit width of the generated summand
133  constant c_shifted_value_width_NYXZ : integer := SUMMAND_BIT_WIDTHS(c_shifted_value_index_NYXZ);
134  -- Bits to shift
135  constant c_shift_bits_NYXZ : natural := log2_ceil_zero(abs(c_shift_factor_NYXZ));
136 
137  -- Result value (summand)
138  signal shifted_value_NYXZ : signed(c_shifted_value_width_NYXZ - 1 downto 0) := (others => '0');
139  signal shifted_value_resized_NYXZ : signed(LARGEST_SUMMAND_BIT_WIDTH - 1 downto 0);
140 
141  begin
142  -- Multiply the input value with the current weight factor (guaranteed powers of two +/-)
143  shifted_value_NYXZ <= to_signed(
144  to_integer(signed(input_value_NYX)) * c_shift_factor_NYXZ, c_shifted_value_width_NYXZ
145  );
146  gen_assign_unused : if c_shifted_value_width_NYXZ < LARGEST_SUMMAND_BIT_WIDTH generate
147  shifted_value_resized_NYXZ(LARGEST_SUMMAND_BIT_WIDTH - 1 downto c_shifted_value_width_NYXZ) <= (others => shifted_value_NYXZ(c_shifted_value_width_NYXZ - 1));
148  end generate;
149  shifted_value_resized_NYXZ(c_shifted_value_width_NYXZ - 1 downto 0) <= shifted_value_NYXZ;
150  unsorted_elements(c_shifted_value_index_NYXZ) <= shifted_value_resized_NYXZ;
151 
152  end generate gen_shifted_values_NYX;
153  end generate gen_produce_summands_NY;
154  end generate gen_produce_summands_N;
155  end generate gen_produce_summands;
156 
157  -- !No register stage!
158  p_data_out : process(unsorted_elements) is
159  begin
160  for N in 0 to SUMMAND_COUNT - 1 loop
161  data_out((N + 1) * LARGEST_SUMMAND_BIT_WIDTH - 1 downto N * LARGEST_SUMMAND_BIT_WIDTH)
162  <= std_logic_vector(unsorted_elements(N));
163  end loop;
164  end process;
165 
166 end architecture;

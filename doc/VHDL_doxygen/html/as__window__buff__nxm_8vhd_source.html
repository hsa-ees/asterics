80 architecture RTL of as_window_buff_nxm is
81  type word_array is array(0 to WINDOW_Y - 1)
82  of std_logic_vector(DATA_WIDTH - 1 downto 0);
83  type t_generic_line_array is array(natural range<>)
84  of t_generic_line(0 to WINDOW_X - 1, DATA_WIDTH - 1 downto 0);
85 
86  -- Array for window area (lines)
87  signal window_line_array : t_generic_line_array(0 to WINDOW_Y - 1) := (others => (others => (others => '0')));
88  -- Array for connecting the window line buffer in and outputs
89  signal buffer_in_array : word_array := (others => (others => '0'));
90 
91 begin
92 
93  buffer_in_array(0) <= data_in;
94  data_out <= f_get_vector_of_generic_line(window_line_array(WINDOW_Y - 1), WINDOW_X - 1);
95 
96  -- Map the window line array from the pipeline line buffers to the output window
97  window_output : process(window_line_array) is
98  begin
99  for Y in window_out'range(1) loop
100  for X in window_out'range(2) loop
101  for N in window_out'range(3) loop
102  window_out(X, Y, N) <= window_line_array(Y)(X, N);
103  end loop;
104  end loop;
105  end loop;
106  end process;
107 
108  -- Line buffers:
109  buffer_lines_generator : if WINDOW_Y > 1 generate
110  buffer_lines_generator_loop : for N in 0 to WINDOW_Y - 2 generate
111  buffer_line_N : entity as_pipeline_row
112  generic map(
113  DATA_WIDTH => DATA_WIDTH,
114  WINDOW_WIDTH => WINDOW_X,
115  -- Plus 1 as pipeline row expects a register between data_out and buff_in!
116  LINE_WIDTH => LINE_WIDTH + 1 ,
117  MINIMUM_LENGTH_FOR_BRAM => MINIMUM_LENGTH_FOR_BRAM
118  )
119  port map(
120  clk => clk,
121  reset => reset,
122  strobe => strobe,
123  buff_in => buffer_in_array(N),
124  line_out => window_line_array(N),
125  data_out => buffer_in_array(N + 1)
126  );
127  end generate;
128  end generate;
129 
130  -- Last line of window, implemented as flipflops
131  shift_last_line : process(clk) is
132  variable temp_shift : std_logic_vector(DATA_WIDTH - 1 downto 0);
133  begin
134  if rising_edge(clk) then
135  if reset = '1' then
136  temp_shift := (others => '0');
137  for N in 0 to WINDOW_X - 1 loop
138  f_set_vector_of_generic_line(window_line_array(WINDOW_Y - 1), N, temp_shift);
139  end loop;
140  else
141  if strobe = '1' then
142  f_set_vector_of_generic_line(window_line_array(WINDOW_Y - 1), 0, buffer_in_array(WINDOW_Y - 1));
143  for N in 1 to WINDOW_X - 1 loop
144  temp_shift := f_get_vector_of_generic_line(window_line_array(WINDOW_Y - 1), N - 1);
145  f_set_vector_of_generic_line(window_line_array(WINDOW_Y - 1), N, temp_shift);
146  end loop;
147  end if;
148  end if;
149  end if;
150  end process;

156 -- Implement axi_awready generation
157 -- axi_awready is asserted for one S_AXI_ACLK clock cycle when both
158 -- S_AXI_AWVALID and S_AXI_WVALID are asserted. axi_awready is
159 -- de-asserted when reset is low.
160 
161 process (S_AXI_ACLK)
162 begin
163  if rising_edge(S_AXI_ACLK) then
164  if S_AXI_ARESETN = '0' then
165  axi_awready <= '0';
166  else
167  if (axi_awready = '0' and S_AXI_AWVALID = '1' and S_AXI_WVALID = '1') then
168  -- slave is ready to accept write address when
169  -- there is a valid write address and write data
170  -- on the write address and data bus. This design
171  -- expects no outstanding transactions.
172  axi_awready <= '1';
173  else
174  axi_awready <= '0';
175  end if;
176  end if;
177  end if;
178 end process;

180 -- Implement axi_awaddr latching
181 -- This process is used to latch the address when both
182 -- S_AXI_AWVALID and S_AXI_WVALID are valid.
183 
184 process (S_AXI_ACLK)
185 begin
186  if rising_edge(S_AXI_ACLK) then
187  if S_AXI_ARESETN = '0' then
188  axi_awaddr <= (others => '0');
189  else
190  if (axi_awready = '0' and S_AXI_AWVALID = '1' and S_AXI_WVALID = '1') then
191  -- Write Address latching
192  axi_awaddr <= S_AXI_AWADDR;
193  end if;
194  end if;
195  end if;
196 end process;

227 -- Slave register write enable is asserted when valid address and data are available
228 -- and the slave is ready to accept the write address and write data.

232 variable loc_addr :std_logic_vector(OPT_MEM_ADDR_BITS downto 0);
233 begin
234  if rising_edge(S_AXI_ACLK) then
235  if S_AXI_ARESETN = '0' then
236  slv_reg0 <= (others => '0');
237  --~ slv_reg1 <= (others => '0');
238  slv_reg2 <= (others => '0');

241  slv_reg5 <= (others => '0');
242  slv_reg6 <= (others => '0');

248  when b"000" =>

251  -- Respective byte enables are asserted as per write strobes
252  -- slave registor 0
253  slv_reg0(byte_index*8+7 downto byte_index*8) <= S_AXI_WDATA(byte_index*8+7 downto byte_index*8);
254  end if;
255  end loop;
256  --~ when b"001" =>
257  --~ for byte_index in 0 to (C_S_AXI_DATA_WIDTH/8-1) loop
258  --~ if ( S_AXI_WSTRB(byte_index) = '1' ) then
259  --~ -- Respective byte enables are asserted as per write strobes
260  --~ -- slave registor 1
261  --~ slv_reg1(byte_index*8+7 downto byte_index*8) <= S_AXI_WDATA(byte_index*8+7 downto byte_index*8);
262  --~ end if;
263  --~ end loop;
264  when b"010" =>
265  for byte_index in 0 to (C_S_AXI_DATA_WIDTH/8-1) loop
266  if ( S_AXI_WSTRB(byte_index) = '1' ) then
267  -- Respective byte enables are asserted as per write strobes
268  -- slave registor 2
269  slv_reg2(byte_index*8+7 downto byte_index*8) <= S_AXI_WDATA(byte_index*8+7 downto byte_index*8);

271  end loop;
272  when b"011" =>
273  for byte_index in 0 to (C_S_AXI_DATA_WIDTH/8-1) loop
274  if ( S_AXI_WSTRB(byte_index) = '1' ) then
275  -- Respective byte enables are asserted as per write strobes
276  -- slave registor 3

278  end if;

292  -- slave registor 5

327 -- Implement write response logic generation
328 -- The write response and response valid signals are asserted by the slave
329 -- when axi_wready, S_AXI_WVALID, axi_wready and S_AXI_WVALID are asserted.
330 -- This marks the acceptance of address and indicates the status of
331 -- write transaction.
332 
333 process (S_AXI_ACLK)
334 begin
335  if rising_edge(S_AXI_ACLK) then
336  if S_AXI_ARESETN = '0' then
337  axi_bvalid <= '0';
338  axi_bresp <= "00"; --need to work more on the responses
339  else
340  if (axi_awready = '1' and S_AXI_AWVALID = '1' and axi_wready = '1' and S_AXI_WVALID = '1' and axi_bvalid = '0' ) then
341  axi_bvalid <= '1';
342  axi_bresp <= "00";
343  elsif (S_AXI_BREADY = '1' and axi_bvalid = '1') then --check if bready is asserted while bvalid is high)
344  axi_bvalid <= '0'; -- (there is a possibility that bready is always asserted high)
345  end if;
346  end if;
347  end if;
348 end process;

350 -- Implement axi_arready generation
351 -- axi_arready is asserted for one S_AXI_ACLK clock cycle when
352 -- S_AXI_ARVALID is asserted. axi_awready is
353 -- de-asserted when reset (active low) is asserted.
354 -- The read address is also latched when S_AXI_ARVALID is
355 -- asserted. axi_araddr is reset to zero on reset assertion.
356 
357 process (S_AXI_ACLK)
358 begin
359  if rising_edge(S_AXI_ACLK) then
360  if S_AXI_ARESETN = '0' then
361  axi_arready <= '0';
362  axi_araddr <= (others => '1');
363  else
364  if (axi_arready = '0' and S_AXI_ARVALID = '1') then
365  -- indicates that the slave has acceped the valid read address
366  axi_arready <= '1';
367  -- Read Address latching
368  axi_araddr <= S_AXI_ARADDR;
369  else
370  axi_arready <= '0';
371  end if;
372  end if;
373  end if;
374 end process;
375 
376 -- Implement axi_arvalid generation
377 -- axi_rvalid is asserted for one S_AXI_ACLK clock cycle when both
378 -- S_AXI_ARVALID and axi_arready are asserted. The slave registers
379 -- data are available on the axi_rdata bus at this instance. The
380 -- assertion of axi_rvalid marks the validity of read data on the
381 -- bus and axi_rresp indicates the status of read transaction.axi_rvalid
382 -- is deasserted on reset (active low). axi_rresp and axi_rdata are
383 -- cleared to zero on reset (active low).
384 process (S_AXI_ACLK)
385 begin
386  if rising_edge(S_AXI_ACLK) then
387  if S_AXI_ARESETN = '0' then
388  axi_rvalid <= '0';
389  axi_rresp <= "00";
390  else
391  if (axi_arready = '1' and S_AXI_ARVALID = '1' and axi_rvalid = '0') then
392  -- Valid read data is available at the read data bus
393  axi_rvalid <= '1';
394  axi_rresp <= "00"; -- 'OKAY' response
395  elsif (axi_rvalid = '1' and S_AXI_RREADY = '1') then
396  -- Read data is accepted by the master
397  axi_rvalid <= '0';
398  end if;
399  end if;
400  end if;
401 end process;
402 
403 -- Implement memory mapped register select and read logic generation
404 -- Slave register read enable is asserted when valid address is available
405 -- and the slave is ready to accept the read address.
406 slv_reg_rden <= axi_arready and S_AXI_ARVALID and (not axi_rvalid) ;

408 process (slv_reg0, slv_reg1, slv_reg2, slv_reg3, slv_reg4, slv_reg5, slv_reg6, slv_reg7, axi_araddr, S_AXI_ARESETN, slv_reg_rden)
409 variable loc_addr :std_logic_vector(OPT_MEM_ADDR_BITS downto 0);
410 begin
411  -- Address decoding for reading registers
412  loc_addr := axi_araddr(ADDR_LSB + OPT_MEM_ADDR_BITS downto ADDR_LSB);
413  case loc_addr is
414  when b"000" =>
415  reg_data_out <= slv_reg0;
416  when b"001" =>
417  reg_data_out <= slv_reg1;
418  when b"010" =>
419  reg_data_out <= slv_reg2;
420  when b"011" =>
421  reg_data_out <= slv_reg3;
422  when b"100" =>
423  reg_data_out <= slv_reg4;
424  when b"101" =>
425  reg_data_out <= slv_reg5;
426  when b"110" =>
427  reg_data_out <= slv_reg6;
428  when b"111" =>
429  reg_data_out <= slv_reg7;
430  when others =>
431  reg_data_out <= (others => '0');
432  end case;
433 end process;
434 
435 -- Output register or memory read data
436 process( S_AXI_ACLK ) is
437 begin
438  if (rising_edge (S_AXI_ACLK)) then
439  if ( S_AXI_ARESETN = '0' ) then
440  axi_rdata <= (others => '0');
441  else
442  if (slv_reg_rden = '1') then
443  -- When there is a valid read address (S_AXI_ARVALID) with
444  -- acceptance of read address by the slave (axi_arready),
445  -- output the read dada
446  -- Read address mux
447  axi_rdata <= reg_data_out; -- register read data
448  end if;
449  end if;
450  end if;
451 end process;

454 -- Add user logic here
455 slv_reg_0 <= slv_reg0;
456 slv_reg1 <= slv_reg_1; -- status output (HW to SW)
457 slv_reg_2 <= slv_reg2;
458 slv_reg_3 <= slv_reg3;
459 slv_reg_4 <= slv_reg4;
460 slv_reg_5 <= slv_reg5;
461 slv_reg_6 <= slv_reg6;
462 slv_reg_7 <= slv_reg7;
463 -- User logic ends

465 end arch_imp;

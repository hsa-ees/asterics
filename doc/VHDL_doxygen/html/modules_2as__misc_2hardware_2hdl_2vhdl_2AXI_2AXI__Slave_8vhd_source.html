189 process (S_AXI_ACLK)
190 begin
191  if rising_edge(S_AXI_ACLK) then
192  if S_AXI_ARESETN = '0' then
193  axi_awready <= '0';
194  else
195  if (axi_awready = '0' and S_AXI_AWVALID = '1' and S_AXI_WVALID = '1') then
196  -- slave is ready to accept write address when
197  -- there is a valid write address and write data
198  -- on the write address and data bus. This design
199  -- expects no outstanding transactions.
200  axi_awready <= '1';
201  else
202  axi_awready <= '0';
203  end if;
204  end if;
205  end if;
206 end process;

211 process (S_AXI_ACLK)
212 begin
213  if rising_edge(S_AXI_ACLK) then
214  if S_AXI_ARESETN = '0' then
215  axi_awaddr <= (others => '0');
216  else
217  if (axi_awready = '0' and S_AXI_AWVALID = '1' and S_AXI_WVALID = '1') then
218  -- Write Address latching
219  axi_awaddr <= S_AXI_AWADDR;
220  end if;
221  end if;
222  end if;
223 end process;

229 process (S_AXI_ACLK)
230 begin

233  axi_wready <= '0';
234  else
235  if (axi_wready = '0' and S_AXI_WVALID = '1' and S_AXI_AWVALID = '1') then
236  -- slave is ready to accept write data when
237  -- there is a valid write address and write data
238  -- on the write address and data bus. This design
239  -- expects no outstanding transactions.
240  axi_wready <= '1';
241  else
242  axi_wready <= '0';
243  end if;
244  end if;
245  end if;
246 end process;

255 slv_reg_wren <= axi_wready and S_AXI_WVALID and axi_awready and S_AXI_AWVALID ;
256 
262 process (S_AXI_ACLK)
263 begin
264  if rising_edge(S_AXI_ACLK) then
265  if S_AXI_ARESETN = '0' then
266  axi_bvalid <= '0';
267  axi_bresp <= "00"; --need to work more on the responses
268  else
269  if (axi_awready = '1' and S_AXI_AWVALID = '1' and axi_wready = '1' and S_AXI_WVALID = '1' and axi_bvalid = '0' ) then
270  axi_bvalid <= '1';
271  axi_bresp <= "00";
272  elsif (S_AXI_BREADY = '1' and axi_bvalid = '1') then --check if bready is asserted while bvalid is high)
273  axi_bvalid <= '0'; -- (there is a possibility that bready is always asserted high)
274  end if;
275  end if;
276  end if;
277 end process;

285 process (S_AXI_ACLK)
286 begin
287  if rising_edge(S_AXI_ACLK) then
288  if S_AXI_ARESETN = '0' then
289  axi_arready <= '0';
290  axi_araddr <= (others => '1');
291  else
292  if (axi_arready = '0' and S_AXI_ARVALID = '1') then
293  -- indicates that the slave has acceped the valid read address
294  axi_arready <= '1';
295  -- Read Address latching
296  axi_araddr <= S_AXI_ARADDR;
297  else
298  axi_arready <= '0';
299  end if;
300  end if;
301  end if;
302 end process;

312 process (S_AXI_ACLK)
313 begin
314  if rising_edge(S_AXI_ACLK) then
315  if S_AXI_ARESETN = '0' then
316  axi_rvalid <= '0';
317  axi_rresp <= "00";
318  else
319  if (axi_arready = '1' and S_AXI_ARVALID = '1' and axi_rvalid = '0') then
320  -- Valid read data is available at the read data bus
321  axi_rvalid <= '1';
322  axi_rresp <= "00"; -- 'OKAY' response
323  elsif (axi_rvalid = '1' and S_AXI_RREADY = '1') then
324  -- Read data is accepted by the master
325  axi_rvalid <= '0';
326  end if;
327  end if;
328  end if;
329 end process;

331 -- Implement memory mapped register select and read logic generation
332 -- Slave register read enable is asserted when valid address is available
333 -- and the slave is ready to accept the read address.
334 slv_reg_rden <= axi_arready and S_AXI_ARVALID and (not axi_rvalid) ;
335 
336 -- Output register or memory read data
337 process( S_AXI_ACLK ) is
338 begin
339  if (rising_edge (S_AXI_ACLK)) then
340  if ( S_AXI_ARESETN = '0' ) then
341  axi_rdata <= (others => '0');
342  else
343  if (slv_reg_rden = '1') then
344  -- When there is a valid read address (S_AXI_ARVALID) with
345  -- acceptance of read address by the slave (axi_arready),
346  -- output the read dada
347  -- Read address mux
348  axi_rdata <= axi_slv_reg_read_data; -- register read data
349  end if;
350  end if;
351  end if;
352 end process;

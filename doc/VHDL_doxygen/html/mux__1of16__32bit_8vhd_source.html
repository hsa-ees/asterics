50 entity mux_1of16_32bit is
51  port(
52  data_sel : in std_logic_vector( 3 downto 0); -- switch for choosing one of the 16 pixels
53  data_in : in std_logic_vector(31 downto 0); -- one address (16 pixels) data abut on input
54  data_out : out std_logic_vector( 1 downto 0) -- one pixel (2 bit) data output
55  );
56 end mux_1of16_32bit;

61 begin

63 process (data_sel, data_in)
64 begin
65  case data_sel is
66  when "1111" => data_out(1 downto 0) <= data_in( 1 downto 0);
67  when "1110" => data_out(1 downto 0) <= data_in( 3 downto 2);
68  when "1101" => data_out(1 downto 0) <= data_in( 5 downto 4);
69  when "1100" => data_out(1 downto 0) <= data_in( 7 downto 6);
70  when "1011" => data_out(1 downto 0) <= data_in( 9 downto 8);
71  when "1010" => data_out(1 downto 0) <= data_in(11 downto 10);
72  when "1001" => data_out(1 downto 0) <= data_in(13 downto 12);
73  when "1000" => data_out(1 downto 0) <= data_in(15 downto 14);
74  when "0111" => data_out(1 downto 0) <= data_in(17 downto 16);
75  when "0110" => data_out(1 downto 0) <= data_in(19 downto 18);
76  when "0101" => data_out(1 downto 0) <= data_in(21 downto 20);
77  when "0100" => data_out(1 downto 0) <= data_in(23 downto 22);
78  when "0011" => data_out(1 downto 0) <= data_in(25 downto 24);
79  when "0010" => data_out(1 downto 0) <= data_in(27 downto 26);
80  when "0001" => data_out(1 downto 0) <= data_in(29 downto 28);
81  when "0000" => data_out(1 downto 0) <= data_in(31 downto 30);
82  when others => null;
83  end case;
84 end process;

86 end RTL;

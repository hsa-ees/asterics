41 -- Purpose:
42 -- This module connects to a top level DVI 1.0 source interface comprised of three
43 -- TMDS data channels and one TMDS clock channel. It includes the necessary
44 -- clock infrastructure (optional), encoding and serialization logic.
45 -- On the input side it has 24-bit RGB video data bus, pixel clock and synchronization
46 -- signals.
47 --

63  Generic (

186 -- Swap each bit
187 D0_direct: if kD0Swap = false generate
188  pDataOutRaw_q(0) <= pDataOutRaw(0);
189 end generate;
190 D0_reverse: if kD0Swap = true generate
191  pDataOutRaw_q(0) <= not pDataOutRaw(0);
192 end generate;
193 D1_direct: if kD1Swap = false generate
194  pDataOutRaw_q(1) <= pDataOutRaw(1);
195 end generate;
196 D1_reverse: if kD1Swap = true generate
197  pDataOutRaw_q(1) <= not pDataOutRaw(1);
198 end generate;
199 D2_direct: if kD2Swap = false generate
200  pDataOutRaw_q(2) <= pDataOutRaw(2);
201 end generate;
202 D2_reverse: if kD2Swap = true generate
203  pDataOutRaw_q(2) <= not pDataOutRaw(2);
204 end generate;
205 Clk_direct: if kClkSwap = false generate
206  pClkOut <= "1111100000";
207 end generate;
208 Clk_reverse: if kClkSwap = true generate
209  pClkOut <= "0000011111";
210 end generate;
211 
212 -- DVI Output conform DVI 1.0
213 -- except that it sends blank pixel during blanking
214 -- for some reason vid_data is packed in RBG order
215 pDataOut(2) <= vid_pData(23 downto 16); -- red is channel 2
216 pDataOut(1) <= vid_pData(7 downto 0); -- green is channel 1
217 pDataOut(0) <= vid_pData(15 downto 8); -- blue is channel 0
218 pC0(2 downto 1) <= (others => '0'); -- default is low for control signals
219 pC1(2 downto 1) <= (others => '0'); -- default is low for control signals
220 pC0(0) <= vid_pHSync; -- channel 0 carries control signals too
221 pC1(0) <= vid_pVSync; -- channel 0 carries control signals too
222 pVde <= vid_pVDE & vid_pVDE & vid_pVDE; -- all of them are either active or blanking at once
223 
224 end Behavioral;
